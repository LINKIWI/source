    159 // DownloadFileByName implements the b2_download_file_by_name API.
    160 // Reference: https://www.backblaze.com/b2/docs/b2_download_file_by_name.html
    161 func (c *Client) DownloadFileByName(ctx context.Context, request *DownloadFileByNameRequest) (*DownloadFileByNameResponse, error) {
    162 	r, err := c.createAPIRequest(ctx, request.DownloadURL, "", http.MethodGet, request)
    163 	if err != nil {
    164 		return nil, err
    165 	}
    166 
    167 	ref := &url.URL{Path: path.Join("file", request.Bucket, request.File)}
    168 	r.URL = r.URL.ResolveReference(ref)
    169 
    170 	if request.Range.Start > 0 && request.Range.End > 0 {
    171 		r.Header.Set(
    172 			"Range",
    173 			fmt.Sprintf("%s=%d-%d", request.Range.Unit, request.Range.Start, request.Range.End),
    174 		)
    175 	} else if request.Range.Start > 0 {
    176 		r.Header.Set(
    177 			"Range",
    178 			fmt.Sprintf("%s=%d-", request.Range.Unit, request.Range.Start),
    179 		)
    180 	} else if request.Range.End > 0 {
    181 		r.Header.Set(
    182 			"Range",
    183 			fmt.Sprintf("%s=0-%d", request.Range.Unit, request.Range.End),
    184 		)
    185 	}
    186 
    187 	// Note that the response body is not closed as this layer since it is passed directly to
    188 	// the client as an io.ReadCloser; it is the responsibility of the caller to close the
    189 	// body when it has been consumed.
    190 	response, err := c.doRaw(r)
    191 	if err != nil {
    192 		return nil, err
    193 	}
    194 
    195 	size, err := strconv.Atoi(response.Header.Get("Content-Length"))
    196 	if err != nil {
    197 		return nil, statusFromError(err)
    198 	}
    199 
    200 	contentRange := Range{}
    201 	contentRangeRe := regexp.MustCompile("^(?P<unit>[^\\s]+)\\s(?P<start>\\d+)-(?P<end>\\d+)/(?P<total>\\d+)$")
    202 	matches := contentRangeRe.FindStringSubmatch(response.Header.Get("Content-Range"))
    203 	if len(matches) == 5 {
    204 		start, err := strconv.Atoi(matches[contentRangeRe.SubexpIndex("start")])
    205 		if err != nil {
    206 			return nil, statusFromError(err)
    207 		}
    208 
    209 		end, err := strconv.Atoi(matches[contentRangeRe.SubexpIndex("end")])
    210 		if err != nil {
    211 			return nil, statusFromError(err)
    212 		}
    213 
    214 		total, err := strconv.Atoi(matches[contentRangeRe.SubexpIndex("total")])
    215 		if err != nil {
    216 			return nil, statusFromError(err)
    217 		}
    218 
    219 		contentRange.Unit = matches[contentRangeRe.SubexpIndex("unit")]
    220 		contentRange.Start = start
    221 		contentRange.End = end
    222 		contentRange.Total = total
    223 	}
    224 
    225 	fileInfo := make(map[string]string)
    226 	for header := range response.Header {
    227 		if strings.HasPrefix(header, "X-Bz-Info-") {
    228 			fileInfo[strings.ToLower(header[len("X-Bz-Info-"):])] = response.Header.Get(header)
    229 		}
    230 	}
    231 
    232 	var uploadTimestamp UnixTimestamp
    233 	if err := json.Unmarshal([]byte(response.Header.Get("X-Bz-Upload-Timestamp")), &uploadTimestamp); err != nil {
    234 		return nil, statusFromError(err)
    235 	}
    236 
    237 	return &DownloadFileByNameResponse{
    238 		FileID:          response.Header.Get("X-Bz-File-Id"),
    239 		FileName:        response.Header.Get("X-Bz-File-Name"),
    240 		FileInfo:        fileInfo,
    241 		ContentSHA1:     response.Header.Get("X-Bz-Content-Sha1"),
    242 		Size:            size,
    243 		ContentType:     response.Header.Get("Content-Type"),
    244 		Range:           contentRange,
    245 		UploadTimestamp: &uploadTimestamp,
    246 		Data:            response.Body,
    247 	}, nil
    248 }

    401 // createAPIRequest creates a http.Request object, obeying B2 conventions, based on the specified
    402 // base URL, endpoint, HTTP method, and request struct.
    403 func (c *Client) createAPIRequest(ctx context.Context, baseURL *url.URL, endpoint string, method string, request interface{}) (*http.Request, error) {
    404 	var body io.ReadCloser
    405 
    406 	u := baseURL
    407 	if endpoint != "" {
    408 		u = baseURL.ResolveReference(&url.URL{Path: path.Join(APIV2BasePath, endpoint)})
    409 	}
    410 
    411 	headers, err := marshalHeaders(request)
    412 	if err != nil {
    413 		return nil, statusFromError(err)
    414 	}
    415 
    416 	headers.Set("User-Agent", c.identity)
    417 
    418 	if method == http.MethodPost {
    419 		payload, err := json.Marshal(request)
    420 		if err != nil {
    421 			return nil, statusFromError(err)
    422 		}
    423 
    424 		body = io.NopCloser(bytes.NewReader(payload))
    425 	}
    426 
    427 	r := &http.Request{
    428 		Method: method,
    429 		URL:    u,
    430 		Header: headers,
    431 		Body:   body,
    432 	}
    433 
    434 	return r.WithContext(ctx), nil
    435 }

    437 // doAPI executes the request as a canonical API request, whose response body can be parsed as JSON
    438 // into the provided container.
    439 func (c *Client) doAPI(request *http.Request, data interface{}) (*http.Response, error) {
    440 	response, err := c.doRaw(request)
    441 	if err != nil {
    442 		return nil, statusFromError(err)
    443 	}
    444 
    445 	defer response.Body.Close()
    446 
    447 	body, err := io.ReadAll(response.Body)
    448 	if err != nil {
    449 		return response, statusFromError(err)
    450 	}
    451 
    452 	if data != nil {
    453 		if err := json.Unmarshal(body, &data); err != nil {
    454 			return response, statusFromError(err)
    455 		}
    456 	}
    457 
    458 	return response, nil
    459 }

    461 // doRaw executes the request, handling HTTP errors semantically as API errors, and returns the
    462 // response as-is without any response body manipulation.
    463 func (c *Client) doRaw(request *http.Request) (*http.Response, error) {
    464 	var status Status
    465 
    466 	c.log.Debug(
    467 		"dispatching HTTP request",
    468 		zap.String("method", request.Method),
    469 		zap.Any("headers", request.Header),
    470 		zap.Stringer("url", request.URL),
    471 	)
    472 
    473 	response, err := c.http.Do(request)
    474 	if err != nil {
    475 		c.log.Error(
    476 			"fatal transport error during HTTP dispatch",
    477 			zap.String("method", request.Method),
    478 			zap.Stringer("url", request.URL),
    479 			zap.Error(err),
    480 		)
    481 
    482 		return nil, statusFromError(err)
    483 	}
    484 
    485 	if response.StatusCode >= http.StatusBadRequest {
    486 		body, err := io.ReadAll(response.Body)
    487 		if err != nil {
    488 			return response, statusFromError(err)
    489 		}
    490 
    491 		if err := json.Unmarshal(body, &status); err != nil {
    492 			return response, statusFromError(err)
    493 		}
    494 
    495 		response.Body.Close()
    496 
    497 		c.log.Debug(
    498 			"received application-level error from B2 API",
    499 			zap.Int("status", status.Status),
    500 			zap.String("code", status.Code),
    501 			zap.String("message", status.Message),
    502 		)
    503 
    504 		return response, &status
    505 	}
    506 
    507 	c.log.Debug(
    508 		"successful B2 API response",
    509 		zap.String("method", response.Request.Method),
    510 		zap.Stringer("url", response.Request.URL),
    511 		zap.Int("status", response.StatusCode),
    512 		zap.Int64("size", response.ContentLength),
    513 	)
    514 
    515 	return response, nil
