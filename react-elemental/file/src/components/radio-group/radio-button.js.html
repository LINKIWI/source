     11 class RadioButton extends Component {

     23       PropTypes.string,

     29     isHover: PropTypes.bool.isRequired,

     33     handleFocus: PropTypes.func.isRequired,

     37   componentDidUpdate(prevProps, prevState, { shouldFocus }) {
     38     if (shouldFocus && this.button.current) {

     41   }
