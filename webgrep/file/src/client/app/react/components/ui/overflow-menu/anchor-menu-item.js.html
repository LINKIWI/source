      8 /**
      9  * Single overflow menu item acting as a native anchor element.
     10  */
     11 const AnchorMenuItem = ({
     12   label,
     13   icon,
     14   href,
     15   onClick,
     16   handleMouseEnter,
     17   handleMouseLeave,
     18   handleMouseDown,
     19   handleMouseUp,
     20   isHover,
     21   isClicked,
     22 }) => {
     23   const baseStyle = {
     24     backgroundColor: 'white',
     25     border: 0,
     26     cursor: 'pointer',
     27     textAlign: 'left',
     28     transition: transition.all.DEFAULT,
     29     width: '100%',
     30   };
     31 
     32   const hoverStyle = {
     33     backgroundColor: background.primary.EPSILON,
     34   };
     35 
     36   const clickStyle = {
     37     backgroundColor: text.dark.GAMMA,
     38   };
     39 
     40   const anchorStyle = {
     41     alignItems: 'center',
     42     color: 'currentColor',
     43     display: 'flex',
     44     padding: '10px 16px',
     45     textDecoration: 'none',
     46     width: '100%',
     47   };
     48 
     49   return (
     50     <Text
     51       color={text.dark.ALPHA}
     52       size="kilo"
     53       onMouseEnter={handleMouseEnter}
     54       onMouseLeave={handleMouseLeave}
     55       onMouseDown={handleMouseDown}
     56       onMouseUp={handleMouseUp}
     57       style={{
     58         ...baseStyle,
     59         ...isHover && hoverStyle,
     60         ...isHover && isClicked && clickStyle,
     61       }}
     62     >
     63       <a href={href} onClick={onClick} style={anchorStyle}>
     64         {icon && (
     65           <Spacing size="small" style={{ display: 'inherit' }} right inline>
     66             <Text color={text.dark.BETA} size="kilo" inline>
     67               {icon}
     68             </Text>
     69           </Spacing>
     70         )}
     71         {label}
     72       </a>
     73     </Text>
     74   );
     75 };
     76 
     77 AnchorMenuItem.propTypes = {
     78   label: PropTypes.string.isRequired,
     79   icon: PropTypes.node,
     80   href: PropTypes.string.isRequired,
     81   onClick: PropTypes.func,
     82   // HOC props
     83   handleMouseEnter: PropTypes.func.isRequired,
     84   handleMouseLeave: PropTypes.func.isRequired,
     85   handleMouseDown: PropTypes.func.isRequired,
     86   handleMouseUp: PropTypes.func.isRequired,
     87   isHover: PropTypes.bool.isRequired,
     88   isClicked: PropTypes.bool.isRequired,
     89 };
     90 
     91 AnchorMenuItem.defaultProps = {
     92   onClick: () => {},
     93   icon: null,
     94 };
     95 
     96 export default compose(
     97   withToggleState({
     98     key: 'isHover',
     99     enable: 'handleMouseEnter',
    100     disable: 'handleMouseLeave',
    101   }),
    102   withToggleState({
    103     key: 'isClicked',
    104     enable: 'handleMouseDown',
    105     disable: 'handleMouseUp',
    106   }),
    107 )(AnchorMenuItem);
