     30       window.removeEventListener('hashchange', this.handleHashChange);

     38       window.location.hash = serializer(value);
     39     }

     41     _handleHashChange() {
     42       const hash = window.location.hash ? deserializer(window.location.hash) : null;

     44     }

     59     }
     60   }
