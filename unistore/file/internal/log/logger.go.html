     21 	defer mutex.Unlock()
     22 
     23 	level := zap.NewAtomicLevel()

     26 	}

     30 	encoder.FunctionKey = "func"
     31 
     32 	logCfg := zap.Config{
     33 		Level:            level,
     34 		Encoding:         "json",

     36 		OutputPaths:      cfg.Outputs,
     37 		ErrorOutputPaths: cfg.Outputs,

     39 	}
     40 
     41 	logger, err := logCfg.Build()

     43 		return fmt.Errorf("log: error building logging configuration: err=%v", err)
     44 	}
     45 
     46 	zap.ReplaceGlobals(logger)

     48 	return nil
