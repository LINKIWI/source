     11 import Slider from 'client/app/react/components/ui/slider';
     12 import { setPreference } from 'client/app/redux/actions/preferences';

     15 // Upper limit for the number of context lines that can be set with the interactive slider.
     16 const MAX_CONTEXT_LINES = 20;
     17 
     18 // Upper limit for the match limit that can be set with the interactive slider.
     19 const MAX_INITIAL_MATCH_LIMIT = 200;
     20 
     21 // Upper limit for the collapsed file results that can be set with the interactive slider.
     22 const MAX_FILE_RESULTS_LIMIT = 50;
     23 
     24 class AdminPreferencesSearchContainer extends Component {
     25   static propTypes = {
     26     // HOC props
     27     form: PropTypes.shape({
     28       contextSlider: PropTypes.number.isRequired,
     29       matchLimitSlider: PropTypes.number.isRequired,
     30       fileResultsLimitSlider: PropTypes.number.isRequired,
     31     }).isRequired,
     32     handleFormChange: PropTypes.func.isRequired,
     33     preferences: PropTypes.shape({
     34       [PREFERENCE_KEYS.CODE_SEARCH_CONTEXT_LINES]: PropTypes.number.isRequired,
     35     }).isRequired,
     36     actions: PropTypes.shape({
     37       setPreference: PropTypes.func.isRequired,
     38     }).isRequired,
     39   };
     40 
     41   componentDidUpdate(prevProps) {
     42     const { preferences, handleFormChange } = this.props;
     43     const { preferences: prevPreferences } = prevProps;
     44 
     45     const contextLines = preferences[PREFERENCE_KEYS.CODE_SEARCH_CONTEXT_LINES];
     46     const prevContextLines = prevPreferences[PREFERENCE_KEYS.CODE_SEARCH_CONTEXT_LINES];
     47     const matchLimit = preferences[PREFERENCE_KEYS.CODE_SEARCH_INITIAL_MATCH_LIMIT];
     48     const prevMatchLimit = prevPreferences[PREFERENCE_KEYS.CODE_SEARCH_INITIAL_MATCH_LIMIT];
     49     const fileResultsLimit = preferences[PREFERENCE_KEYS.CODE_SEARCH_FILE_RESULTS_LIMIT];
     50     const prevFileResultsLimit = prevPreferences[PREFERENCE_KEYS.CODE_SEARCH_FILE_RESULTS_LIMIT];
     51 
     52     // Redux-induced change in number of slider values; propagate this change as an adjustment to
     53     // slider position
     54     if (prevContextLines !== contextLines) {
     55       handleFormChange('contextSlider')(contextLines / MAX_CONTEXT_LINES);
     56     }
     57     if (matchLimit !== prevMatchLimit) {
     58       handleFormChange('matchLimitSlider')(matchLimit / MAX_INITIAL_MATCH_LIMIT);
     59     }
     60     if (fileResultsLimit !== prevFileResultsLimit) {
     61       handleFormChange('fileResultsLimitSlider')(fileResultsLimit / MAX_FILE_RESULTS_LIMIT);
     62     }
     63   }
     64 
     65   handleChoiceChange = this._handleChoiceChange.bind(this);
     66 
     67   _handleChoiceChange(key) {
     68     return (value) => this.props.actions.setPreference(key, value);
     69   }
     70 
     71   render() {
     72     const {
     73       form: { contextSlider, matchLimitSlider, fileResultsLimitSlider },
     74       handleFormChange,
     75     } = this.props;
     76     const {
     77       CODE_SEARCH_CONTEXT_LINES,
     78       CODE_SEARCH_INITIAL_MATCH_LIMIT,
     79       CODE_SEARCH_FILE_RESULTS_LIMIT,
     80     } = PREFERENCE_KEYS;
     81 
     82     const contextLines = Math.round(contextSlider * MAX_CONTEXT_LINES);
     83     const matchLimit = Math.round(matchLimitSlider * MAX_INITIAL_MATCH_LIMIT);
     84     const fileResultsLimit = Math.round(fileResultsLimitSlider * MAX_FILE_RESULTS_LIMIT);
     85 
     86     return (
     87       <AdminPanel
     88         title="Search"
     89         subtitle="Code search preferences, persisted per-browser"
     90         icon={<MdSearch />}
     91       >
     92         <AdminControl
     93           title="Code context lines"
     94           description="Number of lines of surrounding context for each matching search result"
     95         >
     96           <Slider
     97             value={contextSlider}
     98             annotation={contextLines.toString()}
     99             onChange={handleFormChange('contextSlider')}
    100             onFinalize={() =>
    101               this.handleChoiceChange(CODE_SEARCH_CONTEXT_LINES)(contextLines)}
    102           />
    103         </AdminControl>
    104 
    105         <Spacing top bottom>
    106           <Separator />
    107         </Spacing>
    108 
    109         <AdminControl
    110           title="Default match limit"
    111           description="Initial maximum number of matched lines per search query"
    112         >
    113           <Slider
    114             value={matchLimitSlider}
    115             annotation={matchLimit.toString()}
    116             onChange={handleFormChange('matchLimitSlider')}
    117             onFinalize={() =>
    118               this.handleChoiceChange(CODE_SEARCH_INITIAL_MATCH_LIMIT)(matchLimit)}
    119           />
    120         </AdminControl>
    121 
    122         <Spacing top bottom>
    123           <Separator />
    124         </Spacing>
    125 
    126         <AdminControl
    127           title="Collapsed file results limit"
    128           description="Number of file result matches to display per search query before expansion"
    129         >
    130           <Slider
    131             value={fileResultsLimitSlider}
    132             annotation={fileResultsLimit.toString()}
    133             onChange={handleFormChange('fileResultsLimitSlider')}
    134             onFinalize={() =>
    135               this.handleChoiceChange(CODE_SEARCH_FILE_RESULTS_LIMIT)(fileResultsLimit)}
    136           />
    137         </AdminControl>
    138       </AdminPanel>
    139     );
    140   }
    141 }
    142 
    143 const mapStateToProps = ({ preferences }) => ({ preferences });
    144 
    145 const mapDispatchToProps = (dispatch) => ({
    146   actions: bindActionCreators({ setPreference }, dispatch),
    147 });

    149 export default compose(
    150   connect(mapStateToProps, mapDispatchToProps),
    151   withForm({

    153       contextSlider: preferences[PREFERENCE_KEYS.CODE_SEARCH_CONTEXT_LINES] / MAX_CONTEXT_LINES,
    154       matchLimitSlider:
    155         preferences[PREFERENCE_KEYS.CODE_SEARCH_INITIAL_MATCH_LIMIT] / MAX_INITIAL_MATCH_LIMIT,
    156       fileResultsLimitSlider:
    157         preferences[PREFERENCE_KEYS.CODE_SEARCH_FILE_RESULTS_LIMIT] / MAX_FILE_RESULTS_LIMIT,
    158     }),
    159   }),
    160 )(AdminPreferencesSearchContainer);
