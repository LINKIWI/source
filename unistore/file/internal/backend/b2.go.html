     78 	if cfg.Connection.Identity != "" {
     79 		clientID := fmt.Sprintf(
     80 			"unistore/%s (instance:%s)",
     81 			meta.Version,
     82 			cfg.Connection.Identity,
     83 		)
     84 
     85 		opts = append(opts, b2.WithClientID(clientID))
     86 	}
     87 
     88 	client, err := b2.New(opts...)
     89 	if err != nil {
     90 		return nil, err
     91 	}
     92 
     93 	b := &B2{
     94 		client:           client,
     95 		applicationKeyID: cfg.ApplicationKeyID,
     96 		applicationKey:   cfg.ApplicationKey,
     97 		auth:             new(authenticationState),

    240 	if err != nil {
    241 		return nil, b.createError(err)
    242 	}
    243 
    244 	dfReq := &b2.DownloadFileByNameRequest{
    245 		Authorization: auth.authorizationToken,

    249 	}
    250 
    251 	if request.Range != nil {

    260 			// behavior by subtracting 1 from the desired end index for compatibility.
    261 			dfReq.Range.End = int(request.Range.End) - 1
    262 		}
    263 	}

    266 	if err != nil {
    267 		return nil, b.createError(err)
    268 	}
    269 
    270 	defer dfResp.Data.Close()
    271 
    272 	data, err := io.ReadAll(dfResp.Data)
    273 	if err != nil {
    274 		return nil, b.createError(err)
    275 	}
    276 
    277 	attributes, err := schemas.UnmarshalAttributes(dfResp.FileInfo, nil)
    278 	if err != nil {
    279 		return nil, b.createError(err)
    280 	}

    283 		Metadata: &common.Metadata{
    284 			Id:           dfResp.FileID,

    292 		},

    297 // StreamGetObject is a streaming implementation of GetObject. It segments the requests to B2 in
    298 // chunks of the requested size using the Range request directive, and pipes chunked responses into
    299 // a stream channel.
    300 func (b *B2) StreamGetObject(ctx context.Context, request *service.GetObjectStreamRequest) (<-chan *service.GetObjectStreamResponse, chan error) {
    301 	stream := make(chan *service.GetObjectStreamResponse)
    302 	errs := make(chan error, 1)
    303 
    304 	if request.ChunkSize == 0 {
    305 		errs <- status.Error(
    306 			codes.InvalidArgument,
    307 			"backend: B2 streaming get chunk size must be greater than zero",
    308 		)
    309 		return nil, errs
    310 	}
    311 
    312 	auth, err := b.authorizeAccount(ctx)
    313 	if err != nil {
    314 		errs <- b.createError(err)
    315 		return nil, errs
    316 	}
    317 
    318 	go func() {
    319 		var offset uint64
    320 
    321 		if request.Request.Range != nil {
    322 			offset = request.Request.Range.Start
    323 		}
    324 
    325 		for {
    326 			// Read the requested chunk size, but only up to the end index specified by
    327 			// the range, when available.
    328 			end := offset + request.ChunkSize
    329 			if request.Request.Range != nil &&
    330 				request.Request.Range.End > 0 &&
    331 				end > request.Request.Range.End {
    332 				end = request.Request.Range.End
    333 			}
    334 
    335 			dfReq := &b2.DownloadFileByNameRequest{
    336 				Authorization: auth.authorizationToken,
    337 				DownloadURL:   auth.downloadURL,
    338 				Bucket:        request.Request.Resource.Bucket,
    339 				File:          request.Request.Key,
    340 				// Note that the B2 API considers the range end index to be
    341 				// inclusive. To normalize this behavior against the Unistore API,
    342 				// add/subtract 1 from the end index as needed for non-inclusive
    343 				// end index semantics.
    344 				Range: b2.Range{
    345 					Unit:  "bytes",
    346 					Start: int(offset),
    347 					End:   int(end) - 1,
    348 				},
    349 			}
    350 
    351 			dfResp, err := b.client.DownloadFileByName(ctx, dfReq)
    352 			if err != nil {
    353 				errs <- b.createError(err)
    354 				break
    355 			}
    356 
    357 			chunk, err := io.ReadAll(dfResp.Data)
    358 			if err != nil {
    359 				dfResp.Data.Close()
    360 				errs <- b.createError(err)
    361 				break
    362 			}
    363 
    364 			dfResp.Data.Close()
    365 
    366 			attributes, err := schemas.UnmarshalAttributes(dfResp.FileInfo, nil)
    367 			if err != nil {
    368 				errs <- b.createError(err)
    369 				break
    370 			}
    371 
    372 			stream <- &service.GetObjectStreamResponse{
    373 				Response: &service.GetObjectResponse{
    374 					Metadata: &common.Metadata{
    375 						Id:           dfResp.FileID,
    376 						Key:          dfResp.FileName,
    377 						Size:         uint64(dfResp.Size),
    378 						ObjectType:   common.Object_REGULAR,
    379 						MimeType:     dfResp.ContentType,
    380 						ModifiedTime: timestamppb.New(dfResp.UploadTimestamp.Time),
    381 						Checksum:     dfResp.ContentSHA1,
    382 						Attributes:   attributes,
    383 					},
    384 					Data: chunk,
    385 				},
    386 				Range: &common.Range{
    387 					Unit:  "bytes",
    388 					Start: uint64(dfResp.Range.Start),
    389 					End:   uint64(dfResp.Range.End) + 1,
    390 					Total: uint64(dfResp.Range.Total),
    391 				},
    392 			}
    393 
    394 			offset += uint64(dfResp.Range.End - dfResp.Range.Start + 1)
    395 
    396 			if offset >= uint64(dfResp.Range.Total) {
    397 				break
    398 			}
    399 
    400 			if request.Request.Range != nil &&
    401 				request.Request.Range.End > 0 &&
    402 				offset >= request.Request.Range.End {
    403 				break
    404 			}
    405 		}
    406 
    407 		close(stream)
    408 	}()
    409 
    410 	return stream, errs
    411 }

    413 // PutObject uploads the request file in full.
    414 func (b *B2) PutObject(ctx context.Context, request *service.PutObjectRequest) (*service.PutObjectResponse, error) {
    415 	retryAttempt := 0
    416 	retryDelay := uploadRetryInitialBackoff
    417 
    418 	auth, err := b.authorizeAccount(ctx)
    419 	if err != nil {
    420 		return nil, b.createError(err)
    421 	}
    422 
    423 	lbReq := &b2.ListBucketsRequest{
    424 		Authorization: auth.authorizationToken,
    425 		BaseURL:       auth.apiURL,
    426 		AccountID:     auth.accountID,
    427 		BucketName:    request.Resource.Bucket,
    428 	}
    429 
    430 	lbResp, err := b.client.ListBuckets(ctx, lbReq)
    431 	if err != nil {
    432 		return nil, b.createError(err)
    433 	}
    434 
    435 	if len(lbResp.Buckets) != 1 {
    436 		return nil, status.Errorf(
    437 			codes.NotFound,
    438 			"backend: unexpected number of buckets matched by name: name=%s buckets=%d",
    439 			request.Resource.Bucket,
    440 			len(lbResp.Buckets),
    441 		)
    442 	}
    443 
    444 	for {
    445 		uuReq := &b2.GetUploadURLRequest{
    446 			Authorization: auth.authorizationToken,
    447 			BaseURL:       auth.apiURL,
    448 			BucketID:      lbResp.Buckets[0].BucketID,
    449 		}
    450 
    451 		uuResp, err := b.client.GetUploadURL(ctx, uuReq)
    452 		if err != nil {
    453 			return nil, b.createError(err)
    454 		}
    455 
    456 		attributes, err := schemas.MarshalAttributes(request.Attributes, nil)
    457 		if err != nil {
    458 			return nil, b.createError(err)
    459 		}
    460 
    461 		hash := sha1.New()
    462 		hash.Write(request.Data)
    463 		checksum := fmt.Sprintf("%x", hash.Sum(nil))
    464 
    465 		ufReq := &b2.UploadFileRequest{
    466 			Authorization: uuResp.AuthorizationToken,
    467 			UploadURL:     uuResp.UploadURL.URL,
    468 			FileName:      request.Key,
    469 			FileInfo:      attributes,
    470 			ContentType:   "b2/x-auto",
    471 			ContentLength: len(request.Data),
    472 			ContentSHA1:   checksum,
    473 			Data:          io.NopCloser(bytes.NewReader(request.Data)),
    474 		}
    475 
    476 		ufResp, err := b.client.UploadFile(ctx, ufReq)
    477 		if err != nil {
    478 			if e, ok := err.(*b2.Status); ok && e.Status >= http.StatusInternalServerError && retryAttempt < uploadRetryBudget {
    479 				zap.L().Warn(
    480 					"full file upload failed with server error; retrying after delay",
    481 					zap.Int("attempt", retryAttempt),
    482 					zap.Duration("delay", retryDelay),
    483 					zap.Error(err),
    484 				)
    485 
    486 				select {
    487 				case <-ctx.Done():
    488 					return nil, status.FromContextError(ctx.Err()).Err()
    489 				case <-time.After(retryDelay):
    490 				}
    491 
    492 				retryAttempt++
    493 				retryDelay *= 2 // Exponential backoff
    494 				continue
    495 			}
    496 
    497 			return nil, b.createError(err)
    498 		}
    499 
    500 		return &service.PutObjectResponse{
    501 			Metadata: &common.Metadata{
    502 				Id:           ufResp.FileID,
    503 				Key:          ufResp.FileName,
    504 				Size:         uint64(ufResp.ContentLength),
    505 				ObjectType:   common.Object_REGULAR,
    506 				MimeType:     ufResp.ContentType,
    507 				ModifiedTime: timestamppb.New(ufResp.UploadTimestamp.Time),
    508 				Checksum:     ufResp.ContentSHA1,
    509 				Attributes:   request.Attributes,
    510 			},
    511 		}, nil
    512 	}
    513 }

    515 // StreamPutObject is a streaming implementation of PutObject. It identifies the target file, starts
    516 // a large file, authorizes a chunked upload URL, uploads each incoming chunk in a single request,
    517 // and finishes the file when all chunks have been uploaded.
    518 func (b *B2) StreamPutObject(ctx context.Context, stream chan *service.PutObjectStreamRequest) (*service.PutObjectStreamResponse, error) {
    519 	var authorization string
    520 	var uploadURL *url.URL
    521 	var fileID string
    522 	var checksums []string
    523 
    524 	part := 1
    525 
    526 	auth, err := b.authorizeAccount(ctx)
    527 	if err != nil {
    528 		return nil, b.createError(err)
    529 	}
    530 
    531 	for request := range stream {
    532 		retryAttempt := 0
    533 		retryDelay := uploadRetryInitialBackoff
    534 
    535 		if request.Range != nil {
    536 			return nil, status.Errorf(
    537 				codes.InvalidArgument,
    538 				"backend: B2 does not support range specifications for streaming puts: range=%v",
    539 				request.Range,
    540 			)
    541 		}
    542 
    543 		if fileID == "" {
    544 			lbReq := &b2.ListBucketsRequest{
    545 				Authorization: auth.authorizationToken,
    546 				BaseURL:       auth.apiURL,
    547 				AccountID:     auth.accountID,
    548 				BucketName:    request.Request.Resource.Bucket,
    549 			}
    550 
    551 			lbResp, err := b.client.ListBuckets(ctx, lbReq)
    552 			if err != nil {
    553 				return nil, b.createError(err)
    554 			}
    555 
    556 			if len(lbResp.Buckets) != 1 {
    557 				return nil, status.Errorf(
    558 					codes.NotFound,
    559 					"backend: unexpected number of buckets matched by name: name=%s buckets=%d",
    560 					request.Request.Resource.Bucket,
    561 					len(lbResp.Buckets),
    562 				)
    563 			}
    564 
    565 			// The B2 API only supports writing file info when the large file is
    566 			// started, but at this point, the total size of the file is not yet known.
    567 			// Explicitly override its value to 0 to indicate that the original size is
    568 			// not known.
    569 			request.Request.Attributes.Size = 0
    570 			attributes, err := schemas.MarshalAttributes(request.Request.Attributes, nil)
    571 			if err != nil {
    572 				return nil, b.createError(err)
    573 			}
    574 
    575 			sfReq := &b2.StartLargeFileRequest{
    576 				Authorization: auth.authorizationToken,
    577 				BaseURL:       auth.apiURL,
    578 				BucketID:      lbResp.Buckets[0].BucketID,
    579 				FileName:      request.Request.Key,
    580 				FileInfo:      attributes,
    581 				ContentType:   "b2/x-auto",
    582 			}
    583 
    584 			sfResp, err := b.client.StartLargeFile(ctx, sfReq)
    585 			if err != nil {
    586 				return nil, b.createError(err)
    587 			}
    588 
    589 			fileID = sfResp.FileID
    590 		}
    591 
    592 		for {
    593 			if uploadURL == nil {
    594 				uuReq := &b2.GetUploadPartURLRequest{
    595 					Authorization: auth.authorizationToken,
    596 					BaseURL:       auth.apiURL,
    597 					FileID:        fileID,
    598 				}
    599 
    600 				uuResp, err := b.client.GetUploadPartURL(ctx, uuReq)
    601 				if err != nil {
    602 					return nil, b.createError(err)
    603 				}
    604 
    605 				authorization = uuResp.AuthorizationToken
    606 				uploadURL = uuResp.UploadURL.URL
    607 			}
    608 
    609 			hash := sha1.New()
    610 			hash.Write(request.Request.Data)
    611 			checksum := fmt.Sprintf("%x", hash.Sum(nil))
    612 
    613 			upReq := &b2.UploadPartRequest{
    614 				Authorization: authorization,
    615 				UploadURL:     uploadURL,
    616 				PartNumber:    part,
    617 				ContentLength: len(request.Request.Data),
    618 				ContentSHA1:   checksum,
    619 				Data:          io.NopCloser(bytes.NewReader(request.Request.Data)),
    620 			}
    621 
    622 			if _, err := b.client.UploadPart(ctx, upReq); err != nil {
    623 				if e, ok := err.(*b2.Status); ok && e.Status >= http.StatusInternalServerError && retryAttempt < uploadRetryBudget {
    624 					zap.L().Warn(
    625 						"chunked file upload failed with server error; retrying after delay",
    626 						zap.Int("attempt", retryAttempt),
    627 						zap.Duration("delay", retryDelay),
    628 						zap.Error(err),
    629 					)
    630 
    631 					select {
    632 					case <-ctx.Done():
    633 						return nil, status.FromContextError(ctx.Err()).Err()
    634 					case <-time.After(retryDelay):
    635 					}
    636 
    637 					// Explicitly reset the upload URL so that the retry also
    638 					// obtains a new upload URL before retrying the actual
    639 					// upload. Such client behavior is mandated by the B2 API.
    640 					// Reference: https://www.backblaze.com/b2/docs/uploading.html
    641 					uploadURL = nil
    642 
    643 					retryAttempt++
    644 					retryDelay *= 2 // Exponential backoff
    645 					continue
    646 				}
    647 
    648 				return nil, b.createError(err)
    649 			}
    650 
    651 			part++
    652 			checksums = append(checksums, checksum)
    653 			break
    654 		}
    655 	}
    656 
    657 	ffReq := &b2.FinishLargeFileRequest{
    658 		Authorization: auth.authorizationToken,
    659 		BaseURL:       auth.apiURL,
    660 		FileID:        fileID,
    661 		PartSHA1Array: checksums,
    662 	}
    663 
    664 	ffResp, err := b.client.FinishLargeFile(ctx, ffReq)
    665 	if err != nil {
    666 		return nil, b.createError(err)
    667 	}
    668 
    669 	attributes, err := schemas.UnmarshalAttributes(ffResp.FileInfo, nil)
    670 	if err != nil {
    671 		return nil, b.createError(err)
    672 	}
    673 
    674 	return &service.PutObjectStreamResponse{
    675 		Response: &service.PutObjectResponse{
    676 			Metadata: &common.Metadata{
    677 				Id:           ffResp.FileID,
    678 				Key:          ffResp.FileName,
    679 				Size:         uint64(ffResp.ContentLength),
    680 				ObjectType:   common.Object_REGULAR,
    681 				MimeType:     ffResp.ContentType,
    682 				ModifiedTime: timestamppb.New(ffResp.UploadTimestamp.Time),
    683 				Checksum:     ffResp.ContentSHA1,
    684 				Attributes:   attributes,
    685 			},
    686 		},
    687 	}, nil
    688 }

    690 // DeleteObject deletes the requested file.
    691 func (b *B2) DeleteObject(ctx context.Context, request *service.DeleteObjectRequest) (*service.DeleteObjectResponse, error) {
    692 	auth, err := b.authorizeAccount(ctx)
    693 	if err != nil {
    694 		return nil, b.createError(err)
    695 	}
    696 
    697 	lbReq := &b2.ListBucketsRequest{
    698 		Authorization: auth.authorizationToken,
    699 		BaseURL:       auth.apiURL,
    700 		AccountID:     auth.accountID,
    701 		BucketName:    request.Resource.Bucket,
    702 	}
    703 
    704 	lbResp, err := b.client.ListBuckets(ctx, lbReq)
    705 	if err != nil {
    706 		return nil, b.createError(err)
    707 	}
    708 
    709 	if len(lbResp.Buckets) != 1 {
    710 		return nil, status.Errorf(
    711 			codes.NotFound,
    712 			"backend: unexpected number of buckets matched by name: name=%s buckets=%d",
    713 			request.Resource.Bucket,
    714 			len(lbResp.Buckets),
    715 		)
    716 	}
    717 
    718 	lfReq := &b2.ListFileNamesRequest{
    719 		Authorization: auth.authorizationToken,
    720 		BaseURL:       auth.apiURL,
    721 		BucketID:      lbResp.Buckets[0].BucketID,
    722 		StartFileName: request.Key,
    723 		MaxFileCount:  1,
    724 		Prefix:        request.Key,
    725 		Delimiter:     "/",
    726 	}
    727 
    728 	if request.Recursive {
    729 		lfReq.Delimiter = ""
    730 		lfReq.MaxFileCount = 10000
    731 	}
    732 
    733 	lfResp, err := b.client.ListFileNames(ctx, lfReq)
    734 	if err != nil {
    735 		return nil, b.createError(err)
    736 	}
    737 
    738 	if len(lfResp.Files) == 0 {
    739 		return nil, status.Errorf(
    740 			codes.NotFound,
    741 			"backend: no files match requested name: bucket=%s name=%s",
    742 			request.Resource.Bucket,
    743 			request.Key,
    744 		)
    745 	} else if !request.Recursive && len(lfResp.Files) == 1 && lfResp.Files[0].FileID == "" {
    746 		// To emulate Unistore API semantics, disallow non-recursive deletions on non-empty
    747 		// directories. The B2 API represents this a single matched file with no file ID.
    748 		return nil, status.Errorf(
    749 			codes.FailedPrecondition,
    750 			"backend: cannot delete directory without recursive flag: bucket=%s name=%s",
    751 			request.Resource.Bucket,
    752 			request.Key,
    753 		)
    754 	} else if !request.Recursive && len(lfResp.Files) > 0 && strings.HasSuffix(request.Key, "/") {
    755 		// Directories can only exist in B2 if there are nonzero children (potentially
    756 		// nested arbitrarily deeply). Require such deletions to be recursive.
    757 		return nil, status.Errorf(
    758 			codes.FailedPrecondition,
    759 			"backend: cannot delete non-empty directory without recursive flag: bucket=%s name=%s children=%d",
    760 			request.Resource.Bucket,
    761 			request.Key,
    762 			len(lfResp.Files),
    763 		)
    764 	}
    765 
    766 	for _, file := range lfResp.Files {
    767 		dfReq := &b2.DeleteFileVersionRequest{
    768 			Authorization: auth.authorizationToken,
    769 			BaseURL:       auth.apiURL,
    770 			FileName:      file.FileName,
    771 			FileID:        file.FileID,
    772 		}
    773 
    774 		if _, err := b.client.DeleteFileVersion(ctx, dfReq); err != nil {
    775 			return nil, b.createError(err)
    776 		}
    777 	}
    778 
    779 	return &service.DeleteObjectResponse{Deleted: int32(len(lfResp.Files))}, nil
    780 }

    782 // ListBuckets retrieves metadata for all buckets.
    783 func (b *B2) ListBuckets(ctx context.Context, request *service.ListBucketsRequest) (*service.ListBucketsResponse, error) {
    784 	auth, err := b.authorizeAccount(ctx)
    785 	if err != nil {
    786 		return nil, b.createError(err)
    787 	}
    788 
    789 	lbReq := &b2.ListBucketsRequest{
    790 		Authorization: auth.authorizationToken,
    791 		BaseURL:       auth.apiURL,
    792 		AccountID:     auth.accountID,
    793 	}
    794 
    795 	lbResp, err := b.client.ListBuckets(ctx, lbReq)
    796 	if err != nil {
    797 		return nil, b.createError(err)
    798 	}
    799 
    800 	buckets := make([]*service.HeadBucketResponse, len(lbResp.Buckets))
    801 
    802 	for i, bucket := range lbResp.Buckets {
    803 		var visibility common.Visibility
    804 
    805 		switch bucket.BucketType {
    806 		case "allPublic":
    807 			visibility = common.Visibility_PUBLIC
    808 		case "allPrivate", "snapshot":
    809 			visibility = common.Visibility_PRIVATE
    810 		default:
    811 			visibility = common.Visibility_DEFAULT
    812 		}
    813 
    814 		lifecyclePolicies := make([]*common.LifecyclePolicy, len(bucket.LifecycleRules))
    815 
    816 		for j, rule := range bucket.LifecycleRules {
    817 			pattern := fmt.Sprintf("^%s", regexp.QuoteMeta(rule.FileNamePrefix))
    818 			ttl := time.Duration(rule.DaysFromUploadingToHiding+rule.DaysFromHidingToDeleting) * 24 * time.Hour
    819 
    820 			lifecyclePolicies[j] = &common.LifecyclePolicy{
    821 				Pattern:    pattern,
    822 				Expiration: durationpb.New(ttl),
    823 			}
    824 		}
    825 
    826 		buckets[i] = &service.HeadBucketResponse{
    827 			Bucket: &common.Bucket{
    828 				Id:                bucket.BucketID,
    829 				Name:              bucket.BucketName,
    830 				Visibility:        visibility,
    831 				LifecyclePolicies: lifecyclePolicies,
    832 				Tags:              bucket.BucketInfo,
    833 			},
    834 		}
    835 	}
    836 
    837 	return &service.ListBucketsResponse{Buckets: buckets}, nil
    838 }

    933 // authorizeAccount invokes the B2 account authorization API to obtain a short-lived authorization
    934 // token for subsequent API calls. Authentication state is cached local to the backend instance and
    935 // refreshed at a regular interval.
    936 func (b *B2) authorizeAccount(ctx context.Context) (*authenticationState, error) {
    937 	b.authMutex.Lock()
    938 	defer b.authMutex.Unlock()
    939 
    940 	// Authorization tokens are good for at most 24 hours after they have been issued by a call
    941 	// to b2_authorize_account. Somewhat conservatively, this opts to refresh the token if more
    942 	// than 3 hours have elapsed since the last successful refresh.
    943 	if time.Since(b.auth.refresh) < authenticationRefreshInterval {
    944 		return b.auth, nil
    945 	}
    946 
    947 	req := &b2.AuthorizeAccountRequest{
    948 		ApplicationKeyID: b.applicationKeyID,
    949 		ApplicationKey:   b.applicationKey,
    950 	}
    951 
    952 	resp, err := b.client.AuthorizeAccount(ctx, req)
    953 	if err != nil {
    954 		return nil, err
    955 	}
    956 
    957 	b.auth = &authenticationState{
    958 		accountID:          resp.AccountID,
    959 		authorizationToken: resp.AuthorizationToken,
    960 		apiURL:             resp.APIURL.URL,
    961 		downloadURL:        resp.DownloadURL.URL,
    962 		refresh:            time.Now(),
    963 	}
    964 
    965 	return b.auth, nil
    966 }

    968 // createError creates a semantically accurate gRPC status for an error status from the B2 API.
    969 func (b *B2) createError(err error) error {
    970 	b2Errors := map[int]codes.Code{
    971 		http.StatusBadRequest:                   codes.InvalidArgument,
    972 		http.StatusUnauthorized:                 codes.PermissionDenied,
    973 		http.StatusForbidden:                    codes.PermissionDenied,
    974 		http.StatusNotFound:                     codes.NotFound,
    975 		http.StatusMethodNotAllowed:             codes.InvalidArgument,
    976 		http.StatusRequestTimeout:               codes.DeadlineExceeded,
    977 		http.StatusRequestedRangeNotSatisfiable: codes.OutOfRange,
    978 		http.StatusTooManyRequests:              codes.ResourceExhausted,
    979 		http.StatusInternalServerError:          codes.Unknown,
    980 		http.StatusServiceUnavailable:           codes.Unavailable,
    981 	}
    982 
    983 	switch e := err.(type) {
    984 	case *b2.Status:
    985 		code, ok := b2Errors[e.Status]
    986 		if !ok {
    987 			code = codes.Unknown
    988 		}
    989 
    990 		return status.Error(code, fmt.Sprintf("%s (code %s)", e.Message, e.Code))
    991 	default:
    992 		return status.Error(codes.Unknown, e.Error())
    993 	}
    994 }
