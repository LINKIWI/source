      7 import withTransactionalTunnel from 'client/app/react/containers/hoc/with-transactional-tunnel';
      8 import { setIndexMeta } from 'client/app/redux/actions/meta';
      9 import { TELEMETRY_ACTIONS } from 'shared/constants/telemetry';
     10 
     11 // Threshold number of consecutive, consistent metadata messages before considering the metadata
     12 // accurate.
     13 const METADATA_COMMIT_THRESHOLD = 3;
     14 
     15 /**
     16  * Dummy container that polls index metadata at a regular intervals and updates global state.
     17  */
     18 class MetadataPollContainer extends Component {
     19   static propTypes = {
     20     pollInterval: PropTypes.number.isRequired,
     21     // HOC props
     22     info: PropTypes.shape({
     23       err: PropTypes.object,
     24       messages: PropTypes.array.isRequired,
     25       isConnected: PropTypes.bool.isRequired,
     26       sendMessage: PropTypes.func.isRequired,
     27     }).isRequired,
     28     actions: PropTypes.shape({
     29       setIndexMeta: PropTypes.func.isRequired,
     30     }).isRequired,
     31     recordTelemetryEvent: PropTypes.func.isRequired,
     32   };
     33 
     34   componentDidMount() {
     35     this._tick();
     36   }
     37 
     38   componentDidUpdate() {
     39     const { info: { messages }, actions, recordTelemetryEvent } = this.props;
     40 
     41     if (messages.length !== METADATA_COMMIT_THRESHOLD) {
     42       return;
     43     }
     44 
     45     // Consider the metadata to be valid only if all messages in the buffer have a consistent index
     46     // name and timestamp. This effectively requires the server-side metadata to be consistent for
     47     // an extended duration of time, to accommodate clustered livegrep deployments which might be
     48     // serving different versions of the index at the same time while the index is being updated.
     49     const metadata = messages.reduce(
     50       (acc, message) => (
     51         (acc &&
     52           !message.err &&
     53           message.data &&
     54           acc.name === message.data.name &&
     55           acc.timestamp === message.data.timestamp) ? acc : null
     56       ),
     57       messages[0].data || {},
     58     );
     59 
     60     if (metadata) {
     61       actions.setIndexMeta(metadata);
     62       recordTelemetryEvent(TELEMETRY_ACTIONS.COMMIT_SERVER_METADATA);
     63     }
     64   }
     65 
     66   componentWillUnmount() {
     67     window.clearInterval(this.timeout);
     68   }
     69 
     70   _tick() {
     71     const { info, pollInterval, recordTelemetryEvent } = this.props;
     72 
     73     this.timeout = window.setTimeout(() => {
     74       if (!document.hidden) {
     75         info.sendMessage();
     76         recordTelemetryEvent(TELEMETRY_ACTIONS.POLL_SERVER_METADATA);
     77       }
     78 
     79       this._tick();
     80     }, pollInterval);
     81   }
     82 
     83   render() {
     84     return null;
     85   }
     86 }
     87 
     88 const mapDispatchToProps = (dispatch) => ({
     89   actions: bindActionCreators({ setIndexMeta }, dispatch),
     90 });
     91 
     92 export default compose(
     93   connect(null, mapDispatchToProps),
     94   withTelemetry,
     95   withTransactionalTunnel({
     96     key: 'info',
     97     endpoint: '/api/meta/info',
     98     messageBufferSize: METADATA_COMMIT_THRESHOLD,
     99   }),
    100 )(MetadataPollContainer);
