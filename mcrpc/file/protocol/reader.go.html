     54 // ReadASCIICommand reads a full ASCII command from the underlying io.Reader, exercising heuristics
     55 // to determine how far into the stream data should be read.
     56 func (r *Reader) ReadASCIICommand() ([]byte, error) {
     57 	buf, err := r.readUntil(crlf)
     58 	if err != nil {
     59 		return nil, err
     60 	}
     61 
     62 	// Storage commands are the only protocol request messages that conclude on the second CRLF
     63 	// sequence, which is treated as a separate case.
     64 	for _, cmd := range storageCommands {
     65 		if bytes.HasPrefix(buf, cmd) {
     66 			data, err := r.readUntil(crlf)
     67 			if err != nil {
     68 				return nil, err
     69 			}
     70 
     71 			return append(buf, data...), nil
     72 		}
     73 	}
     74 
     75 	return buf, nil
     76 }

     78 // readUntil incrementally reads from the buffered stream in chunks until the delimiter sequence is
     79 // located. Only those bytes that represent the fully-formed message are consumed from the buffered
     80 // reader. If the sequence is located before EOF, a byte slice with the full message contents is
     81 // returned; otherwise, an error indicates the command is malformed.
     82 func (r *Reader) readUntil(sequence []byte) ([]byte, error) {
     83 	var buf []byte
     84 	var terminate bool
     85 
     86 	for !terminate {
     87 		chunk, err := r.b.Peek(scanChunkSize)
     88 		if err != nil {
     89 			if err == io.EOF && len(buf) > 0 {
     90 				// Terminate the chunk-read loop on the next iteration, relevant
     91 				// only when a nonzero buffer has already been accumulated.
     92 				terminate = true
     93 			} else {
     94 				return nil, err
     95 			}
     96 		}
     97 
     98 		buf = append(buf, chunk...)
     99 
    100 		idx := bytes.Index(buf, sequence)
    101 		if idx != -1 {
    102 			offset := idx + len(sequence)
    103 			r.b.Discard(offset - len(buf) + len(chunk))
    104 			return buf[:offset], nil
    105 		}
    106 
    107 		r.b.Discard(len(chunk))
    108 	}
    109 
    110 	return nil, ErrMalformedRequest
