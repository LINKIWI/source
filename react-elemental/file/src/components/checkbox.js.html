     11 /**
     12  * Styled checkbox element.

     14 class Checkbox extends Component {

     16     checked: PropTypes.bool,
     17     label: PropTypes.string,
     18     style: PropTypes.object,
     19     disabled: PropTypes.bool,
     20     onChange: PropTypes.func,
     21     children: PropTypes.node,

     26   };

     28   static defaultProps = {
     29     checked: false,
     30     label: null,

     33     onChange: noop,

     38   state = { isHover: false, isFocus: false };

     40   handleClick = () => {
     41     const { checked, onChange, disabled } = this.props;
     42 
     43     return !disabled && onChange(!checked);
     44   };
     45 
     46   handleHoverChange = (isHover) => () => this.setState({ isHover });

     49

     52       checked,
     53       label,
     54       style: overrides,

     56       children,

     64       if (checked) {
     65         return colors.primary;
     66       }
     67 
     68       return disabled ? colors.gray5 : 'rgb(250, 250, 250)';
     69     })();
     70 
     71     const borderColor = (() => {
     72       if (checked) {
     73         return colors.primary;
     74       }
     75 
     76       return isActive ? colors.gray20 : colors.gray10;
     77     })();
     78 
     79     const containerStyle = {
     80       alignItems: 'center',
     81       background: 'inherit',
     82       border: 0,
     83       cursor: disabled ? 'inherit' : 'pointer',
     84       display: 'inline-flex',
     85       ...overrides,
     86     };
     87 
     88     const checkboxStyle = {
     89       alignItems: 'center',
     90       backgroundColor,
     91       border: `1px solid ${borderColor}`,
     92       display: 'flex',

     99 
    100     const checkStyle = {

    105     };
    106 
    107     return (
    108       <button

    117         onBlur={this.handleFocusChange(false)}

    120         <div style={checkboxStyle}>
    121           <Check style={checkStyle} />
    122         </div>

    124         {label && (
    125           <Spacing size="small" left inline>

    135   }
