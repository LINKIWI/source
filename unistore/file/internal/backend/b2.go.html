    243 func (b *B2) GetObject(ctx context.Context, request *service.GetObjectRequest) (*service.GetObjectResponse, error) {
    244 	auth, err := b.authorizeAccount(ctx)
    245 	if err != nil {
    246 		return nil, b.createError(err)
    247 	}
    248 
    249 	dfReq := &b2.DownloadFileByNameRequest{
    250 		Authorization: auth.authorizationToken,
    251 		DownloadURL:   auth.downloadURL,
    252 		Bucket:        request.Resource.Bucket,

    254 	}
    255 
    256 	if request.Range != nil {
    257 		dfReq.Range = b2.Range{
    258 			Unit:  request.Range.Unit,
    259 			Start: int(request.Range.Start),
    260 		}

    262 		if request.Range.End > 0 {
    263 			// Note that the B2 API considers the range end index to be inclusive, while
    264 			// the Unistore API considers the end index to be exclusive. Normalize this
    265 			// behavior by subtracting 1 from the desired end index for compatibility.
    266 			dfReq.Range.End = int(request.Range.End) - 1
    267 		}
    268 	}
    269 
    270 	dfResp, err := b.client.DownloadFileByName(ctx, dfReq)
    271 	if err != nil {
    272 		return nil, b.createError(err)
    273 	}
    274 
    275 	defer dfResp.Data.Close()
    276 
    277 	data, err := io.ReadAll(dfResp.Data)
    278 	if err != nil {
    279 		return nil, b.createError(err)
    280 	}
    281 
    282 	attributes, err := schemas.UnmarshalAttributes(dfResp.FileInfo, nil)
    283 	if err != nil {
    284 		return nil, b.createError(err)
    285 	}
    286 
    287 	return &service.GetObjectResponse{
    288 		Metadata: &common.Metadata{
    289 			Id:           dfResp.FileID,
    290 			Key:          dfResp.FileName,
    291 			Size:         uint64(dfResp.Size),
    292 			ObjectType:   common.Object_REGULAR,
    293 			MimeType:     dfResp.ContentType,
    294 			ModifiedTime: timestamppb.New(dfResp.UploadTimestamp.Time),

    296 			Attributes:   attributes,
    297 		},
    298 		Data: data,
    299 	}, nil

    302 // StreamGetObject is a streaming implementation of GetObject. It segments the requests to B2 in
    303 // chunks of the requested size using the Range request directive, and pipes chunked responses into
    304 // a stream channel.
    305 func (b *B2) StreamGetObject(ctx context.Context, request *service.GetObjectStreamRequest) (<-chan *service.GetObjectStreamResponse, chan error) {
    306 	stream := make(chan *service.GetObjectStreamResponse)
    307 	errs := make(chan error, 1)
    308 
    309 	if request.ChunkSize == 0 {
    310 		errs <- status.Error(
    311 			codes.InvalidArgument,
    312 			"backend: B2 streaming get chunk size must be greater than zero",
    313 		)
    314 		return nil, errs
    315 	}
    316 
    317 	auth, err := b.authorizeAccount(ctx)
    318 	if err != nil {
    319 		errs <- b.createError(err)
    320 		return nil, errs
    321 	}
    322 
    323 	go func() {
    324 		var offset uint64
    325 
    326 		if request.Request.Range != nil {
    327 			offset = request.Request.Range.Start
    328 		}
    329 
    330 		for {
    331 			// Read the requested chunk size, but only up to the end index specified by
    332 			// the range, when available.
    333 			end := offset + request.ChunkSize
    334 			if request.Request.Range != nil &&
    335 				request.Request.Range.End > 0 &&
    336 				end > request.Request.Range.End {
    337 				end = request.Request.Range.End
    338 			}
    339 
    340 			dfReq := &b2.DownloadFileByNameRequest{
    341 				Authorization: auth.authorizationToken,
    342 				DownloadURL:   auth.downloadURL,
    343 				Bucket:        request.Request.Resource.Bucket,
    344 				File:          request.Request.Key,
    345 				// Note that the B2 API considers the range end index to be
    346 				// inclusive. To normalize this behavior against the Unistore API,
    347 				// add/subtract 1 from the end index as needed for non-inclusive
    348 				// end index semantics.
    349 				Range: b2.Range{
    350 					Unit:  "bytes",
    351 					Start: int(offset),
    352 					End:   int(end) - 1,
    353 				},
    354 			}
    355 
    356 			dfResp, err := b.client.DownloadFileByName(ctx, dfReq)
    357 			if err != nil {
    358 				errs <- b.createError(err)
    359 				break
    360 			}
    361 
    362 			chunk, err := io.ReadAll(dfResp.Data)
    363 			if err != nil {
    364 				dfResp.Data.Close()
    365 				errs <- b.createError(err)
    366 				break
    367 			}
    368 
    369 			dfResp.Data.Close()
    370 
    371 			attributes, err := schemas.UnmarshalAttributes(dfResp.FileInfo, nil)
    372 			if err != nil {
    373 				errs <- b.createError(err)
    374 				break
    375 			}
    376 
    377 			stream <- &service.GetObjectStreamResponse{
    378 				Response: &service.GetObjectResponse{
    379 					Metadata: &common.Metadata{
    380 						Id:           dfResp.FileID,
    381 						Key:          dfResp.FileName,
    382 						Size:         uint64(dfResp.Size),
    383 						ObjectType:   common.Object_REGULAR,
    384 						MimeType:     dfResp.ContentType,
    385 						ModifiedTime: timestamppb.New(dfResp.UploadTimestamp.Time),
    386 						Checksum:     dfResp.ContentSHA1,
    387 						Attributes:   attributes,
    388 					},
    389 					Data: chunk,
    390 				},
    391 				Range: &common.Range{
    392 					Unit:  "bytes",
    393 					Start: uint64(dfResp.Range.Start),
    394 					End:   uint64(dfResp.Range.End) + 1,
    395 					Total: uint64(dfResp.Range.Total),
    396 				},
    397 			}
    398 
    399 			offset += uint64(dfResp.Range.End - dfResp.Range.Start + 1)
    400 
    401 			if offset >= uint64(dfResp.Range.Total) {
    402 				break
    403 			}
    404 
    405 			if request.Request.Range != nil &&
    406 				request.Request.Range.End > 0 &&
    407 				offset >= request.Request.Range.End {
    408 				break
    409 			}
    410 		}
    411 
    412 		close(stream)
    413 	}()
    414 
    415 	return stream, errs
    416 }

    418 // PutObject uploads the request file in full.
    419 func (b *B2) PutObject(ctx context.Context, request *service.PutObjectRequest) (*service.PutObjectResponse, error) {
    420 	retryAttempt := 0
    421 	retryDelay := uploadRetryInitialBackoff
    422 
    423 	auth, err := b.authorizeAccount(ctx)
    424 	if err != nil {
    425 		return nil, b.createError(err)
    426 	}
    427 
    428 	lbReq := &b2.ListBucketsRequest{
    429 		Authorization: auth.authorizationToken,
    430 		BaseURL:       auth.apiURL,
    431 		AccountID:     auth.accountID,
    432 		BucketName:    request.Resource.Bucket,
    433 	}
    434 
    435 	lbResp, err := b.client.ListBuckets(ctx, lbReq)
    436 	if err != nil {
    437 		return nil, b.createError(err)
    438 	}
    439 
    440 	if len(lbResp.Buckets) != 1 {
    441 		return nil, status.Errorf(
    442 			codes.NotFound,
    443 			"backend: unexpected number of buckets matched by name: name=%s buckets=%d",
    444 			request.Resource.Bucket,
    445 			len(lbResp.Buckets),
    446 		)
    447 	}
    448 
    449 	for {
    450 		uuReq := &b2.GetUploadURLRequest{
    451 			Authorization: auth.authorizationToken,
    452 			BaseURL:       auth.apiURL,
    453 			BucketID:      lbResp.Buckets[0].BucketID,
    454 		}
    455 
    456 		uuResp, err := b.client.GetUploadURL(ctx, uuReq)
    457 		if err != nil {
    458 			return nil, b.createError(err)
    459 		}
    460 
    461 		attributes, err := schemas.MarshalAttributes(request.Attributes, nil)
    462 		if err != nil {
    463 			return nil, b.createError(err)
    464 		}
    465 
    466 		hash := sha1.New()
    467 		hash.Write(request.Data)
    468 		checksum := fmt.Sprintf("%x", hash.Sum(nil))
    469 
    470 		ufReq := &b2.UploadFileRequest{
    471 			Authorization: uuResp.AuthorizationToken,
    472 			UploadURL:     uuResp.UploadURL.URL,
    473 			FileName:      request.Key,
    474 			FileInfo:      attributes,
    475 			ContentType:   "b2/x-auto",
    476 			ContentLength: len(request.Data),
    477 			ContentSHA1:   checksum,
    478 			Data:          io.NopCloser(bytes.NewReader(request.Data)),
    479 		}
    480 
    481 		ufResp, err := b.client.UploadFile(ctx, ufReq)
    482 		if err != nil {
    483 			if e, ok := err.(*b2.Status); ok && e.Status >= http.StatusInternalServerError && retryAttempt < uploadRetryBudget {
    484 				zap.L().Warn(
    485 					"full file upload failed with server error; retrying after delay",
    486 					zap.Int("attempt", retryAttempt),
    487 					zap.Duration("delay", retryDelay),
    488 					zap.Error(err),
    489 				)
    490 
    491 				select {
    492 				case <-ctx.Done():
    493 					return nil, status.FromContextError(ctx.Err()).Err()
    494 				case <-time.After(retryDelay):
    495 				}
    496 
    497 				retryAttempt++
    498 				retryDelay *= 2 // Exponential backoff
    499 				continue
    500 			}
    501 
    502 			return nil, b.createError(err)
    503 		}
    504 
    505 		return &service.PutObjectResponse{
    506 			Metadata: &common.Metadata{
    507 				Id:           ufResp.FileID,
    508 				Key:          ufResp.FileName,
    509 				Size:         uint64(ufResp.ContentLength),
    510 				ObjectType:   common.Object_REGULAR,
    511 				MimeType:     ufResp.ContentType,
    512 				ModifiedTime: timestamppb.New(ufResp.UploadTimestamp.Time),
    513 				Checksum:     ufResp.ContentSHA1,
    514 				Attributes:   request.Attributes,
    515 			},
    516 		}, nil
    517 	}
    518 }

    520 // StreamPutObject is a streaming implementation of PutObject. It identifies the target file, starts
    521 // a large file, authorizes a chunked upload URL, uploads each incoming chunk in a single request,
    522 // and finishes the file when all chunks have been uploaded.
    523 func (b *B2) StreamPutObject(ctx context.Context, stream chan *service.PutObjectStreamRequest) (*service.PutObjectStreamResponse, error) {
    524 	var authorization string
    525 	var uploadURL *url.URL
    526 	var fileID string
    527 	var checksums []string
    528 
    529 	part := 1
    530 
    531 	auth, err := b.authorizeAccount(ctx)
    532 	if err != nil {
    533 		return nil, b.createError(err)
    534 	}
    535 
    536 	for request := range stream {
    537 		retryAttempt := 0
    538 		retryDelay := uploadRetryInitialBackoff
    539 
    540 		if request.Range != nil {
    541 			return nil, status.Errorf(
    542 				codes.InvalidArgument,
    543 				"backend: B2 does not support range specifications for streaming puts: range=%v",
    544 				request.Range,
    545 			)
    546 		}
    547 
    548 		if fileID == "" {
    549 			lbReq := &b2.ListBucketsRequest{
    550 				Authorization: auth.authorizationToken,
    551 				BaseURL:       auth.apiURL,
    552 				AccountID:     auth.accountID,
    553 				BucketName:    request.Request.Resource.Bucket,
    554 			}
    555 
    556 			lbResp, err := b.client.ListBuckets(ctx, lbReq)
    557 			if err != nil {
    558 				return nil, b.createError(err)
    559 			}
    560 
    561 			if len(lbResp.Buckets) != 1 {
    562 				return nil, status.Errorf(
    563 					codes.NotFound,
    564 					"backend: unexpected number of buckets matched by name: name=%s buckets=%d",
    565 					request.Request.Resource.Bucket,
    566 					len(lbResp.Buckets),
    567 				)
    568 			}
    569 
    570 			// The B2 API only supports writing file info when the large file is
    571 			// started, but at this point, the total size of the file is not yet known.
    572 			// Explicitly override its value to 0 to indicate that the original size is
    573 			// not known.
    574 			request.Request.Attributes.Size = 0
    575 			attributes, err := schemas.MarshalAttributes(request.Request.Attributes, nil)
    576 			if err != nil {
    577 				return nil, b.createError(err)
    578 			}
    579 
    580 			sfReq := &b2.StartLargeFileRequest{
    581 				Authorization: auth.authorizationToken,
    582 				BaseURL:       auth.apiURL,
    583 				BucketID:      lbResp.Buckets[0].BucketID,
    584 				FileName:      request.Request.Key,
    585 				FileInfo:      attributes,
    586 				ContentType:   "b2/x-auto",
    587 			}
    588 
    589 			sfResp, err := b.client.StartLargeFile(ctx, sfReq)
    590 			if err != nil {
    591 				return nil, b.createError(err)
    592 			}
    593 
    594 			fileID = sfResp.FileID
    595 		}
    596 
    597 		for {
    598 			if uploadURL == nil {
    599 				uuReq := &b2.GetUploadPartURLRequest{
    600 					Authorization: auth.authorizationToken,
    601 					BaseURL:       auth.apiURL,
    602 					FileID:        fileID,
    603 				}
    604 
    605 				uuResp, err := b.client.GetUploadPartURL(ctx, uuReq)
    606 				if err != nil {
    607 					return nil, b.createError(err)
    608 				}
    609 
    610 				authorization = uuResp.AuthorizationToken
    611 				uploadURL = uuResp.UploadURL.URL
    612 			}
    613 
    614 			hash := sha1.New()
    615 			hash.Write(request.Request.Data)
    616 			checksum := fmt.Sprintf("%x", hash.Sum(nil))
    617 
    618 			upReq := &b2.UploadPartRequest{
    619 				Authorization: authorization,
    620 				UploadURL:     uploadURL,
    621 				PartNumber:    part,
    622 				ContentLength: len(request.Request.Data),
    623 				ContentSHA1:   checksum,
    624 				Data:          io.NopCloser(bytes.NewReader(request.Request.Data)),
    625 			}
    626 
    627 			if _, err := b.client.UploadPart(ctx, upReq); err != nil {
    628 				if e, ok := err.(*b2.Status); ok && e.Status >= http.StatusInternalServerError && retryAttempt < uploadRetryBudget {
    629 					zap.L().Warn(
    630 						"chunked file upload failed with server error; retrying after delay",
    631 						zap.Int("attempt", retryAttempt),
    632 						zap.Duration("delay", retryDelay),
    633 						zap.Error(err),
    634 					)
    635 
    636 					select {
    637 					case <-ctx.Done():
    638 						return nil, status.FromContextError(ctx.Err()).Err()
    639 					case <-time.After(retryDelay):
    640 					}
    641 
    642 					// Explicitly reset the upload URL so that the retry also
    643 					// obtains a new upload URL before retrying the actual
    644 					// upload. Such client behavior is mandated by the B2 API.
    645 					// Reference: https://www.backblaze.com/b2/docs/uploading.html
    646 					uploadURL = nil
    647 
    648 					retryAttempt++
    649 					retryDelay *= 2 // Exponential backoff
    650 					continue
    651 				}
    652 
    653 				return nil, b.createError(err)
    654 			}
    655 
    656 			part++
    657 			checksums = append(checksums, checksum)
    658 			break
    659 		}
    660 	}
    661 
    662 	ffReq := &b2.FinishLargeFileRequest{
    663 		Authorization: auth.authorizationToken,
    664 		BaseURL:       auth.apiURL,
    665 		FileID:        fileID,
    666 		PartSHA1Array: checksums,
    667 	}
    668 
    669 	ffResp, err := b.client.FinishLargeFile(ctx, ffReq)
    670 	if err != nil {
    671 		return nil, b.createError(err)
    672 	}
    673 
    674 	attributes, err := schemas.UnmarshalAttributes(ffResp.FileInfo, nil)
    675 	if err != nil {
    676 		return nil, b.createError(err)
    677 	}
    678 
    679 	return &service.PutObjectStreamResponse{
    680 		Response: &service.PutObjectResponse{
    681 			Metadata: &common.Metadata{
    682 				Id:           ffResp.FileID,
    683 				Key:          ffResp.FileName,
    684 				Size:         uint64(ffResp.ContentLength),
    685 				ObjectType:   common.Object_REGULAR,
    686 				MimeType:     ffResp.ContentType,
    687 				ModifiedTime: timestamppb.New(ffResp.UploadTimestamp.Time),
    688 				Checksum:     ffResp.ContentSHA1,
    689 				Attributes:   attributes,
    690 			},
    691 		},
    692 	}, nil
    693 }

    695 // DeleteObject deletes the requested file.
    696 func (b *B2) DeleteObject(ctx context.Context, request *service.DeleteObjectRequest) (*service.DeleteObjectResponse, error) {
    697 	auth, err := b.authorizeAccount(ctx)
    698 	if err != nil {
    699 		return nil, b.createError(err)
    700 	}
    701 
    702 	lbReq := &b2.ListBucketsRequest{
    703 		Authorization: auth.authorizationToken,
    704 		BaseURL:       auth.apiURL,
    705 		AccountID:     auth.accountID,
    706 		BucketName:    request.Resource.Bucket,
    707 	}
    708 
    709 	lbResp, err := b.client.ListBuckets(ctx, lbReq)
    710 	if err != nil {
    711 		return nil, b.createError(err)
    712 	}
    713 
    714 	if len(lbResp.Buckets) != 1 {
    715 		return nil, status.Errorf(
    716 			codes.NotFound,
    717 			"backend: unexpected number of buckets matched by name: name=%s buckets=%d",
    718 			request.Resource.Bucket,
    719 			len(lbResp.Buckets),
    720 		)
    721 	}
    722 
    723 	lfReq := &b2.ListFileNamesRequest{
    724 		Authorization: auth.authorizationToken,
    725 		BaseURL:       auth.apiURL,
    726 		BucketID:      lbResp.Buckets[0].BucketID,
    727 		StartFileName: request.Key,
    728 		MaxFileCount:  1,
    729 		Prefix:        request.Key,
    730 		Delimiter:     "/",
    731 	}
    732 
    733 	if request.Recursive {
    734 		lfReq.Delimiter = ""
    735 		lfReq.MaxFileCount = 10000
    736 	}
    737 
    738 	lfResp, err := b.client.ListFileNames(ctx, lfReq)
    739 	if err != nil {
    740 		return nil, b.createError(err)
    741 	}
    742 
    743 	if len(lfResp.Files) == 0 {
    744 		return nil, status.Errorf(
    745 			codes.NotFound,
    746 			"backend: no files match requested name: bucket=%s name=%s",
    747 			request.Resource.Bucket,
    748 			request.Key,
    749 		)
    750 	} else if !request.Recursive && len(lfResp.Files) == 1 && lfResp.Files[0].FileID == "" {
    751 		// To emulate Unistore API semantics, disallow non-recursive deletions on non-empty
    752 		// directories. The B2 API represents this a single matched file with no file ID.
    753 		return nil, status.Errorf(
    754 			codes.FailedPrecondition,
    755 			"backend: cannot delete directory without recursive flag: bucket=%s name=%s",
    756 			request.Resource.Bucket,
    757 			request.Key,
    758 		)
    759 	} else if !request.Recursive && len(lfResp.Files) > 0 && strings.HasSuffix(request.Key, "/") {
    760 		// Directories can only exist in B2 if there are nonzero children (potentially
    761 		// nested arbitrarily deeply). Require such deletions to be recursive.
    762 		return nil, status.Errorf(
    763 			codes.FailedPrecondition,
    764 			"backend: cannot delete non-empty directory without recursive flag: bucket=%s name=%s children=%d",
    765 			request.Resource.Bucket,
    766 			request.Key,
    767 			len(lfResp.Files),
    768 		)
    769 	}
    770 
    771 	for _, file := range lfResp.Files {
    772 		dfReq := &b2.DeleteFileVersionRequest{
    773 			Authorization: auth.authorizationToken,
    774 			BaseURL:       auth.apiURL,
    775 			FileName:      file.FileName,
    776 			FileID:        file.FileID,
    777 		}
    778 
    779 		if _, err := b.client.DeleteFileVersion(ctx, dfReq); err != nil {
    780 			return nil, b.createError(err)
    781 		}
    782 	}
    783 
    784 	return &service.DeleteObjectResponse{Deleted: int32(len(lfResp.Files))}, nil
    785 }

    787 // ListBuckets retrieves metadata for all buckets.
    788 func (b *B2) ListBuckets(ctx context.Context, request *service.ListBucketsRequest) (*service.ListBucketsResponse, error) {
    789 	auth, err := b.authorizeAccount(ctx)
    790 	if err != nil {
    791 		return nil, b.createError(err)
    792 	}
    793 
    794 	lbReq := &b2.ListBucketsRequest{
    795 		Authorization: auth.authorizationToken,
    796 		BaseURL:       auth.apiURL,
    797 		AccountID:     auth.accountID,
    798 	}
    799 
    800 	lbResp, err := b.client.ListBuckets(ctx, lbReq)
    801 	if err != nil {
    802 		return nil, b.createError(err)
    803 	}
    804 
    805 	buckets := make([]*service.HeadBucketResponse, len(lbResp.Buckets))
    806 
    807 	for i, bucket := range lbResp.Buckets {
    808 		var visibility common.Visibility
    809 
    810 		switch bucket.BucketType {
    811 		case "allPublic":
    812 			visibility = common.Visibility_PUBLIC
    813 		case "allPrivate", "snapshot":
    814 			visibility = common.Visibility_PRIVATE
    815 		default:
    816 			visibility = common.Visibility_DEFAULT
    817 		}
    818 
    819 		lifecyclePolicies := make([]*common.LifecyclePolicy, len(bucket.LifecycleRules))
    820 
    821 		for j, rule := range bucket.LifecycleRules {
    822 			pattern := fmt.Sprintf("^%s", regexp.QuoteMeta(rule.FileNamePrefix))
    823 			ttl := time.Duration(rule.DaysFromUploadingToHiding+rule.DaysFromHidingToDeleting) * 24 * time.Hour
    824 
    825 			lifecyclePolicies[j] = &common.LifecyclePolicy{
    826 				Pattern:    pattern,
    827 				Expiration: durationpb.New(ttl),
    828 			}
    829 		}
    830 
    831 		buckets[i] = &service.HeadBucketResponse{
    832 			Bucket: &common.Bucket{
    833 				Id:                bucket.BucketID,
    834 				Name:              bucket.BucketName,
    835 				Visibility:        visibility,
    836 				LifecyclePolicies: lifecyclePolicies,
    837 				Tags:              bucket.BucketInfo,
    838 			},
    839 		}
    840 	}
    841 
    842 	return &service.ListBucketsResponse{Buckets: buckets}, nil
    843 }

    943 // authorizeAccount invokes the B2 account authorization API to obtain a short-lived authorization
    944 // token for subsequent API calls. Authentication state is cached local to the backend instance and
    945 // refreshed at a regular interval.
    946 func (b *B2) authorizeAccount(ctx context.Context) (*authenticationState, error) {
    947 	b.authMutex.Lock()
    948 	defer b.authMutex.Unlock()
    949 
    950 	// Authorization tokens are good for at most 24 hours after they have been issued by a call
    951 	// to b2_authorize_account. Somewhat conservatively, this opts to refresh the token if more
    952 	// than 3 hours have elapsed since the last successful refresh.
    953 	if time.Since(b.auth.refresh) < authenticationRefreshInterval {
    954 		return b.auth, nil
    955 	}
    956 
    957 	req := &b2.AuthorizeAccountRequest{
    958 		ApplicationKeyID: b.applicationKeyID,
    959 		ApplicationKey:   b.applicationKey,
    960 	}
    961 
    962 	resp, err := b.client.AuthorizeAccount(ctx, req)
    963 	if err != nil {
    964 		return nil, err
    965 	}
    966 
    967 	b.auth = &authenticationState{
    968 		accountID:          resp.AccountID,
    969 		authorizationToken: resp.AuthorizationToken,
    970 		apiURL:             resp.APIURL.URL,
    971 		downloadURL:        resp.DownloadURL.URL,
    972 		refresh:            time.Now(),
    973 	}
    974 
    975 	return b.auth, nil
    976 }

    978 // createError creates a semantically accurate gRPC status for an error status from the B2 API.
    979 func (b *B2) createError(err error) error {
    980 	b2Errors := map[int]codes.Code{
    981 		http.StatusBadRequest:                   codes.InvalidArgument,
    982 		http.StatusUnauthorized:                 codes.PermissionDenied,
    983 		http.StatusForbidden:                    codes.PermissionDenied,
    984 		http.StatusNotFound:                     codes.NotFound,
    985 		http.StatusMethodNotAllowed:             codes.InvalidArgument,
    986 		http.StatusRequestTimeout:               codes.DeadlineExceeded,
    987 		http.StatusRequestedRangeNotSatisfiable: codes.OutOfRange,
    988 		http.StatusTooManyRequests:              codes.ResourceExhausted,
    989 		http.StatusInternalServerError:          codes.Unknown,
    990 		http.StatusServiceUnavailable:           codes.Unavailable,
    991 	}
    992 
    993 	switch e := err.(type) {
    994 	case *b2.Status:
    995 		code, ok := b2Errors[e.Status]
    996 		if !ok {
    997 			code = codes.Unknown
    998 		}
    999 
   1000 		return status.Error(code, fmt.Sprintf("%s (code %s)", e.Message, e.Code))
   1001 	default:
   1002 		return status.Error(codes.Unknown, e.Error())
   1003 	}
   1004 }
