     27 // Server is the main Courier server entry point.

     29 	cfg     *config.Config
     30 	rp      http.Handler

     34 // New creates a new Courier server from the configuration.
     35 func New(cfg *config.Config) (*Server, error) {
     36 	rp, err := newReverseProxy(cfg.Server.VirtualHosts, cfg.Server.Proxy)
     37 	if err != nil {
     38 		return nil, &util.Error{
     39 			Namespace:    "server",
     40 			Message:      "failed to initialize reverse proxy",
     41 			StackedError: err,
     42 		}
     43 	}
     44 
     45 	return &Server{
     46 		cfg:     cfg,
     47 		rp:      rp,
     48 		servers: make([]*instance, len(cfg.Server.Listeners)),
     49 	}, nil
     50 }

     52 // Serve starts all individual server instances and serves indefinitely.
     53 func (s *Server) Serve() error {
     54 	errs := make(chan error)
     55 
     56 	for i, listener := range s.cfg.Server.Listeners {
     57 		s.servers[i] = newInstance(listener, s.rp)
     58 
     59 		go func(inst *instance) {
     60 			err := inst.serve()
     61 
     62 			// Ignore http.ErrServerClosed errors, since they are always returned when
     63 			// the underlying server is closed and not indicative of a real error during
     64 			// serving. Propagate all other errors or non-errors directly as-is.
     65 			if err == http.ErrServerClosed {
     66 				errs <- nil
     67 			} else {
     68 				errs <- err
     69 			}
     70 		}(s.servers[i])
     71 	}
     72 
     73 	return <-errs
     74 }
     75 
     76 // Close requests graceful shutdown of all server instances.
     77 func (s *Server) Close() error {
     78 	for _, inst := range s.servers {
     79 		if err := inst.close(); err != nil {
     80 			zap.L().Warn(
     81 				"error closing server instance",
     82 				zap.String("name", inst.cfg.Name),
     83 				zap.Stringer("address", inst.cfg.Address),
     84 				zap.Error(err),
     85 			)
     86 		}
     87 	}
     88 
     89 	return nil
     90 }

     98 // newInstance creates an instance wrapping a single http.Server for one listener specification.
     99 func newInstance(cfg *config.Listener, handler http.Handler) *instance {
    100 	var connStats sync.Map
    101 
    102 	connStats.Store(http.StateNew, new(int64))
    103 	connStats.Store(http.StateActive, new(int64))
    104 	connStats.Store(http.StateIdle, new(int64))
    105 	connStats.Store(http.StateHijacked, new(int64))
    106 	connStats.Store(http.StateClosed, new(int64))
    107 
    108 	handler = h2c.NewHandler(handler, &http2.Server{})
    109 	handler = log.NewRequestLogHandler(cfg.Name, handler)
    110 
    111 	server := &http.Server{
    112 		Handler:           handler,
    113 		ReadTimeout:       cfg.Connection.ReadTimeout,
    114 		ReadHeaderTimeout: cfg.Connection.ReadHeaderTimeout,
    115 		WriteTimeout:      cfg.Connection.WriteTimeout,
    116 		IdleTimeout:       cfg.Connection.IdleTimeout,
    117 		ErrorLog:          log.StdErrorLogger,
    118 		ConnState: func(conn net.Conn, state http.ConnState) {
    119 			tags := map[string]interface{}{
    120 				"listener": cfg.Name,
    121 				"state":    state.String(),
    122 			}
    123 
    124 			stateTotal, _ := connStats.Load(state)
    125 			atomic.AddInt64(stateTotal.(*int64), 1)
    126 
    127 			metrics.Client.Incr(metricConnectionStateTransition, tags)
    128 			metrics.Client.Gauge(
    129 				metricConnectionStateTotal,
    130 				float64(atomic.LoadInt64(stateTotal.(*int64))),
    131 				tags,
    132 			)
    133 		},
    134 		ConnContext: func(ctx context.Context, conn net.Conn) context.Context {
    135 			return context.WithValue(ctx, ctxLocalConn, conn)
    136 		},
    137 	}
    138 
    139 	return &instance{
    140 		cfg:    cfg,
    141 		server: server,
    142 	}
    143 }

    145 // serve configures and creates the listener and starts the underlying http.Server.
    146 func (i *instance) serve() error {
    147 	listenNet, listenAddr := i.cfg.Address.Address()
    148 	zap.L().Debug(
    149 		"created server listener",
    150 		zap.String("name", i.cfg.Name),
    151 		zap.String("net", listenNet),
    152 		zap.String("address", listenAddr),
    153 	)
    154 
    155 	ln, err := net.Listen(listenNet, listenAddr)
    156 	if err != nil {
    157 		return &util.Error{
    158 			Namespace: "server",
    159 			Message:   "failed to listen on address",
    160 			Tags: map[string]interface{}{
    161 				"name":    i.cfg.Name,
    162 				"address": i.cfg.Address.String(),
    163 			},
    164 			StackedError: err,
    165 		}
    166 	}
    167 
    168 	if i.cfg.Connection.ActiveLimit > 0 {
    169 		zap.L().Debug(
    170 			"enabling active connections limit for listener",
    171 			zap.String("name", i.cfg.Name),
    172 			zap.Int("limit", i.cfg.Connection.ActiveLimit),
    173 		)
    174 
    175 		ln = netutil.LimitListener(ln, i.cfg.Connection.ActiveLimit)
    176 	}
    177 
    178 	if len(i.cfg.AuthorizedSources) > 0 {
    179 		for _, source := range i.cfg.AuthorizedSources {
    180 			zap.L().Debug(
    181 				"enabling restricted IP source for listener",
    182 				zap.String("name", i.cfg.Name),
    183 				zap.Stringer("source", source),
    184 			)
    185 		}
    186 
    187 		ln = newRestrictedListener(i.cfg.Name, i.cfg.AuthorizedSources, ln)
    188 	}
    189 
    190 	switch i.cfg.Protocol {
    191 	case "haproxy":
    192 		zap.L().Debug(
    193 			"enabling proxy protocol header parser for listener",
    194 			zap.String("name", i.cfg.Name),
    195 		)
    196 
    197 		headerTimeout := i.cfg.Connection.ReadTimeout
    198 		if i.cfg.Connection.ReadHeaderTimeout > 0 {
    199 			headerTimeout = i.cfg.Connection.ReadHeaderTimeout
    200 		}
    201 
    202 		ln = newProxyListener(i.cfg.Name, headerTimeout, ln)
    203 	}
    204 
    205 	if i.cfg.TLSContext != nil {
    206 		zap.L().Debug(
    207 			"enabling server TLS termination",
    208 			zap.String("name", i.cfg.Name),
    209 			zap.Stringer("address", i.cfg.Address),
    210 			zap.Any("context", i.cfg.TLSContext),
    211 		)
    212 
    213 		i.server.TLSConfig, err = i.cfg.TLSContext.ServerConfig()
    214 		if err != nil {
    215 			return &util.Error{
    216 				Namespace: "server",
    217 				Message:   "failed to build server TLS context",
    218 				Tags: map[string]interface{}{
    219 					"name": i.cfg.Name,
    220 					"ctx":  i.cfg.TLSContext,
    221 				},
    222 				StackedError: err,
    223 			}
    224 		}
    225 
    226 		ln = tls.NewListener(ln, i.server.TLSConfig)
    227 
    228 		if i.cfg.TLSContext.HandshakeTimeout > 0 {
    229 			ln = newEagerTLSListener(i.cfg.Name, i.cfg.TLSContext.HandshakeTimeout, ln)
    230 		}
    231 	}
    232 
    233 	if err := http2.ConfigureServer(i.server, &http2.Server{}); err != nil {
    234 		return &util.Error{
    235 			Namespace: "server",
    236 			Message:   "error configuring HTTP/2 on server",
    237 			Tags: map[string]interface{}{
    238 				"name":    i.cfg.Name,
    239 				"address": i.cfg.Address.String(),
    240 			},
    241 			StackedError: err,
    242 		}
    243 	}
    244 
    245 	return i.server.Serve(ln)
    246 }
