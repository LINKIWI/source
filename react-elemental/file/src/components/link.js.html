      8 /**
      9  * Styled link element.
     10  */
     11 class Link extends Component {
     12   static propTypes = {
     13     type: PropTypes.oneOf(['regular', 'plain', 'underline']),
     14     href: PropTypes.string,
     15     activeColor: PropTypes.string,
     16     style: PropTypes.object,
     17     children: PropTypes.node,
     18     // HOC props
     19     handleMouseOver: PropTypes.func.isRequired,
     20     handleMouseOut: PropTypes.func.isRequired,
     21     handleMouseDown: PropTypes.func.isRequired,
     22     handleMouseUp: PropTypes.func.isRequired,
     23     handleFocus: PropTypes.func.isRequired,
     24     handleBlur: PropTypes.func.isRequired,
     25     isHover: PropTypes.bool.isRequired,
     26     isActive: PropTypes.bool.isRequired,
     27     isFocus: PropTypes.bool.isRequired,
     28     forwardedRef: PropTypes.oneOfType([
     29       PropTypes.shape({ current: PropTypes.instanceOf(Element) }),
     30       PropTypes.func,
     31     ]),
     32   };
     33 
     34   static defaultProps = {
     35     type: 'regular',
     36     href: '#',
     37     activeColor: 'currentColor',
     38     style: {},
     39     children: null,
     40     forwardedRef: null,
     41   };
     42 
     43   handleMouseOut = () => {
     44     const { handleMouseOut, handleMouseUp, handleBlur } = this.props;
     45 
     46     // Also simulate a mouseup and blur event so that the active state is properly reset if the user
     47     // moves the mouse outside of the container without releasing the mouse.
     48     handleMouseOut();
     49     handleMouseUp();
     50     handleBlur();
     51   };
     52 
     53   handleKeyDown = ({ keyCode }) => (keyCode === KEY_CODE_ENTER) && this.props.handleMouseDown();
     54 
     55   handleKeyUp = ({ keyCode }) => (keyCode === KEY_CODE_ENTER) && this.props.handleMouseUp();
     56 
     57   render() {
     58     const {
     59       type,
     60       activeColor,
     61       href,
     62       style: overrides,
     63       children,
     64       handleMouseOver,
     65       handleMouseDown,
     66       handleMouseUp,
     67       handleFocus,
     68       handleBlur,
     69       isHover,
     70       isActive,
     71       isFocus,
     72       forwardedRef,
     73       ...props
     74     } = this.props;
     75 
     76     const proxyProps = omit(props, ['handleMouseOut']);
     77 
     78     const borderSize = type === 'plain' ? '0' : '2px';
     79     const style = {
     80       color: 'currentColor',
     81       opacity: 0.8,
     82       textDecoration: 'none',
     83       ...transitionStyle('all', 'alpha'),
     84       ...(isHover || isFocus) && {
     85         borderBottom: `${borderSize} solid currentColor`,
     86         opacity: '1.0',
     87       },
     88       ...isActive && {
     89         borderBottom: `${borderSize} solid currentColor`,
     90         color: activeColor,
     91       },
     92       ...type === 'underline' && {
     93         borderBottom: '2px solid currentColor',
     94       },
     95       ...overrides,
     96     };
     97 
     98     return (
     99       <a
    100         ref={forwardedRef}
    101         href={href}
    102         style={style}
    103         onMouseOver={handleMouseOver}
    104         onMouseOut={this.handleMouseOut}
    105         onMouseDown={handleMouseDown}
    106         onMouseUp={handleMouseUp}
    107         onTouchStart={handleMouseDown}
    108         onTouchEnd={handleMouseUp}
    109         onFocus={handleFocus}
    110         onBlur={handleBlur}
    111         onKeyDown={this.handleKeyDown}
    112         onKeyUp={this.handleKeyUp}
    113         {...proxyProps}
    114       >
    115         {children}
    116       </a>
    117     );
    118   }
    119 }
    120 
    121 export default compose(
    122   withForwardedRef,
    123   withToggleState({ key: 'isHover', enable: 'handleMouseOver', disable: 'handleMouseOut' }),
    124   withToggleState({ key: 'isActive', enable: 'handleMouseDown', disable: 'handleMouseUp' }),
    125   withToggleState({ key: 'isFocus', enable: 'handleFocus', disable: 'handleBlur' }),
    126 )(Link);
