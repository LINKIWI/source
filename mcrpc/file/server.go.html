     37 // Serve starts the server on the specified listener.
     38 func (s *Server) Serve(ln net.Listener) error {
     39 	for {
     40 		conn, err := ln.Accept()
     41 		if err != nil {
     42 			if s.errorLog != nil {
     43 				s.errorLog.Printf(
     44 					"server: error accepting new connection on listener: err=%v",
     45 					err,
     46 				)
     47 			}
     48 
     49 			continue
     50 		}
     51 
     52 		if s.debugLog != nil {
     53 			s.debugLog.Printf(
     54 				"server: accepted client connection: laddr=%v raddr=%v",
     55 				conn.LocalAddr(),
     56 				conn.RemoteAddr(),
     57 			)
     58 		}
     59 
     60 		conn = &server.TimeoutConn{
     61 			ReadTimeout:  s.connReadTimeout,
     62 			WriteTimeout: s.connWriteTimeout,
     63 			Conn:         conn,
     64 		}
     65 
     66 		go s.handle(conn)
     67 	}
     68 }

     70 // handle processes the client connection.
     71 func (s *Server) handle(conn net.Conn) {
     72 	reader := protocol.NewReader(conn)
     73 
     74 	// Start a loop reading and processing requests from the client, in order to support
     75 	// connection reuse across multiple requests.
     76 	for {
     77 		if s.debugLog != nil {
     78 			s.debugLog.Printf(
     79 				"server: read-evaluation loop init: conn=%v",
     80 				conn.RemoteAddr(),
     81 			)
     82 		}
     83 
     84 		terminate, err := s.dispatch(conn, reader)
     85 		if err != nil && s.errorLog != nil {
     86 			s.errorLog.Printf("server: error handling connection: err=%v", err)
     87 		}
     88 
     89 		if terminate {
     90 			if s.debugLog != nil {
     91 				s.debugLog.Printf(
     92 					"server: terminating read-evaluation loop: conn=%v",
     93 					conn.RemoteAddr(),
     94 				)
     95 			}
     96 
     97 			conn.Close()
     98 			return
     99 		}
    100 	}
    101 }

    103 // dispatch parses the request and invokes the underlying service handler.
    104 func (s *Server) dispatch(conn net.Conn, reader *protocol.Reader) (bool, error) {
    105 	var req protocol.Request
    106 	var resp protocol.Response
    107 	var hErr error
    108 
    109 	buf, err := reader.ReadASCIICommand()
    110 	if err == io.EOF {
    111 		return true, nil
    112 	} else if err != nil {
    113 		return true, fmt.Errorf(
    114 			"server: error buffering command for parse: buf=%#v err=%v",
    115 			string(buf),
    116 			err,
    117 		)
    118 	}
    119 
    120 	req, err = protocol.NewASCIIParser().Parse(buf)
    121 	if err == protocol.ErrInvalidParse {
    122 		resp = &protocol.ClientError{Err: err}
    123 		if _, cErr := conn.Write([]byte(resp.String())); cErr != nil {
    124 			return true, cErr
    125 		}
    126 		return false, nil
    127 	} else if err != nil {
    128 		return true, fmt.Errorf(
    129 			"server: non-recoverable error during command parsing: err=%v",
    130 			err,
    131 		)
    132 	}
    133 
    134 	if s.debugLog != nil {
    135 		s.debugLog.Printf("server: parsed request: request=%#v", req)
    136 	}
    137 
    138 	ctx := context.WithValue(context.Background(), ClientConnContextKey, conn)
    139 	ctx = context.WithValue(ctx, ClientRawRequestContextKey, buf)
    140 
    141 	switch r := req.(type) {
    142 	case *protocol.VersionRequest:
    143 		resp, hErr = s.handler.Version(ctx, r)
    144 	case *protocol.ShutdownRequest:
    145 		resp, hErr = s.handler.Shutdown(ctx, r)
    146 	case *protocol.FlushRequest:
    147 		resp, hErr = s.handler.Flush(ctx, r)
    148 	case *protocol.QuitRequest:
    149 		resp, hErr = s.handler.Quit(ctx, r)
    150 	case *protocol.StatsRequest:
    151 		resp, hErr = s.handler.Stats(ctx, r)
    152 	case *protocol.WatchRequest:
    153 		resp, hErr = s.handler.Watch(ctx, r)
    154 	case *protocol.TouchRequest:
    155 		resp, hErr = s.handler.Touch(ctx, r)
    156 	case *protocol.DeleteRequest:
    157 		resp, hErr = s.handler.Delete(ctx, r)
    158 	case *protocol.IncrRequest:
    159 		resp, hErr = s.handler.Incr(ctx, r)
    160 	case *protocol.DecrRequest:
    161 		resp, hErr = s.handler.Decr(ctx, r)
    162 	case *protocol.GetRequest:
    163 		resp, hErr = s.handler.Get(ctx, r)
    164 	case *protocol.GetsRequest:
    165 		resp, hErr = s.handler.Gets(ctx, r)
    166 	case *protocol.GatRequest:
    167 		resp, hErr = s.handler.Gat(ctx, r)
    168 	case *protocol.GatsRequest:
    169 		resp, hErr = s.handler.Gats(ctx, r)
    170 	case *protocol.SetRequest:
    171 		resp, hErr = s.handler.Set(ctx, r)
    172 	case *protocol.AddRequest:
    173 		resp, hErr = s.handler.Add(ctx, r)
    174 	case *protocol.ReplaceRequest:
    175 		resp, hErr = s.handler.Replace(ctx, r)
    176 	case *protocol.AppendRequest:
    177 		resp, hErr = s.handler.Append(ctx, r)
    178 	case *protocol.PrependRequest:
    179 		resp, hErr = s.handler.Prepend(ctx, r)
    180 	case *protocol.CasRequest:
    181 		resp, hErr = s.handler.Cas(ctx, r)
    182 	default:
    183 		return false, fmt.Errorf("server: unsupported request type: request=%T", req)
    184 	}
    185 
    186 	if req.IsNoReply() {
    187 		return false, nil
    188 	}
    189 
    190 	if hErr != nil {
    191 		if s.debugLog != nil {
    192 			s.debugLog.Printf("server: handler error response: err=%v", hErr)
    193 		}
    194 
    195 		if _, cErr := conn.Write([]byte(hErr.Error())); cErr != nil {
    196 			return true, cErr
    197 		}
    198 	} else {
    199 		if s.debugLog != nil {
    200 			s.debugLog.Printf("server: handler response: response=%#v", resp)
    201 		}
    202 
    203 		if _, cErr := conn.Write([]byte(resp.String())); cErr != nil {
    204 			return true, cErr
    205 		}
    206 	}
    207 
    208 	return false, nil
    209 }
