     22 // New creates a configuration object from a path on disk.

     42 		return nil, &util.Error{
     43 			Namespace: "config",
     44 			Message:   "error deserializing config file contents",
     45 			Tags: map[string]interface{}{
     46 				"path": path,
     47 			},
     48 			StackedError: err,
     49 		}
     50 	}
     51 
     52 	if err := cfg.Validate(); err != nil {
     53 		return nil, &util.Error{
     54 			Namespace: "config",
     55 			Message:   "error validating config",
     56 			Tags: map[string]interface{}{
     57 				"path": path,

     60 		}
     61 	}

     64 }

    405 // ClientTransport creates a new http.RoundTripper transport derived from the upstream connection
    406 // configuration. Note that this method initializes new resources and the returned transport should
    407 // thus be reused across multiple connections for this upstream in order to avoid resource leaks.
    408 func (u *Upstream) ClientTransport() (http.RoundTripper, error) {
    409 	var err error
    410 	var tlsCfg *tls.Config
    411 
    412 	if u.TLSContext != nil {
    413 		zap.L().Debug(
    414 			"creating client TLS origination context",
    415 			zap.String("upstream", u.Address.String()),
    416 		)
    417 
    418 		tlsCfg, err = u.TLSContext.ClientConfig()
    419 		if err != nil {
    420 			return nil, &util.Error{
    421 				Namespace:    "config",
    422 				Message:      "error creating client TLS origination context",
    423 				StackedError: err,
    424 			}
    425 		}
    426 	}
    427 
    428 	switch u.Transport {
    429 	case "http2":
    430 		zap.L().Debug(
    431 			"creating client HTTP/2 transport",
    432 			zap.String("upstream", u.Address.String()),
    433 			zap.Stringer("proxy", u.Proxy),
    434 			zap.Any("connection", u.Connection),
    435 		)
    436 
    437 		return &http2.Transport{
    438 			AllowHTTP:       true,
    439 			TLSClientConfig: tlsCfg,
    440 			DialTLS: func(network string, addr string, cfg *tls.Config) (net.Conn, error) {
    441 				var dialer proxy.Dialer
    442 				var cancel context.CancelFunc
    443 
    444 				dialer = &net.Dialer{
    445 					Timeout:   u.Connection.ConnectTimeout,
    446 					KeepAlive: u.Connection.KeepaliveInterval,
    447 				}
    448 
    449 				if u.TLSContext != nil {
    450 					dialer = &tls.Dialer{
    451 						NetDialer: dialer.(*net.Dialer),
    452 						Config:    cfg,
    453 					}
    454 				}
    455 
    456 				if u.Proxy != nil {
    457 					dialer, err = proxy.FromURL(u.Proxy.URL, dialer)
    458 					if err != nil {
    459 						return nil, &util.Error{
    460 							Namespace: "config",
    461 							Message:   "error creating HTTP/2 proxy dialer",
    462 							Tags: map[string]interface{}{
    463 								"name":  u.Name,
    464 								"addr":  u.Address,
    465 								"proxy": u.Proxy,
    466 							},
    467 							StackedError: err,
    468 						}
    469 					}
    470 				}
    471 
    472 				network, addr = u.Address.Address()
    473 				conn, err := dialer.Dial(network, addr)
    474 				if err != nil {
    475 					return nil, &util.Error{
    476 						Namespace: "config",
    477 						Message:   "error dialing upstream address over HTTP/2",
    478 						Tags: map[string]interface{}{
    479 							"name": u.Name,
    480 							"addr": u.Address,
    481 						},
    482 						StackedError: err,
    483 					}
    484 				}
    485 
    486 				if u.TLSContext != nil {
    487 					ctx := context.Background()
    488 					if u.TLSContext.HandshakeTimeout > 0 {
    489 						ctx, cancel = context.WithTimeout(
    490 							context.Background(),
    491 							u.TLSContext.HandshakeTimeout,
    492 						)
    493 						defer cancel()
    494 					}
    495 
    496 					if err := conn.(*tls.Conn).HandshakeContext(ctx); err != nil {
    497 						return nil, &util.Error{
    498 							Namespace: "config",
    499 							Message:   "error during upstream TLS handshake",
    500 							Tags: map[string]interface{}{
    501 								"name": u.Name,
    502 								"addr": u.Address,
    503 							},
    504 							StackedError: err,
    505 						}
    506 					}
    507 				}
    508 
    509 				return newTimeoutConn(
    510 					u.Connection.ReadTimeout,
    511 					u.Connection.WriteTimeout,
    512 					conn,
    513 				), nil
    514 			},
    515 		}, nil
    516 	default:

    518 
    519 		zap.L().Debug(
    520 			"creating client HTTP/1.1 transport",
    521 			zap.String("upstream", u.Address.String()),
    522 			zap.Stringer("proxy", u.Proxy),
    523 			zap.Any("connection", u.Connection),
    524 		)
    525 
    526 		if u.TLSContext != nil {
    527 			tlsHandshakeTimeout = u.TLSContext.HandshakeTimeout
    528 		}
    529

    544 			DialContext: func(ctx context.Context, network string, addr string) (net.Conn, error) {

    592 // ClientConfig transforms the TLS context into a tls.Config appropriate for purposes of TLS
    593 // origination from a client.
    594 func (t *TLSContext) ClientConfig() (*tls.Config, error) {
    595 	var verifyConnection func(tls.ConnectionState) error
    596 	var minVersion uint16
    597 
    598 	cs, err := newCertificateStore(t, func() (*x509.CertPool, error) {
    599 		return x509.SystemCertPool()
    600 	})
    601 	if err != nil {
    602 		return nil, &util.Error{
    603 			Namespace:    "config",
    604 			Message:      "failed to initialize client certificate store",
    605 			StackedError: err,
    606 		}
    607 	}
    608 
    609 	if t.VerifyPeer != nil {
    610 		pv := &peerVerifier{
    611 			pattern: t.VerifyPeer,
    612 			roots:   cs.cas,
    613 		}
    614 
    615 		verifyConnection = pv.verifyServer
    616 	}
    617 
    618 	switch t.ProtocolVersion {
    619 	case "lenient":
    620 		minVersion = tls.VersionTLS11
    621 	case "standard":
    622 		minVersion = tls.VersionTLS12
    623 	case "modern":
    624 		minVersion = tls.VersionTLS13
    625 	default:
    626 		minVersion = tls.VersionTLS12
    627 	}
    628 
    629 	return &tls.Config{
    630 		GetClientCertificate: cs.getClientCertificate,
    631 		RootCAs:              cs.cas,
    632 		// Defer to the overridden connection verification in place of default validation
    633 		// when explicit peer verification is enabled.
    634 		InsecureSkipVerify: t.VerifyMode == "none" ||
    635 			t.VerifyMode == "relaxed" ||
    636 			t.VerifyPeer != nil,
    637 		MinVersion:         minVersion,
    638 		ClientSessionCache: tls.NewLRUClientSessionCache(128),
    639 		VerifyConnection:   verifyConnection,
    640 	}, nil
    641 }
    642 
    643 // ServerConfig transforms the TLS context into a tls.Config appropriate for purposes of TLS
    644 // termination at a server.
    645 func (t *TLSContext) ServerConfig() (*tls.Config, error) {
    646 	var verifyMode tls.ClientAuthType
    647 	var verifyConnection func(tls.ConnectionState) error
    648 	var minVersion uint16
    649 
    650 	// Explicitly use a nil (empty) seed CA certificate pool in order to use only the CAs
    651 	// specified in TLS configuration for client certificate authentication.
    652 	cs, err := newCertificateStore(t, nil)
    653 	if err != nil {
    654 		return nil, &util.Error{
    655 			Namespace:    "config",
    656 			Message:      "failed to initialize server certificate store",
    657 			StackedError: err,
    658 		}
    659 	}
    660 
    661 	switch t.VerifyMode {
    662 	case "none":
    663 		verifyMode = tls.NoClientCert
    664 	case "relaxed":
    665 		verifyMode = tls.VerifyClientCertIfGiven
    666 	case "strict":
    667 		verifyMode = tls.RequireAndVerifyClientCert
    668 	default:
    669 		verifyMode = tls.NoClientCert
    670 	}
    671 
    672 	if t.VerifyPeer != nil {
    673 		pv := &peerVerifier{
    674 			pattern: t.VerifyPeer,
    675 			roots:   cs.cas,
    676 		}
    677 
    678 		verifyConnection = pv.verifyClient
    679 	}
    680 
    681 	switch t.ProtocolVersion {
    682 	case "lenient":
    683 		minVersion = tls.VersionTLS11
    684 	case "standard":
    685 		minVersion = tls.VersionTLS12
    686 	case "modern":
    687 		minVersion = tls.VersionTLS13
    688 	default:
    689 		minVersion = tls.VersionTLS12
    690 	}
    691 
    692 	return &tls.Config{
    693 		GetCertificate:   cs.getServerCertificate,
    694 		ClientCAs:        cs.cas,
    695 		ClientAuth:       verifyMode,
    696 		MinVersion:       minVersion,
    697 		VerifyConnection: verifyConnection,
    698 	}, nil
    699 }
