+/**

+const CodeSnippet = ({

+  numMatches,

+import PropTypes from 'prop-types';
+import React, { Fragment, PureComponent } from 'react';
+import { Spacing } from 'react-elemental';
+import { compose } from '@linkiwi/hoc';
+import withClipboard from 'client/app/react/containers/hoc/with-clipboard';
+import withHashState from 'client/app/react/containers/hoc/with-hash-state';
+import withToast from 'client/app/react/containers/hoc/with-toast';
+import KeyboardListener from 'client/app/react/components/passive/keyboard-listener';
+import CodeSnippet from 'client/app/react/components/search/results/code-snippet';
+import { modulo } from 'client/app/util/number';
+import { background } from 'client/app/util/style/color';
+import { transition } from 'client/app/util/style/transition';
+
+/**
+ * List of results for matching code snippets.
+ */
+class CodeResultsContainer extends PureComponent {
+  static propTypes = {
+    snippets: PropTypes.arrayOf(PropTypes.shape({
+      repo: PropTypes.string.isRequired,
+      version: PropTypes.string.isRequired,
+      path: PropTypes.string.isRequired,
+      lines: PropTypes.array.isRequired,
+    })).isRequired,
+    repositories: PropTypes.object.isRequired,
+    // HOC props
+    hash: PropTypes.number,
+    setHash: PropTypes.func.isRequired,
+    toast: PropTypes.func.isRequired,
+    clipboard: PropTypes.shape({
+      read: PropTypes.func.isRequired,
+      write: PropTypes.func.isRequired,
+    }).isRequired,
+  };
+
+  static defaultProps = {
+    hash: null,
+  };
+
+  componentDidMount() {
+    const { hash } = this.props;
+
+    if (hash) {
+      this._scrollPermalink(hash);
+    }
+  }
+
+  componentDidUpdate(prevProps) {
+    if (this.props.hash !== prevProps.hash) {
+      this._scrollPermalink(this.props.hash);
+    }
+  }
+
+  navigateResults = this._navigateResults.bind(this);
+
+  handleClipboardCopy = this._handleClipboardCopy.bind(this);
+
+  _scrollPermalink(idx) {  // eslint-disable-line class-methods-use-this
+    const [node] = document.getElementsByName(`#R${idx}`);
+    if (node) {
+      node.scrollIntoView({ behavior: 'smooth', block: 'start' });
+    }
+  }
+
+  _navigateResults(delta) {
+    const { snippets, hash, setHash } = this.props;
+
+    return (evt) => {
+      // Don't allow results navigation when focused on the search field
+      if (evt.target.id !== 'search-query-field') {
+        if (hash === null) {
+          setHash(0);
+        } else {
+          // Bidirectionally cycle through all available snippets
+          setHash(modulo((hash) + delta, snippets.length));
+        }
+      }
+    };
+  }
+
+  _handleClipboardCopy(path) {
+    const { clipboard, toast } = this.props;
+
+    return () => clipboard.write(path, (err) => toast(err ?
+      'There was an error writing to the system clipboard.' :
+      `Copied to clipboard: ${path}`));
+  }
+
+  render() {
+    const { snippets, repositories, hash } = this.props;
+
+    return (
+      <Fragment>
+        <KeyboardListener
+          character="j"
+          handler={this.navigateResults(1)}
+        />
+        <KeyboardListener
+          character="k"
+          handler={this.navigateResults(-1)}
+        />
+
+        {snippets.map((snippet, idx) => {
+          const numMatches = snippet.lines.reduce((acc, line) =>
+            (line.bounds ? acc + 1 : acc), 0);
+
+          const style = hash === idx ? {
+            borderRadius: '3px',
+            boxShadow: `0 0 0 1.5px ${background.primary.GAMMA}`,
+            transition: transition.all.DEFAULT,
+          } : {};
+
+          return (
+            <Spacing
+              key={`${snippet.repo}-${snippet.path}`}
+              name={`#R${idx}`}
+              style={style}
+              bottom={idx < snippets.length - 1}
+            >
+              <CodeSnippet
+                repo={snippet.repo}
+                path={snippet.path}
+                version={snippet.version}
+                lines={snippet.lines}
+                urlPattern={repositories[snippet.repo].url}
+                numMatches={numMatches}
+                permalink={`#R${idx}`}
+                onClipboardCopy={this.handleClipboardCopy}
+              />
+            </Spacing>
+          );
+        })}
+      </Fragment>
+    );
+  }
+}
+
+export default compose(
+  withClipboard,
+  withToast,
+  withHashState({
+    serializer: (value) => `#R${value}`,
+    deserializer: (hash) => parseInt(hash.substr(2), 10),
+  }),
+)(CodeResultsContainer);

+import PropTypes from 'prop-types';
+import React, { Component, Fragment } from 'react';
+import { Alert, Spacing } from 'react-elemental';
+import { connect } from 'react-redux';
+import { CODE_INVALID_PARAMETERS } from 'supercharged/shared/constants/error';
+import CodeResultsContainer from 'client/app/react/containers/search/results/code-results';
+import RecentSearches from 'client/app/react/components/search/history/recent-searches';
+import SearchStats from 'client/app/react/components/search/meta/stats';
+import FileResults from 'client/app/react/components/search/results/file-results';
+
+/**
+ * Logic container for rendering of both file and code search results.
+ */
+class SearchResultsContainer extends Component {
+  static propTypes = {
+    results: PropTypes.shape({
+      message: PropTypes.object,
+    }).isRequired,
+    onQueryChange: PropTypes.func.isRequired,
+    // HOC props
+    repositories: PropTypes.object.isRequired,
+    history: PropTypes.array.isRequired,
+  };
+
+  shouldComponentUpdate(nextProps) {
+    return nextProps.results.message !== this.props.results.message;
+  }
+
+  render() {
+    const {
+      results: { message },
+      repositories,
+      history,
+      onQueryChange,
+    } = this.props;
+    const { err, data: scData } = message || {};
+    const data = scData || { files: [], code: [], stats: { time: 0, exitReason: 0 } };
+
+    // A request/response transaction has not yet been initiated, or the client supplied an empty
+    // search query
+    if (!message || (err && err.code === CODE_INVALID_PARAMETERS)) {
+      return history.length ? (
+        <div style={{ alignItems: 'center', display: 'flex', flexDirection: 'column' }}>
+          <RecentSearches
+            history={history}
+            onHistoricalSearch={(query) => onQueryChange({ target: { value: query } })}
+          />
+        </div>
+      ) : null;
+    }
+
+    if (err) {
+      return (
+        <Alert
+          type="error"
+          title="Error"
+          message={err.message}
+        />
+      );
+    }
+
+    return (
+      <Fragment>
+        <Spacing size="large" bottom>
+          <SearchStats
+            numPaths={data.files.length}
+            numFiles={data.code.length}
+            time={data.stats.time}
+            exitReason={data.stats.exitReason}
+          />
+        </Spacing>
+
+        <Spacing size="large" bottom>
+          <FileResults
+            files={data.files}
+            repositories={repositories}
+          />
+        </Spacing>
+
+        <CodeResultsContainer
+          snippets={data.code}
+          repositories={repositories}
+        />
+      </Fragment>
+    );
+  }
+}
+
+const mapStateToProps = ({ meta, search }) => ({
+  repositories: meta.repositories,
+  history: search.history,
+});
+
+export default connect(mapStateToProps)(SearchResultsContainer);
