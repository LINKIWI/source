     37 // Get performs a non-transactional lookup of the requested key.
     38 func (t *TiKVHandler) Get(ctx context.Context, request *protocol.GetRequest) (*protocol.GetResponse, error) {
     39 	kvctx, cancel := context.WithTimeout(ctx, kvTimeout)
     40 	defer cancel()
     41 
     42 	keys := make([][]byte, len(request.Keys))
     43 	for idx, rawKey := range request.Keys {
     44 		key, err := schema.NewKey(rawKey)
     45 		if err != nil {
     46 			return nil, &protocol.ServerError{Err: err}
     47 		}
     48 
     49 		keyb, err := key.MarshalBinary()
     50 		if err != nil {
     51 			return nil, &protocol.ServerError{Err: err}
     52 		}
     53 
     54 		keys[idx] = keyb
     55 	}
     56 
     57 	values, err := t.kv.BatchGet(kvctx, keys)
     58 	if err != nil {
     59 		return nil, &protocol.ServerError{Err: err}
     60 	}
     61 
     62 	retrievals := make([]*protocol.Retrieval, 0, len(request.Keys))
     63 	for idx, rawValue := range values {
     64 		if len(rawValue) == 0 {
     65 			// Key not found; skip this retrieval data block.
     66 			continue
     67 		}
     68 
     69 		value := &schema.Value{}
     70 		if err := value.UnmarshalBinary(rawValue); err != nil {
     71 			return nil, &protocol.ServerError{Err: err}
     72 		}
     73 
     74 		retrievals = append(retrievals, &protocol.Retrieval{
     75 			Key:       request.Keys[idx],
     76 			Flags:     value.Flags,
     77 			Size:      len(value.Data),
     78 			CasUnique: value.CasUnique,
     79 			Data:      value.Data,
     80 		})
     81 	}
     82 
     83 	return &protocol.GetResponse{Values: retrievals}, nil
     84 }

     86 // Set performs a non-transactional write of a key-value pair, optionally with a finite TTL.
     87 func (t *TiKVHandler) Set(ctx context.Context, request *protocol.SetRequest) (*protocol.SetResponse, error) {
     88 	kvctx, cancel := context.WithTimeout(ctx, kvTimeout)
     89 	defer cancel()
     90 
     91 	key, err := schema.NewKey(request.Payload.Key)
     92 	if err != nil {
     93 		return nil, &protocol.ServerError{Err: err}
     94 	}
     95 
     96 	keyb, err := key.MarshalBinary()
     97 	if err != nil {
     98 		return nil, &protocol.ServerError{Err: err}
     99 	}
    100 
    101 	value, err := schema.NewValue(
    102 		schema.Simple,
    103 		request.Payload.Flags,
    104 		uint32(request.Payload.Expiration.Seconds()),
    105 		request.Payload.CasUnique,
    106 		request.Payload.Data,
    107 	)
    108 	if err != nil {
    109 		return nil, &protocol.ServerError{Err: err}
    110 	}
    111 
    112 	valueb, err := value.MarshalBinary()
    113 	if err != nil {
    114 		return nil, &protocol.ServerError{Err: err}
    115 	}
    116 
    117 	if request.Payload.Expiration > 0 {
    118 		if err := t.kv.PutWithTTL(kvctx, keyb, valueb, uint64(request.Payload.Expiration.Seconds())); err != nil {
    119 			return nil, &protocol.ServerError{Err: err}
    120 		}
    121 	} else {
    122 		if err := t.kv.Put(kvctx, keyb, valueb); err != nil {
    123 			return nil, &protocol.ServerError{Err: err}
    124 		}
    125 	}
    126 
    127 	return &protocol.SetResponse{Status: protocol.Stored}, nil
    128 }
