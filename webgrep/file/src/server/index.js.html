      1 import Express from 'express';
      2 import { supercharge } from 'supercharged/server';
      3 import path from 'path';
      4 import yargs from 'yargs';
      5 import * as Sentry from '@sentry/node';
      6 import Context from 'server/context';
      7 import handlers from 'server/handlers';
      8 
      9 const { VERSION } = process.env;
     10 
     11 const main = () => {
     12   const params = yargs
     13     .usage('$0 [options]', 'webgrep: web frontend for Livegrep code search')
     14     .option('config', {
     15       alias: 'c',
     16       default: 'config.yaml',
     17       description: 'Path to the configuration file on disk',
     18     })
     19     .option('verbosity', {
     20       alias: 'v',
     21       default: 'info',
     22       description: 'Log verbosity level',
     23       choices: ['error', 'warn', 'info', 'debug'],
     24     })
     25     .version(VERSION)
     26     .argv;
     27 
     28   const app = Express();
     29   const ctx = new Context({
     30     config: params.config,
     31     verbosity: params.verbosity,
     32   });
     33 
     34   ctx.log.info(
     35     'main: starting webgrep: version=%s config=%s verbosity=%s',
     36     VERSION,
     37     params.config,
     38     params.verbosity,
     39   );
     40 
     41   const sentryDSN = ctx.config.get('server.sentry_dsn');
     42   if (sentryDSN) {
     43     Sentry.init({ dsn: sentryDSN });
     44     Sentry.configureScope((scope) => {
     45       scope.setTag('version', VERSION);
     46     });
     47     ctx.log.debug('main: enabled Sentry reporting: dsn=%s', sentryDSN);
     48   }
     49 
     50   app.use(Sentry.Handlers.requestHandler());
     51   app.use('/assets', Express.static(path.join(__dirname, '../../dist/client')));
     52   supercharge(app, handlers, {
     53     createHandler: (HandlerClass) => (...args) => new HandlerClass(...args, ctx),
     54     ws: { perMessageDeflate: true },
     55   });
     56   app.use(Sentry.Handlers.errorHandler());
     57 
     58   const transport = ctx.config.get('server.listen.transport');
     59   const address = ctx.config.get('server.listen.address');
     60   switch (transport) {
     61     case 'tcp': {
     62       const [host, port] = address.split(':', 2);
     63       ctx.log.info('main: serving indefinitely over TCP: host=%s port=%d', host, port);
     64       app.listen(port, host);
     65       return 0;
     66     }
     67     case 'unix':
     68       ctx.log.info('main: serving indefinitely over Unix domain socket: socket=%s', address);
     69       app.listen(address);
     70       return 0;
     71     default:
     72       ctx.log.error('main: unsupported server listen transport: transport=%s', transport);
     73       return 1;
     74   }
     75 };
     76 
     77 const code = main();
     78 if (code) {
     79   process.exit(code);
     80 }
