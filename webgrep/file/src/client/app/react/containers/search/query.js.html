     10 /**
     11  * Wrapper for search query concerns, including the query entry itself and associated query options.
     12  */
     13 class SearchQueryContainer extends Component {
     14   static propTypes = {
     15     value: PropTypes.string.isRequired,
     16     regex: PropTypes.bool.isRequired,
     17     caseSensitive: PropTypes.bool.isRequired,
     18     maxMatches: PropTypes.number.isRequired,
     19     filePath: PropTypes.string.isRequired,
     20     repositories: PropTypes.array.isRequired,
     21     results: PropTypes.shape({
     22       err: PropTypes.object,
     23       data: PropTypes.object,
     24     }),
     25     onQueryChange: PropTypes.func.isRequired,
     26     onRegexChange: PropTypes.func.isRequired,
     27     onCaseSensitivityChange: PropTypes.func.isRequired,
     28     onMaxMatchesChange: PropTypes.func.isRequired,
     29     onFilePathChange: PropTypes.func.isRequired,
     30     onRepositoriesChange: PropTypes.func.isRequired,
     31     onQueryRecord: PropTypes.func.isRequired,
     32     connectivity: PropTypes.string.isRequired,
     33     isCompact: PropTypes.bool.isRequired,
     34   };
     35 
     36   static defaultProps = {
     37     results: null,
     38   };
     39 
     40   handleQueryFieldBlur = this._handleQueryFieldBlur.bind(this);
     41 
     42   handleRepositoryToggle = this._handleRepositoryToggle.bind(this);
     43 
     44   _handleQueryFieldBlur(evt) {
     45     if (evt.target.value) {
     46       this.props.onQueryRecord(evt.target.value);
     47     }
     48   }
     49 
     50   _handleRepositoryToggle(toggled) {
     51     const { repositories, onRepositoriesChange } = this.props;
     52 
     53     // Reset filter list entirely
     54     if (!toggled) {
     55       return onRepositoriesChange([]);
     56     }
     57 
     58     const filteredRepos = repositories
     59       .filter((repo) => (repo.name === toggled.name ? !repo.isSelected : repo.isSelected))
     60       .map((repo) => repo.name);
     61 
     62     return onRepositoriesChange(filteredRepos);
     63   }
     64 
     65   render() {
     66     const {
     67       value,
     68       regex,
     69       caseSensitive,
     70       maxMatches,
     71       filePath,
     72       repositories,
     73       results,
     74       onQueryChange,
     75       onRegexChange,
     76       onCaseSensitivityChange,
     77       onMaxMatchesChange,
     78       onFilePathChange,
     79       connectivity,
     80       isCompact,
     81     } = this.props;
     82 
     83     const searchResults = [
     84       ...objLookup(results, ['data', 'files']) || [],
     85       ...objLookup(results, ['data', 'code']) || [],
     86     ];
     87 
     88     return (
     89       <Fragment>
     90         <Spacing size="20px" bottom>
     91           <QueryField
     92             value={value}
     93             onChange={onQueryChange}
     94             onBlur={this.handleQueryFieldBlur}
     95           />
     96         </Spacing>
     97 
     98         <div style={{ alignItems: 'center', display: 'flex', justifyContent: 'space-between' }}>
     99           <div style={{ flexGrow: 1 }}>
    100             <QueryOptionControls
    101               regex={regex}
    102               caseSensitive={caseSensitive}
    103               maxMatches={maxMatches}
    104               filePath={filePath}
    105               repositories={repositories}
    106               searchResults={searchResults}
    107               onRegexChange={onRegexChange}
    108               onCaseSensitivityChange={onCaseSensitivityChange}
    109               onMaxMatchesChange={onMaxMatchesChange}
    110               onFilePathChange={onFilePathChange}
    111               onRepositoryToggle={this.handleRepositoryToggle}
    112               isCompact={isCompact}
    113             />
    114           </div>
    115 
    116           {!isCompact && (
    117             <Spacing style={{ display: 'inherit' }} left>
    118               <ConnectionStatus connectivity={connectivity} />
    119             </Spacing>
    120           )}
    121         </div>
    122       </Fragment>
    123     );
    124   }
    125 }
    126 
    127 const mapStateToProps = ({ meta }, { filteredRepos }) => ({
    128   repositories: Object.values(meta.repositories).map((repo) => ({

    130     remote: repo.remote,
    131     isSelected: filteredRepos.includes(repo.name),
    132   })),

    135 export default connect(mapStateToProps)(SearchQueryContainer);
