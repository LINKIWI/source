      3 /**
      4  * Logic for reading raw source code.
      5  */
      6 export default class SourceLogic extends BaseLogic {
      7   /**
      8    * Read a source file contents, with transparent caching.
      9    *
     10    * @param {String} repo Repository identifier.
     11    * @param {String} version Git ref or commit-ish at which the file should be fetched.
     12    * @param {String} file Path to the file to read.
     13    * @param {Function} cb Callback invoked with (err, resp) on completion.
     14    */
     15   read(repo, version, file, cb) {
     16     this.ctx.log.debug(
     17       'source: serving file preview: repo=%s version=%s file=%s',
     18       repo,
     19       version,
     20       file,
     21     );
     22 
     23     const transaction = this.ctx.cache.transaction(
     24       'source',
     25       'read',
     26       {
     27         repo,
     28         revision: version,
     29         file,
     30         version: process.env.VERSION,
     31       },
     32       JSON.stringify,
     33       JSON.parse,
     34     );
     35 
     36     return transaction.get((cacheErr, cached) => {
     37       if (!cacheErr && cached) {
     38         return cb(null, cached);
     39       }
     40 
     41       return this.ctx.source.read(repo, version, file, (err, contents) => {
     42         if (err) {
     43           return cb({
     44             status: 500,
     45             message: err.toString(),
     46           });
     47         }
     48 
     49         const resp = { data: contents };
     50 
     51         transaction.set(resp);
     52 
     53         return cb(null, resp);
     54       });
     55     });
     56   }
