     10 import MetaLogic from 'server/logic/meta';

     15 /**
     16  * Format a fully qualified path to a schemas file.
     17  *
     18  * @param {String} schema Name of the schemas filename.
     19  * @return {String} Fully qualified path to the schemas path.
     20  */
     21 const schemasPath = (schema) => path.join(__dirname, `../shared/schemas/${schema}`);

     23 /**
     24  * Server-side context shared by all incoming request handlers.
     25  */
     26 export default class Context {
     27   constructor(options) {
     28     this.config = new ConfigClient(options.config);
     29 
     30     this.metrics = new MetricsClient(
     31       (() => {
     32         const backends = Object.keys(this.config.get('server.metrics') || {});
     33         if (!backends.length) {
     34           return new NoopMetricsClient();
     35         }
     36 
     37         const [backend] = backends;
     38 
     39         switch (backend) {
     40           case 'statsd':
     41             return new StatsdMetricsClient(
     42               this.config.get('server.metrics.statsd.addr') || '',
     43               this.config.get('server.metrics.statsd.prefix') || 'webgrep',
     44               this.config.get('server.metrics.statsd.sample_rate') || 1.0,
     45             );
     46           default:
     47             return new NoopMetricsClient();
     48         }
     49       })(),
     50       { host: os.hostname(), version: process.env.VERSION },
     51     );
     52 
     53     this.cache = (() => {
     54       const noop = new CacheClient(
     55         new NoopCache(),
     56         '',
     57         0,
     58         new MetricsClient(new NoopMetricsClient(), {}),
     59       );
     60 
     61       const backends = Object.keys(this.config.get('server.cache') || {});
     62       if (!backends.length) {
     63         return noop;
     64       }
     65 
     66       const [backend] = backends;
     67 
     68       switch (backend) {
     69         case 'redis':
     70           return new CacheClient(
     71             new RedisCache(
     72               this.config.get('server.cache.redis.socket'),
     73               this.config.get('server.cache.redis.timeout') || 0,
     74             ),
     75             this.config.get('server.cache.redis.prefix') || 'webgrep',
     76             this.config.get('server.cache.redis.ttl') || 10 * 60 * 1000,
     77             this.metrics,
     78           );
     79         case 'memory':
     80           return new CacheClient(
     81             new MemoryCache(),
     82             '',
     83             this.config.get('server.cache.memory.ttl') || 10 * 60 * 1000,
     84             this.metrics,
     85           );
     86         default:
     87           return noop;
     88       }
     89     })();
     90 
     91     this.service = {
     92       codesearch: (() => {
     93         const backends = this.config.get('server.livegrep.servers').map((server) => new GRPCClient(
     94           server.address,
     95           server.authority || '',
     96           'CodeSearch',
     97           schemasPath('livegrep.proto'),
     98           this.metrics,
     99           { requestTimeout: this.config.get('server.livegrep.request_timeout') || 3000 },
    100         ));
    101 
    102         switch (this.config.get('server.livegrep.load_balancing_policy')) {
    103           case 'round-robin':
    104             return new RoundRobinGRPCLoadBalancer(backends);
    105           case 'failover':
    106             return new FailoverGRPCLoadBalancer(backends);
    107           default:
    108             return new StaticGRPCLoadBalancer(backends);
    109         }
    110       })(),
    111     };
    112 
    113     this.source = (() => {
    114       const noop = new SourceClient(
    115         new NoopSourceBackend(),
    116         new MetricsClient(new NoopMetricsClient(), {}),
    117       );
    118 
    119       const backends = Object.keys(this.config.get('server.source') || {});
    120       if (!backends.length) {
    121         return noop;
    122       }
    123 
    124       const [backend] = backends;
    125 
    126       switch (backend) {
    127         case 'local':
    128           return new SourceClient(
    129             new LocalSourceBackend(this.config.get('server.source.local.repositories') || []),
    130             this.metrics,
    131           );
    132         case 'gitlab':
    133           return new SourceClient(
    134             new GitlabSourceBackend(
    135               this.config.get('server.source.gitlab.base_url') || 'https://gitlab.com',
    136               this.config.get('server.source.gitlab.socket_path'),
    137               this.config.get('server.source.gitlab.access_token'),
    138               {
    139                 key: this.config.get('server.source.gitlab.tls_key'),
    140                 cert: this.config.get('server.source.gitlab.tls_cert'),
    141                 ca: this.config.get('server.source.gitlab.tls_ca_cert'),
    142               },
    143             ),
    144             this.metrics,
    145           );
    146         default:
    147           return noop;
    148       }
    149     })();
    150 
    151     this.logic = {
    152       admin: new AdminLogic(this),
    153       meta: new MetaLogic(this),
    154       search: new SearchLogic(this),
    155       source: new SourceLogic(this),
    156       view: new ViewLogic(this),
    157     };
    158 
    159     this.log = winston.createLogger({
    160       level: options.verbosity,
    161       format: winston.format.combine(
    162         winston.format.timestamp(),
    163         winston.format.splat(),
    164         winston.format.printf((fmt) =>
    165           `${fmt.timestamp} ${fmt.level.toUpperCase()}\t${fmt.message}`),
    166       ),
    167       transports: [
    168         new winston.transports.Console({
    169           timestamp: () => Date.now(),
    170         }),
    171       ],
    172     });
    173   }
    174 }
