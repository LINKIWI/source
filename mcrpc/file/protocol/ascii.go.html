    430 func (s *Storage) String() string {
    431 	exptime := fmt.Sprintf("%.f", s.Expiration.Seconds())
    432 
    433 	switch {
    434 	case s.CasUnique != 0 && s.NoReply:
    435 		return fmt.Sprintf(
    436 			"%s %d %s %d %d noreply\r\n%s\r\n",
    437 			s.Key,
    438 			s.Flags,
    439 			exptime,
    440 			s.Size,
    441 			s.CasUnique,
    442 			string(s.Data),
    443 		)
    444 	case s.CasUnique != 0 && !s.NoReply:
    445 		return fmt.Sprintf(
    446 			"%s %d %s %d %d\r\n%s\r\n",
    447 			s.Key,
    448 			s.Flags,
    449 			exptime,
    450 			s.Size,
    451 			s.CasUnique,
    452 			string(s.Data),
    453 		)
    454 	case s.CasUnique == 0 && s.NoReply:
    455 		return fmt.Sprintf(
    456 			"%s %d %s %d noreply\r\n%s\r\n",
    457 			s.Key,
    458 			s.Flags,
    459 			exptime,
    460 			s.Size,
    461 			string(s.Data),
    462 		)
    463 	default:
    464 		return fmt.Sprintf(
    465 			"%s %d %s %d\r\n%s\r\n",
    466 			s.Key,
    467 			s.Flags,
    468 			exptime,
