      7 // Buffered is an abstraction over a Transport that internally buffers the transmission of
      8 // individual payloads with a configurable batch size.
      9 type Buffered struct {
     10 	buf chan []byte
     11 	Transport
     12 }
     13 
     14 // NewBuffered creates a new buffered transport over an existing Transport backend.
     15 func NewBuffered(backendFactory Factory, size int) (Transport, error) {
     16 	if size <= 0 {
     17 		return nil, errors.New(
     18 			"transport",
     19 			"buffered transport batch size must be greater than 0",
     20 		)
     21 	}
     22 
     23 	tport, err := backendFactory()
     24 	if err != nil {
     25 		return nil, err
     26 	}
     27 
     28 	return &Buffered{
     29 		Transport: tport,
     30 		buf:       make(chan []byte, size),
     31 	}, nil
     32 }
     33 
     34 // Send asynchronously queues the payload for transmission. If the queue size has reached the batch
     35 // size, all queued payloads are flushed.
     36 func (t *Buffered) Send(data []byte) (int, error) {
     37 	t.buf <- data
     38 
     39 	if len(t.buf) == cap(t.buf) {
     40 		t.flush()
     41 	}
     42 
     43 	return len(data), nil
     44 }
     45 
     46 // Close flushes all remaining queued payloads and closes the backing transport.
     47 func (t *Buffered) Close() error {
     48 	t.flush()
     49 
     50 	return t.Transport.Close()
     51 }
     52 
     53 // flush sends all queued payloads, ignoring all errors.
     54 func (t *Buffered) flush() {
     55 	elements := len(t.buf)
     56 
     57 	for i := 0; i < elements; i++ {
     58 		t.Transport.Send(<-t.buf)
     59 	}
     60 }
