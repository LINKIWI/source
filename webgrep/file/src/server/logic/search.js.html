      3 import { escapeRegex } from 'server/util/format';

      6 /**
      7  * Logic for executing index searches.
      8  */
      9 export default class SearchLogic extends BaseLogic {
     10   /**
     11    * Execute a search against the livegrep codesearch backend.
     12    *
     13    * @param {Object} params Object of search parameters.
     14    * @param {Function} cb Callback invoked with (err, resp) on completion.
     15    */
     16   executeSearch(params, cb) {
     17     const req = {
     18       line: params.regex ? params.query : escapeRegex(params.query),
     19       file: params.regex ? params.file : escapeRegex(params.file),
     20       // Manually construct a regular expression for a logical disjunction among all repositories
     21       repo: params.repos.map((repo) => `^${escapeRegex(repo)}$`).join('|'),
     22       foldCase: !params.caseSensitive,
     23       maxMatches: params.maxMatches,
     24       contextLines: params.context,
     25     };
     26 
     27     this.ctx.log.debug(
     28       'search: serving query: line=%s file=%s repos=%s case=%s matches=%d context=%d',
     29       req.line,
     30       params.file || '(none)',
     31       params.repos.length ? params.repos : '(all)',
     32       params.caseSensitive,
     33       params.maxMatches,
     34       params.context,
     35     );
     36 
     37     const transaction = this.ctx.cache.transaction(
     38       'search',
     39       'query-results',
     40       {
     41         ...req,
     42         // Encode the current server-side application version into the cache key for robustness
     43         // against potential key schema changes across different versions
     44         version: process.env.VERSION,
     45         // Encode the unique index identity into the cache key for robustness against stale values
     46         // (within TTL) after the server index is rolled
     47         ...params.indexIdentity && {
     48           indexName: params.indexIdentity.name,
     49           indexTimestamp: params.indexIdentity.timestamp,
     50         },
     51       },
     52       JSON.stringify,
     53       JSON.parse,
     54     );
     55 
     56     return transaction.get((cacheErr, cached) => {
     57       if (!cacheErr && cached) {
     58         return cb(null, cached);
     59       }
     60 
     61       return this.ctx.service.codesearch.rpc('search', req, (rpcErr, data) => {
     62         if (rpcErr) {
     63           this.ctx.log.error(
     64             'search: encountered RPC error: method=search code=%d details=%s request=%s',
     65             rpcErr.code,
     66             rpcErr.details,
     67             JSON.stringify(req),
     68           );
     69           return cb(this.formatErr(rpcErr));
     70         }
     71 
     72         // Treat early terminations due to server-side timeout as fatal errors
     73         const { SearchStats: { ExitReason } } = this.ctx.service.codesearch.proto;
     74         if (data.stats.exitReason === ExitReason.TIMEOUT) {
     75           this.ctx.log.error(
     76             'search: exceeded livegrep search timeout: request=%s',
     77             JSON.stringify(req),
     78           );
     79           return cb({
     80             code: ERROR_CODES.SEARCH_TIMEOUT,
     81             message: 'Server-side search timeout exceeded',
     82           });
     83         }
     84 
     85         const { code, stats, files } = this._reshapeResults(data);
     86 
     87         this.ctx.metrics.gauge('gauge.search.code_results', data.results.length);
     88         this.ctx.metrics.gauge('gauge.search.path_results', data.fileResults.length);
     89         this.ctx.metrics.gauge('gauge.search.file_results', code.length);
     90         this.ctx.metrics.timing(
     91           'latency.search.exec',
     92           stats.time,
     93           { exit: invertMap(ExitReason)[stats.exitReason] },
     94         );
     95 
     96         const resp = {
     97           data: {
     98             stats,
     99             code,
    100             files,
    101           },
    102         };
    103 
    104         // Only write to the cache if the search result was produced by the same index identity as
    105         // that specified in the request. Otherwise, this would pollute the cache with data from the
    106         // wrong index for this request.
    107         // Note that a side effect of this logic is that requests without a supplied index identity
    108         // will never write to cache, in order to avoid potential index inconsistency.
    109         const isCacheWriteSafe = params.indexIdentity &&
    110           data.indexName === params.indexIdentity.name &&
    111           parseInt(data.indexTime, 10) === params.indexIdentity.timestamp;
    112 
    113         if (isCacheWriteSafe) {
    114           transaction.set(resp);
    115         }
    116 
    117         return cb(null, resp);
    118       });
    119     });
    120   }
    121 
    122   /**
    123    * Massage the response from livegrep into a form that can be more easily interpreted by the
    124    * webgrep frontend.
    125    *
    126    * @param {Object} data Raw response payload from livegrep.
    127    * @return {Object} Object of code and file results, and search stats.
    128    * @private
    129    */
    130   _reshapeResults(data) {  // eslint-disable-line class-methods-use-this
    131     const code = Object.values(data.results
    132       // Aggregate lines by repo and path, so that each unique (repo, path) combination is
    133       // described by an array of all matching lines and the left/right bounds for each line.
    134       .reduce((aggregated, result) => {
    135         // Aggregation key: combine all results for the same file in the same repo
    136         const key = `${result.tree}-${result.path}`;
    137         // Line number of the matching line
    138         const lineNumber = parseInt(result.lineNumber, 10);
    139         // The existing entry for this repo/path combination, if it exists
    140         const existing = aggregated[key] || {};
    141 
    142         // Create a map of line numbers -> { bounds description, line }, being careful not to
    143         // override the bounds if they have already been specified. Since context lines are
    144         // overlapping, it's possible that a context line does not have a bounds description,
    145         // but it has one from an earlier result.
    146         const contextLines = [
    147           ...result.contextBefore.reverse(),
    148           result.line,
    149           ...result.contextAfter,
    150         ].reduce((lines, line, idx) => {
    151           const contextLno = idx + lineNumber - result.contextBefore.length;
    152           const bounds = (() => {
    153             // Examining the matching line, for which bounds information is available
    154             if (contextLno === lineNumber) {
    155               return [result.bounds.left, result.bounds.right];
    156             }
    157 
    158             // Defer to existing bounds information
    159             return existing.lines &&
    160               existing.lines[contextLno] &&
    161               existing.lines[contextLno].bounds;
    162           })();
    163 
    164           return {
    165             ...lines,
    166             [contextLno]: { bounds, line },
    167           };
    168         }, {});
    169 
    170         return {
    171           ...aggregated,
    172           [key]: {
    173             repo: result.tree,
    174             version: result.version,
    175             path: result.path,
    176             lines: {
    177               ...existing.lines || {},
    178               ...contextLines,
    179             },
    180           },
    181         };
    182       }, {}))
    183       .map((resultGroup) => ({
    184         ...resultGroup,
    185         lines: Object.entries(resultGroup.lines)
    186           .map(([number, details]) => ({
    187             ...details,
    188             number: parseInt(number, 10),
    189           }))
    190           .sort((a, b) => a.number - b.number),
    191       }));
    192 
    193     const files = Object.values(data.fileResults.reduce((acc, file) => ({
    194       ...acc,
    195       // Deduplicate results keyed by its repository and file path
    196       [`${file.tree}-${file.path}`]: {
    197         repo: file.tree,
    198         version: file.version,
    199         path: file.path,
    200         bounds: [file.bounds.left, file.bounds.right],
    201       },
    202     }), {}));
    203 
    204     const stats = {
    205       exitReason: data.stats.exitReason,
    206       time: parseInt(data.stats.totalTime, 10),
    207     };
    208 
    209     return { code, files, stats };
    210   }
    211 }
