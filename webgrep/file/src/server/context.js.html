      5 import ConfigClient from 'server/clients/config';
      6 import GRPCClient, { RoundRobinGRPCLoadBalancer, FailoverGRPCLoadBalancer, StaticGRPCLoadBalancer } from 'server/clients/grpc';
      7 import MetricsClient, { NoopMetricsClient, StatsdMetricsClient } from 'server/clients/metrics';
      8 import SourceClient, { NoopSourceBackend, LocalSourceBackend, GitlabSourceBackend } from 'server/clients/source';
      9 import AdminLogic from 'server/logic/admin';
     10 import MetaLogic from 'server/logic/meta';
     11 import SearchLogic from 'server/logic/search';
     12 import SourceLogic from 'server/logic/source';

     15 /**
     16  * Server-side context shared by all incoming request handlers.
     17  */
     18 export default class Context {
     19   constructor(options) {
     20     this.config = new ConfigClient(options.config);
     21 
     22     this.metrics = new MetricsClient(
     23       (() => {
     24         const backends = Object.keys(this.config.get('server.metrics') || {});
     25         if (!backends.length) {
     26           return new NoopMetricsClient();
     27         }
     28 
     29         const [backend] = backends;
     30 
     31         switch (backend) {
     32           case 'statsd':
     33             return new StatsdMetricsClient(
     34               this.config.get('server.metrics.statsd.addr') || '',
     35               this.config.get('server.metrics.statsd.prefix') || 'webgrep',
     36               this.config.get('server.metrics.statsd.sample_rate') || 1.0,
     37             );
     38           default:
     39             return new NoopMetricsClient();
     40         }
     41       })(),
     42       { host: os.hostname(), version: process.env.VERSION },
     43     );
     44 
     45     this.cache = (() => {
     46       const noop = new CacheClient(
     47         new NoopCache(),
     48         '',
     49         0,
     50         new MetricsClient(new NoopMetricsClient(), {}),
     51       );
     52 
     53       const backends = Object.keys(this.config.get('server.cache') || {});
     54       if (!backends.length) {
     55         return noop;
     56       }
     57 
     58       const [backend] = backends;
     59 
     60       switch (backend) {
     61         case 'redis':
     62           return new CacheClient(
     63             new RedisCache(
     64               this.config.get('server.cache.redis.socket'),
     65               this.config.get('server.cache.redis.timeout') || 0,
     66             ),
     67             this.config.get('server.cache.redis.prefix') || 'webgrep',
     68             this.config.get('server.cache.redis.ttl') || 10 * 60 * 1000,
     69             this.metrics,
     70           );
     71         case 'memory':
     72           return new CacheClient(
     73             new MemoryCache(),
     74             '',
     75             this.config.get('server.cache.memory.ttl') || 10 * 60 * 1000,
     76             this.metrics,
     77           );
     78         default:
     79           return noop;
     80       }
     81     })();
     82 
     83     this.service = {
     84       codesearch: (() => {
     85         const backends = this.config.get('server.livegrep.servers').map((server) => new GRPCClient(
     86           'CodeSearch',
     87           server.address,
     88           server.authority || '',
     89           CodeSearchClient,
     90           this.metrics,
     91           {
     92             requestTimeout: this.config.get('server.livegrep.request_timeout') || 3000,
     93             maxRecvMessageSize: this.config.get('server.livegrep.max_recv_message_size') || 0,
     94             maxSendMessageSize: this.config.get('server.livegrep.max_send_message_size') || 0,
     95           },
     96         ));
     97 
     98         switch (this.config.get('server.livegrep.load_balancing_policy')) {
     99           case 'round-robin':
    100             return new RoundRobinGRPCLoadBalancer(backends);
    101           case 'failover':
    102             return new FailoverGRPCLoadBalancer(backends);
    103           default:
    104             return new StaticGRPCLoadBalancer(backends);
    105         }
    106       })(),
    107     };
    108 
    109     this.source = (() => {
    110       const noop = new SourceClient(
    111         new NoopSourceBackend(),
    112         new MetricsClient(new NoopMetricsClient(), {}),
    113       );
    114 
    115       const backends = Object.keys(this.config.get('server.source') || {});
    116       if (!backends.length) {
    117         return noop;
    118       }
    119 
    120       const [backend] = backends;
    121 
    122       switch (backend) {
    123         case 'local':
    124           return new SourceClient(
    125             new LocalSourceBackend(this.config.get('server.source.local.repositories') || []),
    126             this.metrics,
    127           );
    128         case 'gitlab':
    129           return new SourceClient(
    130             new GitlabSourceBackend(
    131               this.config.get('server.source.gitlab.base_url') || 'https://gitlab.com',
    132               this.config.get('server.source.gitlab.socket_path'),
    133               this.config.get('server.source.gitlab.access_token'),
    134               {
    135                 key: this.config.get('server.source.gitlab.tls_key'),
    136                 cert: this.config.get('server.source.gitlab.tls_cert'),
    137                 ca: this.config.get('server.source.gitlab.tls_ca_cert'),
    138               },
    139             ),
    140             this.metrics,
    141           );
    142         default:
    143           return noop;
    144       }
    145     })();
    146 
    147     this.logic = {
    148       admin: new AdminLogic(this),
    149       meta: new MetaLogic(this),
    150       search: new SearchLogic(this),
    151       source: new SourceLogic(this),
    152       view: new ViewLogic(this),
    153     };
    154 
    155     this.log = winston.createLogger({
    156       level: options.verbosity,
    157       format: winston.format.combine(
    158         winston.format.timestamp(),
    159         winston.format.splat(),
    160         winston.format.printf((fmt) =>
    161           `${fmt.timestamp} ${fmt.level.toUpperCase()}\t${fmt.message}`),
    162       ),
    163       transports: [
    164         new winston.transports.Console({
    165           timestamp: () => Date.now(),
    166         }),
    167       ],
    168     });
    169   }
    170 }
