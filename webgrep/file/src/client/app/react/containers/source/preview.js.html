     47     selection: PropTypes.shape({

     66   _handleSearchPromptClick() {
     67     const { selection, onSearchQueryChange, onHide } = this.props;
     68 
     69     if (selection.text) {
     70       onSearchQueryChange(selection.text);
     71       onHide();
     72       // Also scroll to the top of the document, since the search results have changed
     73       scroll();
     74     }
     75   }

     77   render() {
     78     const {
     79       source: { err, data, isLoaded },
     80       repo,
     81       version,
     82       path,
     83       focusLine,
     84       urlPattern,
     85       onHide,
     86       width,
     87       themePreference,
     88       selection,
     89       onPathCopy,
     90       onSingleFileSearchClick,
     91     } = this.props;
     92 
     93     const { data: contents, ok } = decodeBase64(data);
     94     const language = Object.entries(CANONICAL_FILE_EXTENSION_LANGUAGES)
     95       .reduce((acc, [extension, syntax]) => (path.endsWith(extension) ? syntax : acc), 'text');
     96     const annotations = [
     97       language,
     98       `${isLoaded && ok ? (contents.match(/\n/g) || ['']).length : '-'} lines`,
     99       version,
    100     ];
    101 
    102     const body = (() => {
    103       if (err) {
    104         return (
    105           <ErrorAlert
    106             size="beta"
    107             message={err.message}
    108           />
    109         );
    110       }
    111 
    112       if (!isLoaded) {
    113         return (
    114           <Spinner />
    115         );
    116       }
    117 
    118       if (!ok) {
    119         return (
    120           <ErrorAlert
    121             size="beta"
    122             message="Failure decoding base64-encoded file payload; it is likely corrupted."

    125       }
    126 
    127       const scrollOffset = this.code.current ? this.code.current.scrollLeft : 0;
    128       const codeBounds = this.code.current ?
    129         this.code.current.getBoundingClientRect() :
    130         { x: 0, y: 0 };
    131       const isSearchPromptVisible = selection.text &&
    132         selection.text !== '\n' &&
    133         selection.bounds.height <= 20 &&
    134         this.code.current &&
    135         this.code.current.contains(selection.anchor);
    136 
    137       return (
    138         <Fragment>
    139           <div
    140             ref={this.code}
    141             style={{
    142               marginTop: '-50px',
    143               paddingTop: '50px',
    144               overflowX: 'auto',
    145               position: 'relative',
    146             }}
    147           >
    148             <SourceCodeBlock
    149               language={language}
    150               theme={SYNTAX_HIGHLIGHT_THEME_NAMES[themePreference]}
    151               focusLine={focusLine}
    152             >
    153               {contents}
    154             </SourceCodeBlock>
    155 
    156             <div
    157               style={{
    158                 display: 'inline-block',
    159                 left: (selection.bounds.left + selection.bounds.right) / 2 -
    160                   codeBounds.x +
    161                   scrollOffset,
    162                 marginTop: '2px',
    163                 opacity: 0,
    164                 position: 'absolute',
    165                 top: selection.bounds.y - codeBounds.y - 25,
    166                 transform: 'translateX(-50%) translateY(-50%)',
    167                 transition: [transition.margin.DEFAULT, transition.opacity.DEFAULT].join(','),
    168                 zIndex: -1,
    169                 ...isSearchPromptVisible && {
    170                   marginTop: 0,
    171                   opacity: 1,
    172                   zIndex: 100,
    173                 },
    174               }}
    175             >
    176               <FloatingButton
    177                 value="New search"
    178                 icon={<MdKeyboardReturn />}
    179                 onClick={this.handleSearchPromptClick}
    180               />
    181             </div>
    182           </div>
    183 
    184           <KeyboardListener
    185             character="Enter"
    186             handler={this.handleSearchPromptClick}
    187           />
    188         </Fragment>
    189       );
    190     })();
    191 
    192     return width && (
    193       <Modal
    194         size="alpha"
    195         onHide={onHide}
    196         style={{
    197           borderRadius: '3px',
    198           boxShadow: '0px 12px 24px -2px rgba(0, 0, 0, 0.1)',
    199           maxHeight: '100%',
    200           overflow: 'auto',
    201           width: width < 1325 ? '100%' : '1325px',
    202         }}
    203       >
    204         <div>
    205           <Spacing size="large" top right left bottom padding>
    206             <Spacing bottom>
    207               <div
    208                 style={{
    209                   display: 'flex',
    210                   justifyContent: 'space-between',
    211                   position: 'relative',
    212                   zIndex: 100,
    213                 }}
    214               >
    215                 <SourceMetadataHeader
    216                   repo={repo}
    217                   version={version}
    218                   path={path}
    219                   urlPattern={urlPattern}
    220                   annotations={annotations}
    221                 />
    222 
    223                 <Spacing style={{ display: 'flex' }} left>
    224                   <Spacing size="tiny" right>
    225                     <Tooltip
    226                       description="Copy path to clipboard"
    227                       style={{ left: 'unset', right: 0, width: '85px' }}
    228                     >
    229                       <IconButton icon={MdContentCopy} onClick={onPathCopy} />
    230                     </Tooltip>
    231                   </Spacing>
    232 
    233                   <Spacing size="tiny" right>
    234                     <Tooltip
    235                       description="Search within this file"
    236                       style={{ left: 'unset', right: 0, width: '85px' }}
    237                     >
    238                       <IconButton
    239                         icon={MdSearch}
    240                         onClick={compose(onHide, onSingleFileSearchClick)}
    241                       />
    242                     </Tooltip>

    245                   <div>
    246                     <Tooltip
    247                       description="Close"
    248                       style={{ left: 'unset', right: 0 }}
