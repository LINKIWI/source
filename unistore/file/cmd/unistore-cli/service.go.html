     22 // preRunGlobalTimeout is a pre-run routine executed on all commands that assigns a context to the
     23 // command with an optional timeout.
     24 func preRunGlobalTimeout(cmd *cobra.Command, args []string) error {
     25 	base := context.Background()
     26 
     27 	if flagTimeout == 0 {
     28 		cmd.SetContext(base)
     29 		return nil
     30 	}
     31 
     32 	ctx, cancel := context.WithCancel(base)
     33 	cmd.SetContext(ctx)
     34 
     35 	go func() {
     36 		select {
     37 		case <-time.After(flagTimeout):
     38 			cancel()
     39 		// Propagate cancellation of the base context to the child context to avoid leaks
     40 		case <-base.Done():
     41 			cancel()
     42 		}
     43 	}()
     44 
     45 	return nil
     46 }

    159 // runDownload is a command line interface for the GetObject and StreamGetObject RPCs.
    160 func runDownload(cmd *cobra.Command, args []string) error {
    161 	rdr := newRenderer(flagFormat, os.Stderr)
    162 	bucket, _ := cmd.Flags().GetString("bucket")
    163 	rangeStart, _ := cmd.Flags().GetUint64("range-start")
    164 	rangeEnd, _ := cmd.Flags().GetUint64("range-end")
    165 	stream, _ := cmd.Flags().GetBool("stream")
    166 	chunkSize, _ := cmd.Flags().GetUint64("chunk-size")
    167 	progress, _ := cmd.Flags().GetBool("progress")
    168 
    169 	store, rpc, err := newClientConfig(flagConfig, flagStore)
    170 	if err != nil {
    171 		return err
    172 	}
    173 
    174 	client, err := newUnistoreClient(cmd.Context(), store, rpc)
    175 	if err != nil {
    176 		return err
    177 	}
    178 
    179 	get := &service.GetObjectRequest{
    180 		Resource: &common.Resource{
    181 			Backend: common.Backend(common.Backend_value[strings.ToUpper(store.Backend)]),
    182 			Bucket:  bucket,
    183 		},
    184 		Key: args[0],
    185 	}
    186 
    187 	if rangeStart > 0 || rangeEnd > 0 {
    188 		get.Range = &common.Range{
    189 			Unit:  "bytes",
    190 			Start: rangeStart,
    191 			End:   rangeEnd,
    192 		}
    193 	}
    194 
    195 	if !stream {
    196 		if flagVerbose {
    197 			rdr.proto(get)
    198 		}
    199 
    200 		response, err := client.GetObject(cmd.Context(), get)
    201 		if err != nil {
    202 			return err
    203 		}
    204 
    205 		if flagVerbose {
    206 			rdr.proto(response)
    207 		}
    208 
    209 		_, err = os.Stdout.Write(response.Data)
    210 		return err
    211 	}
    212 
    213 	request := &service.GetObjectStreamRequest{
    214 		Request:   get,
    215 		ChunkSize: chunkSize,
    216 	}
    217 
    218 	if flagVerbose {
    219 		rdr.proto(request)
    220 	}
    221 
    222 	response, err := client.StreamGetObject(cmd.Context(), request)
    223 	if err != nil {
    224 		return err
    225 	}
    226 
    227 	start := time.Now()
    228 	downloaded := new(uint64)
    229 
    230 	if progress {
    231 		interval := 5 * time.Second
    232 		ticker := time.NewTicker(interval)
    233 		defer ticker.Stop()
    234 
    235 		go func() {
    236 			prev := atomic.LoadUint64(downloaded)
    237 
    238 			for range ticker.C {
    239 				current := atomic.LoadUint64(downloaded)
    240 				fmt.Fprintf(
    241 					os.Stderr,
    242 					"%v: downloaded %s (%s/s)\n",
    243 					time.Since(start).Round(interval),
    244 					humanize.Bytes(current),
    245 					humanize.Bytes((current-prev)/5),
    246 				)
    247 				prev = current
    248 			}
    249 		}()
    250 	}
    251 
    252 	for {
    253 		message, err := response.Recv()
    254 		if err == io.EOF {
    255 			break
    256 		} else if err != nil {
    257 			return err
    258 		}
    259 
    260 		atomic.AddUint64(downloaded, uint64(len(message.Response.Data)))
    261 
    262 		if flagVerbose {
    263 			rdr.proto(message)
    264 		}
    265 
    266 		if _, err = os.Stdout.Write(message.Response.Data); err != nil {
    267 			return err
    268 		}
    269 	}
    270 
    271 	if progress {
    272 		fmt.Fprintf(
    273 			os.Stderr,
    274 			"%v: completed download of %s\n",
    275 			time.Since(start),
    276 			humanize.Bytes(atomic.LoadUint64(downloaded)),
    277 		)
    278 	}
    279 
    280 	return nil
    281 }

    283 // runUpload is a command line interface for the PutObject and StreamPutObject RPCs.
    284 func runUpload(cmd *cobra.Command, args []string) error {
    285 	rdr := newRenderer(flagFormat, os.Stdout)
    286 	bucket, _ := cmd.Flags().GetString("bucket")
    287 	timestamp, _ := cmd.Flags().GetUint64("timestamp")
    288 	stream, _ := cmd.Flags().GetBool("stream")
    289 	chunkSize, _ := cmd.Flags().GetUint64("chunk-size")
    290 	checksum, _ := cmd.Flags().GetString("checksum")
    291 	progress, _ := cmd.Flags().GetBool("progress")
    292 
    293 	store, rpc, err := newClientConfig(flagConfig, flagStore)
    294 	if err != nil {
    295 		return err
    296 	}
    297 
    298 	client, err := newUnistoreClient(cmd.Context(), store, rpc)
    299 	if err != nil {
    300 		return err
    301 	}
    302 
    303 	attributes := &common.Attributes{}
    304 	if timestamp > 0 {
    305 		attributes.Timestamp = timestamppb.New(time.Unix(0, int64(timestamp)))
    306 	}
    307 
    308 	put := &service.PutObjectRequest{
    309 		Resource: &common.Resource{
    310 			Backend: common.Backend(common.Backend_value[strings.ToUpper(store.Backend)]),
    311 			Bucket:  bucket,
    312 		},
    313 		Key:        args[0],
    314 		Checksum:   checksum,
    315 		Attributes: attributes,
    316 	}
    317 
    318 	// Wrap os.Stdin with a context-aware reader, so that stdin reads respect any potential
    319 	// downstream context cancellations (e.g. if stdin produces no bytes and thus blocks I/O).
    320 	stdin := util.NewContextReader(cmd.Context(), os.Stdin)
    321 
    322 	if !stream {
    323 		data, err := io.ReadAll(stdin)
    324 		if err != nil {
    325 			return err
    326 		}
    327 
    328 		put.Data = data
    329 
    330 		if flagVerbose {
    331 			rdr.proto(put)
    332 		}
    333 
    334 		response, err := client.PutObject(cmd.Context(), put)
    335 		if err != nil {
    336 			return err
    337 		}
    338 
    339 		return rdr.proto(response)
    340 	}
    341 
    342 	request, err := client.StreamPutObject(cmd.Context())
    343 	if err != nil {
    344 		return err
    345 	}
    346 
    347 	start := time.Now()
    348 	uploaded := new(uint64)
    349 
    350 	if progress {
    351 		interval := 5 * time.Second
    352 		ticker := time.NewTicker(interval)
    353 		defer ticker.Stop()
    354 
    355 		go func() {
    356 			prev := atomic.LoadUint64(uploaded)
    357 
    358 			for range ticker.C {
    359 				current := atomic.LoadUint64(uploaded)
    360 				fmt.Printf(
    361 					"%v: uploaded %s (%s/s)\n",
    362 					time.Since(start).Round(interval),
    363 					humanize.Bytes(current),
    364 					humanize.Bytes((current-prev)/5),
    365 				)
    366 				prev = current
    367 			}
    368 		}()
    369 	}
    370 
    371 	for {
    372 		chunk := make([]byte, chunkSize)
    373 		n, err := stdin.Read(chunk)
    374 		if err == io.EOF {
    375 			break
    376 		} else if err != nil {
    377 			return err
    378 		}
    379 
    380 		message := &service.PutObjectStreamRequest{Request: put}
    381 		message.Request.Data = chunk[:n]
    382 
    383 		atomic.AddUint64(uploaded, uint64(n))
    384 
    385 		if flagVerbose {
    386 			rdr.proto(message)
    387 		}
    388 
    389 		if err := request.Send(message); err != nil {
    390 			break
    391 		}
    392 	}
    393 
    394 	response, err := request.CloseAndRecv()
    395 	if err != nil {
    396 		return err
    397 	}
    398 
    399 	if progress {
    400 		fmt.Printf(
    401 			"%v: completed upload of %s\n",
    402 			time.Since(start),
    403 			humanize.Bytes(atomic.LoadUint64(uploaded)),
    404 		)
    405 	}
    406 
    407 	return rdr.proto(response)
    408 }
