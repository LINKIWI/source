      5 import omit from 'util/omit';

      7 export const GRACE_TIMEOUT_INTERVAL = 150;
      8 
      9 /**
     10  * Wrap an arbitrary element with a tooltip next to the element on hover.
     11  */
     12 export default class Tooltip extends Component {
     13   static propTypes = {
     14     contents: PropTypes.element.isRequired,
     15     persistent: PropTypes.bool,
     16     width: PropTypes.oneOfType([
     17       PropTypes.number,
     18       PropTypes.string,
     19     ]),
     20     offset: PropTypes.number,
     21     top: PropTypes.bool,
     22     bottom: PropTypes.bool,
     23     children: PropTypes.node.isRequired,
     24     style: PropTypes.object,
     25   };
     26 
     27   static defaultProps = {
     28     persistent: false,
     29     width: 'auto',
     30     offset: 0,
     31     top: true,
     32     bottom: false,
     33     style: {},
     34   };
     35 
     36   constructor(props) {
     37     super(props);
     38 
     39     const { persistent } = props;
     40 
     41     this.state = {
     42       displayTooltip: persistent,
     43     };
     44   }
     45 
     46   handleMouseOver = () => {
     47     // If we mouse-over the element again before the timeout expires, simply clear the interval so
     48     // that the display is maintained.
     49     clearInterval(this.interval);
     50 
     51     this.setState({ displayTooltip: true });
     52   };
     53 
     54   handleMouseOut = () => {
     55     const { persistent } = this.props;
     56 
     57     // The logic here is to set a delay before actually changing the state of the component to
     58     // no longer display the tooltip. This allows the mouse to temporarily exit the tooltip zone
     59     // while still preserving display of the tooltip.
     60     this.interval = setTimeout(() => this.setState({ displayTooltip: persistent }),
     61       GRACE_TIMEOUT_INTERVAL);
     62   };
     63 
     64   render() {
     65     const {
     66       contents,
     67       width,
     68       offset,
     69       bottom,
     70       children,
     71       style: overrides,
     72       ...props
     73     } = this.props;
     74     const { displayTooltip } = this.state;
     75 
     76     const proxyProps = omit(props, ['persistent', 'top']);
     77 
     78     const placementProperty = bottom ? 'top' : 'bottom';
     79 
     80     const containerStyle = {
     81       display: 'inline-block',
     82       position: 'relative',
     83     };
     84 
     85     const tooltipStyle = {
     86       background: colors.gray80,
     87       left: offset,
     88       opacity: displayTooltip ? 0.95 : 0,
     89       padding: '7px 15px',
     90       position: 'absolute',
     91       visibility: displayTooltip ? 'inherit' : 'hidden',
     92       width,
     93       [placementProperty]: 'calc(100% + 5px)',
     94       ...transitionStyle(),
     95       ...overrides,
     96     };
     97 
     98     return (
     99       <div
    100         style={containerStyle}
    101         onMouseOver={this.handleMouseOver}
    102         onMouseOut={this.handleMouseOut}
    103       >
    104         {children}
    105 
    106         <span style={tooltipStyle} {...proxyProps}>
    107           {contents}
    108         </span>
    109       </div>
    110     );
    111   }
    112 }
