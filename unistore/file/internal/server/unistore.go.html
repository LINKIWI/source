    276 // StreamGetObject calls into the backend's StreamGetObject.
    277 func (u *unistoreService) StreamGetObject(request *service.GetObjectStreamRequest, stream service.Unistore_StreamGetObjectServer) error {
    278 	responses, errs := u.backend.StreamGetObject(stream.Context(), request)
    279 	if responses == nil {
    280 		return <-errs
    281 	}
    282 
    283 	for {
    284 		select {
    285 		case err := <-errs:
    286 			if err != nil {
    287 				return err
    288 			}
    289 		case response, ok := <-responses:
    290 			if !ok {
    291 				// Channel closed; there are no remaining responses to read
    292 				return nil
    293 			}
    294 
    295 			if err := stream.Send(response); err != nil {
    296 				return err
    297 			}
    298 		}
    299 	}
    300 }

    307 // StreamPutObject calls into the backend's StreamPutObject.
    308 func (u *unistoreService) StreamPutObject(stream service.Unistore_StreamPutObjectServer) error {
    309 	puts := make(chan *service.PutObjectStreamRequest)
    310 
    311 	go func() {
    312 		for {
    313 			request, err := stream.Recv()
    314 			if err != nil {
    315 				break
    316 			}
    317 
    318 			puts <- request
    319 		}
    320 
    321 		close(puts)
    322 	}()
    323 
    324 	response, err := u.backend.StreamPutObject(stream.Context(), puts)
    325 	if err != nil {
    326 		return err
    327 	}
    328 
    329 	return stream.SendAndClose(response)
    330 }
