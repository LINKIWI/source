     56     def get(self, node):
     57         """
     58         Read the value of a configuration directive by its node expression. Nodes are described
     59         hierarchically, delimited with a dot.
     60 
     61         :param node: Node expression for the directive.
     62         :return: Value at the specified node (may be a dict, list, or literal primitive) if it
     63                  exists. Returns a default value if specified; otherwise raises an exception.
     64         """
     65         def lookup(config, path):
     66             if not path:
     67                 return config
     68 
     69             current = path[0]
     70             rest = path[1:]
     71 
     72             if current not in config:
     73                 raise InvalidConfigDirectiveException('Node `{}` does not exist.'.format(current))
     74 
     75             return lookup(config[current], rest)
     76 
     77         try:
     78             return lookup(self.config['repliqate'], node.split('.'))
     79         except InvalidConfigDirectiveException:
     80             if node not in CONFIG_DEFAULTS:
     81                 raise
     82 
     83             return CONFIG_DEFAULTS[node]
