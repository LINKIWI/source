     18 // Config is the client-specified configuration for the statsd client.
     19 type Config struct {
     20 	// URI describing the statsd endpoint. Supported schemes are "udp", "tcp", "unix", and
     21 	// "file" for UDP, TCP, UDS (Unix datagram socket), and local disk file transports,
     22 	// respectively. A UDP or TCP transport should specify the network address, a Unix socket
     23 	// transport should specify the path to a socket file, and a file transport should specify
     24 	// the path to a preexisting or nonexistent file on disk.
     25 	//
     26 	// Multiple addresses may be specified, delimited by a comma, to indicate use of a
     27 	// composite transport that will concurrently write metrics to all addresses. By default,
     28 	// the composite transport operates at the strongest write consistency level; it waits until
     29 	// all writes have completed and will consider the metric emission to have failed if any of
     30 	// the underlying transports fail.
     31 	//
     32 	// For example:
     33 	// A UDP transport should be shaped as "udp://localhost:8125".
     34 	// A TCP transport should be shaped as "tcp://localhost:8125".
     35 	// A UDS transport should be shaped as "unix:///var/run/metrics/statsd.sock".
     36 	// A file transport should be shaped as "file:///var/log/metrics/statsd.log".
     37 	// A composite transport that writes to multiple addresses may be shaped as
     38 	// "udp://localhost:8125,tcp://localhost:8125,file:///var/log/metrics/statsd.log".
     39 	Address string
     40 
     41 	// Prefix to prepend to all emitted metrics.
     42 	Prefix string
     43 
     44 	// Suffix to append to all emitted metrics.
     45 	Suffix string
     46 
     47 	// Set of defaults tags to include on all emitted metrics.
     48 	DefaultTags map[string]interface{}
     49 
     50 	// Sample rate to share across all metric emissions for which sampling is supported.
     51 	SampleRate float64
     52 
     53 	// Used for buffering metric emissions. When non-negative, a buffered transport is used that
     54 	// batches sending of metrics once the threshold queue size has been reached. When zero or
     55 	// negative, buffering is disabled; each metric is sent over the transport immediately.
     56 	BufferSize int
     57 
     58 	// URI describing the address of a SOCKS5 proxy server through which the transport should be
     59 	// established. Supported schemes are "tcp" and "unix" for TCP and Unix domain socket proxy
     60 	// server listeners, respectively.
     61 	//
     62 	// Only statsd emissions through TCP transports support use of a proxy server; this setting
     63 	// is ignored for all other transport types. When omitted, no proxy is used.
     64 	Proxy string
     65 
     66 	// Used for automatic and asynchronous connection reestablishment to the network transport
     67 	// on metric emission errors. When nonzero, this duration defines the interval at which the
     68 	// current transport's connection may be reestablished if the number of transmission errors
     69 	// accumulated in the current window is nonzero. When zero, health probing of the network
     70 	// transport is disabled at the library layer, and it is the responsibility of the client
     71 	// application to reestablish a new instance as needed in response to write errors.
     72 	TransportProbeInterval time.Duration
     73 
     74 	// Requested time interval between keepalive probes for the underlying connection, relevant
     75 	// only when using a reliable transport (i.e. TCP) with support from the underlying OS. When
     76 	// zero or negative, keepalive probes are disabled.
     77 	TransportKeepaliveInterval time.Duration
     78 
     79 	// Absolute timeout for establishing/dialing a new connection, relevant only when using a
     80 	// reliable transport (i.e. TCP). When zero, no application-layer timeout is applied.
     81 	TransportConnectTimeout time.Duration
     82 
     83 	// Absolute timeout for individual payload transmissions over the network, relevant only
     84 	// when using a reliable transport (i.e. TCP). When zero, no application-layer transmission
     85 	// timeout is applied.
     86 	TransportSendTimeout time.Duration
     87 
     88 	// Initialize the backing transport lazily on the first metric emission instead of eagerly
     89 	// on client initialization. This has the effect of potentially masking transport errors
     90 	// until late into the application lifecycle, though this may be desirable when using a
     91 	// reliable transport (e.g. TCP or UDS) to eliminate the requirement that the backend be
     92 	// available before the client is initialized. It is recommended that this option be used in
     93 	// conjunction with active transport probing.
     94 	LazyTransportInitialization bool
     95 
     96 	// Specify a metrics logger used for logging all metrics emitted by this client. When
     97 	// omitted, a noop logger is used, which effectively disables any logging.
     98 	Logger logger.Logger
     99 
    100 	// Override the line protocol serializer implementation. This can be used, for example, to
    101 	// use a custom tags serialization mechanism, or to conform to a different metrics protocol
    102 	// that shares statsd metric types. When omitted, the default statsd serializer is used.
    103 	Serializer protocol.LineSerializer
    104 }

    106 // Validate the supplied configuration parameters.
    107 func (c *Config) validate() error {
    108 	if c.Address != "" {
    109 		for _, addr := range strings.Split(c.Address, ",") {
    110 			uri, err := data.ParseAddressURI(addr)
    111 			if err != nil {
    112 				return errors.New(
    113 					"config",
    114 					"malformed statsd server address URI",
    115 					errors.Tag{Key: "addr", Value: addr},
    116 				)
    117 			}
    118 
    119 			switch uri.Scheme {
    120 			case "udp", "tcp":
    121 				if _, _, err := net.SplitHostPort(uri.Authority); err != nil {
    122 					return errors.Stack(
    123 						"config",
    124 						"failed to parse network address",
    125 						err,
    126 						errors.Tag{Key: "addr", Value: addr},
    127 					)
    128 				}
    129 			case "unix":
    130 				file, err := os.Stat(uri.Authority)
    131 				if err != nil {
    132 					return errors.Stack(
    133 						"config",
    134 						"failed to stat socket path",
    135 						err,
    136 						errors.Tag{Key: "addr", Value: addr},
    137 					)
    138 				} else if file.Mode()&os.ModeSocket != os.ModeSocket {
    139 					return errors.New(
    140 						"config",
    141 						"socket path is not a Unix socket",
    142 						errors.Tag{Key: "addr", Value: addr},
    143 					)
    144 				}
    145 			case "file":
    146 				file, err := os.Stat(uri.Authority)
    147 				if err == nil && !file.Mode().IsRegular() {
    148 					return errors.New(
    149 						"config",
    150 						"file path is not a regular file",
    151 						errors.Tag{Key: "addr", Value: addr},
    152 					)
    153 				}
    154 
    155 				if unix.Access(uri.Authority, unix.W_OK) != nil &&
    156 					unix.Access(path.Dir(uri.Authority), unix.W_OK) != nil {
    157 					return errors.New(
    158 						"config",
    159 						"both file path and its parent directory are not writable",
    160 						errors.Tag{Key: "addr", Value: addr},
    161 					)
    162 				}
    163 			default:
    164 				return errors.New(
    165 					"config",
    166 					"unsupported statsd server address scheme",
    167 					errors.Tag{Key: "addr", Value: addr},
    168 				)
    169 			}
    170 		}
    171 	}
    172 
    173 	if c.Proxy != "" {
    174 		uri, err := data.ParseAddressURI(c.Proxy)
    175 		if err != nil {
    176 			return errors.New(
    177 				"config",
    178 				"malformed proxy server address URI",
    179 				errors.Tag{Key: "addr", Value: c.Proxy},
    180 			)
    181 		}
    182 
    183 		switch uri.Scheme {
    184 		case "tcp":
    185 			if _, _, err := net.SplitHostPort(uri.Authority); err != nil {
    186 				return errors.Stack(
    187 					"config",
    188 					"failed to parse proxy server address",
    189 					err,
    190 					errors.Tag{Key: "proxy", Value: c.Proxy},
    191 				)
    192 			}
    193 		case "unix":
    194 			file, err := os.Stat(uri.Authority)
    195 			if err != nil {
    196 				return errors.Stack(
    197 					"config",
    198 					"failed to stat proxy server socket path",
    199 					err,
    200 					errors.Tag{Key: "addr", Value: c.Proxy},
    201 				)
    202 			} else if file.Mode()&os.ModeSocket != os.ModeSocket {
    203 				return errors.New(
    204 					"config",
    205 					"proxy server socket path is not a Unix socket",
    206 					errors.Tag{Key: "addr", Value: c.Proxy},
    207 				)
    208 			}
    209 		default:
    210 			return errors.New(
    211 				"config",
    212 				"unsupported proxy server address scheme",
    213 				errors.Tag{Key: "addr", Value: c.Proxy},
    214 			)
    215 		}
    216 	}
    217 
    218 	if c.SampleRate < 0.0 || c.SampleRate > 1.0 {
    219 		return errors.New(
    220 			"config",
    221 			"sample rate must be in range [0.0, 1.0]",
    222 			errors.Tag{Key: "sample_rate", Value: c.SampleRate},
    223 		)
    224 	}
    225 
    226 	if c.BufferSize < 0 {
    227 		return errors.New(
    228 			"config",
    229 			"buffer size must be non-negative",
    230 			errors.Tag{Key: "buffer_size", Value: c.BufferSize},
    231 		)
    232 	}
    233 
    234 	if c.TransportProbeInterval < 0 {
    235 		return errors.New(
    236 			"config",
    237 			"transport probe interval must be non-negative",
    238 			errors.Tag{Key: "probe_interval", Value: c.TransportProbeInterval},
    239 		)
    240 	}
    241 
    242 	return nil
    243 }
