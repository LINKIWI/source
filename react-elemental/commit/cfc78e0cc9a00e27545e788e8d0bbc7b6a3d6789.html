+describe('Label', () => {

+    const label = shallow(
+      <Label label="label" />,

+  });
+
+  test('Only renders sublabel', () => {

+  });
+
+  test('Renders both label and sublabel', () => {
+    const label = shallow(
+      <Label label="label" sublabel="sublabel" />,
+    );

+  });
