     15 class MemoryCache(object):
     16     """
     17     MemoryCache is a simple, thread-safe in-memory key-value cache.
     18     """
     19 
     20     def __init__(self):
     21         """
     22         Create a MemoryCache with the default in-memory storage backend.
     23         """
     24         self.lock = threading.Lock()
     25         self.store = {}
     26 
     27     def get(self, key):
     28         """
     29         Retrieve a key's value.
     30 
     31         :param key: Raw key.
     32         :return: Associated value, if it exists and is prior to expiry.
     33         """
     34         with self.lock:
     35             try:
     36                 return self.store[key]
     37             except KeyError:
     38                 return None
     39 
     40     def set(self, key, value):
     41         """
     42         Set a key-value pair.
     43 
     44         :param key: Raw key.
     45         :param value: Associated value.
     46         """
     47         with self.lock:
     48             self.store[key] = str(value)
     49 
     50     def delete(self, key):
     51         """
     52         Delete a key, if it exists.
     53 
     54         :param key: Raw key.
     55         """
     56         with self.lock:
     57             if key in self.store:
     58                 del self.store[key]

     61 class RedisProxyClient(object):
     62     """
     63     Intermediary proxy client in front of Redis that gracefully falls back to an in-memory cache if
     64     Redis fails or is otherwise unavailable, to provide additional resiliency.
     65     """
     66 
     67     def __init__(self, addr, contention_resolution):
     68         """
     69         Create a RedisProxyClient.
     70 
     71         :param addr: Address to the Redis cluster.
     72         :param contention_resolution: Function expressing how to resolve conflicts on read
     73                                       inconsistencies between the primary and failover clients.
     74         """
     75         ip, port = addr.split(':')
     76 
     77         self.contention_resolution = contention_resolution or (lambda primary, secondary: primary)
     78         self.memory = MemoryCache()
     79         self.redis = redis.Redis(
     80             host=ip,
     81             port=port,
     82         )
     83 
     84     def get(self, key):
     85         """
     86         Get the value for a key, prioritizing Redis if available.
     87 
     88         :param key: Raw key.
     89         :return: Associated value.
     90         """
     91         try:
     92             redis_val = self.redis.get(key)
     93             mem_val = self.memory.get(key)
     94 
     95             if redis_val != mem_val:
     96                 resolved_val = self.contention_resolution(redis_val, mem_val)
     97                 self.set(key, resolved_val)
     98                 return resolved_val
     99 
    100             return redis_val
    101         except (ConnectionError, TimeoutError):
    102             return self.memory.get(key)
    103 
    104     def set(self, key, value):
    105         """
    106         Set the value for a key. Dark writes to the backup in-memory store are always performed
    107         to synchronize the state of the in-memory store with Redis, so that read failovers do not
    108         sacrifice the consistency of the underlying data.
    109 
    110         :param key: Raw key.
    111         :param value: Associated value.
    112         """
    113         try:
    114             return self.redis.set(key, value)
    115         except (ConnectionError, TimeoutError):
    116             pass
    117         finally:
    118             return self.memory.set(key, value)
    119 
    120     def delete(self, key):
    121         """
    122         Invalidate a cache entry. Like the other write operation set(), dark writes are always
    123         performed to keep the in-memory cache consistent with Redis in the event of a failover.
    124 
    125         :param key: Raw key.
    126         """
    127         try:
    128             return self.redis.delete(key)
    129         except (ConnectionError, TimeoutError):
    130             pass
    131         finally:
    132             return self.memory.delete(key)

    135 class KeyValueStoreClient(object):

    137     Caching abstractions on top of a key value storage system.

    140     def __init__(self, addr, prefix, contention_resolution=None):
    141         """
    142         Create a key value store client with a Redis backend.
    143 
    144         :param addr: Address of the Redis cluster.
    145         :param prefix: String prefix for all inserted cache keys.
    146         :param contention_resolution: Function expressing how to resolve conflicts on read
    147                                       inconsistencies between the primary and failover clients.
    148         """
    149         self.prefix = prefix
    150 
    151         if addr:
    152             self.backend = RedisProxyClient(addr, contention_resolution)
    153         else:
    154             self.backend = MemoryCache()
    155 
    156     def closure(self, namespace, key, tags={}):
    157         """
    158         Factory for a key value read/write client for a particular key.
    159 
    160         :param namespace: Namespace of the key.
    161         :param key: The key itself.
    162         :param tags: Optional dictionary of tags to qualify the key.
    163         :return: A client with get and set methods within this client's closure.
    164         """
    165         def set_proxy(value):
    166             return self.set(namespace, key, tags, value)
    167 
    168         def get_proxy():
    169             return self.get(namespace, key, tags)
    170 
    171         def delete_proxy():
    172             return self.delete(namespace, key, tags)
    173 
    174         return KeyValueStoreClosure(set_proxy, get_proxy, delete_proxy)
    175 
    176     def get(self, namespace, key, tags={}):
    177         """
    178         Get a value.
    179 
    180         :param namespace: Namespace of the key.
    181         :param key: The key itself.
    182         :param tags: Optional dictionary of tags to qualify the key.
    183         :return: The cached value, if available; None otherwise.
    184         """
    185         return self.backend.get(
    186             key=self._format_key(namespace, key, tags),
    187         )
    188 
    189     def set(self, namespace, key, tags, value):
    190         """
    191         Set a value. This operation treats new entries and updates to existing entries
    192         identically.
    193 
    194         :param namespace: Namespace of the key.
    195         :param key:The key itself.
    196         :param tags: Optional dictionary of tags to qualify the key.
    197         :param value: Value to set.
    198         """
    199         self.backend.set(
    200             key=self._format_key(namespace, key, tags),
    201             value=value,
    202         )
    203 
    204     def delete(self, namespace, key, tags):
    205         """
    206         Delete an entry.
    207 
    208         :param namespace: Namespace of the key.
    209         :param key: The key itself.
    210         :param tags: Optional dictionary of tags to qualify the key.
    211         """
    212         self.backend.delete(
    213             key=self._format_key(namespace, key, tags),
    214         )

    216     def _format_key(self, namespace, key, tags, delimiter=':'):
    217         """
    218         Serialize a (namespace, key, tags) triple to a plain-text string used as the raw key in the
    219         key-value storage backend.
    220 
    221         :param namespace: Namespace of the key.
    222         :param key:The key itself.
    223         :param tags: Optional dictionary of tags to qualify the key.
    224         :param delimiter: String delimiter used for separating the key qualifiers during
    225                           serialization. A colon is conventional for Redis keys.
    226         :return: Formatted key string for insertion into the database.
    227         """
    228         def format_tag_kv_pair(tag_key, tag_value):
    229             if {'=', '&'}.intersection('{}{}'.format(tag_key, tag_value)):
    230                 raise CacheException('Cache tag key or value contains illegal characters')
    231 
    232             return '{}={}'.format(tag_key, tag_value)
    233 
    234         serialized_tags = '&'.join(
    235             sorted(
    236                 format_tag_kv_pair(tag_key, tag_value)
    237                 for tag_key, tag_value in tags.items()
    238             )
    239         )
    240 
    241         return '{prefix}{delimiter}{namespace}{delimiter}{key}{delimiter}{tags}'.format(
    242             delimiter=delimiter,
    243             prefix=self.prefix,
    244             namespace=namespace,
    245             key=key,
    246             tags=serialized_tags,
    247         )
