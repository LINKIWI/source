    112 // newReverseProxy creates a new reverseProxy from a list of virtual host configurations.
    113 func newReverseProxy(vhosts []*config.VirtualHost, opts config.Proxy) (http.Handler, error) {
    114 	filterChains := make(map[string]middleware.Filter)
    115 	healthProbes := make(map[*config.Upstream]*healthProbe)
    116 
    117 	// Set up the filter chains for each virtual host up front, so that they can be trivially
    118 	// accessed in hot request paths.
    119 	for _, vhost := range vhosts {
    120 		var vhostFilters []middleware.Filter
    121 
    122 		zap.L().Debug(
    123 			"assembling virtual host filter chain",
    124 			zap.String("vhost", vhost.Name),
    125 			zap.Int("filters", len(vhost.Filters)),
    126 		)
    127 
    128 		internal, err := filters.NewInternal(nil)
    129 		if err != nil {
    130 			return nil, &util.Error{
    131 				Namespace: "server",
    132 				Message:   "error creating internal filter",
    133 				Tags: map[string]interface{}{
    134 					"vhost": vhost.Name,
    135 				},
    136 			}
    137 		}
    138 
    139 		vhostFilters = append(vhostFilters, internal)
    140 
    141 		for _, cfgFilter := range vhost.Filters {
    142 			zap.L().Debug(
    143 				"initializing filter",
    144 				zap.String("vhost", vhost.Name),
    145 				zap.String("name", cfgFilter.Name),
    146 				zap.String("type", cfgFilter.Type),
    147 			)
    148 
    149 			filterFactory, ok := filterFactories[cfgFilter.Type]
    150 			if !ok {
    151 				return nil, &util.Error{
    152 					Namespace: "server",
    153 					Message:   "unknown filter type",
    154 					Tags: map[string]interface{}{
    155 						"vhost":  vhost.Name,
    156 						"name":   cfgFilter.Name,
    157 						"type":   cfgFilter.Type,
    158 						"params": cfgFilter.Params.Value,
    159 					},
    160 				}
    161 			}
    162 
    163 			filter, err := filterFactory(cfgFilter)
    164 			if err != nil {
    165 				return nil, &util.Error{
    166 					Namespace: "server",
    167 					Message:   "error creating filter",
    168 					Tags: map[string]interface{}{
    169 						"vhost":  vhost.Name,
    170 						"name":   cfgFilter.Name,
    171 						"params": cfgFilter.Params.Value,
    172 					},
    173 					StackedError: err,
    174 				}
    175 			}
    176 
    177 			vhostFilters = append(vhostFilters, filter)
    178 		}
    179 
    180 		filterChains[vhost.Name] = middleware.NewComposedFilter(vhostFilters...)
    181 
    182 		for _, upstream := range vhost.Upstreams {
    183 			if upstream.HealthCheck != nil {
    184 				zap.L().Debug(
    185 					"creating health check for virtual host upstream",
    186 					zap.String("vhost", vhost.Name),
    187 					zap.String("upstream", upstream.Name),
    188 					zap.Stringer("address", upstream.Address),
    189 					zap.String("host", upstream.HealthCheck.Host),
    190 					zap.String("path", upstream.HealthCheck.Path),
    191 					zap.Duration("interval", upstream.HealthCheck.Interval),
    192 					zap.Duration("jitter", upstream.HealthCheck.Jitter),
    193 					zap.Bool("circuit_break", upstream.HealthCheck.EnableCircuitBreak),
    194 				)
    195 
    196 				hp, err := newHealthProbe(vhost.Name, upstream)
    197 				if err != nil {
    198 					return nil, &util.Error{
    199 						Namespace: "server",
    200 						Message:   "error creating health check probe",
    201 						Tags: map[string]interface{}{
    202 							"vhost": vhost.Name,
    203 						},
    204 					}
    205 				}
    206 
    207 				healthProbes[upstream] = hp
    208 			}
    209 		}
    210 	}
    211 
    212 	return &reverseProxy{
    213 		vhosts:       vhosts,
    214 		opts:         opts,
    215 		healthProbes: healthProbes,
    216 		filterChains: filterChains,
    217 		sequenceID:   new(int64),
    218 	}, nil
    219 }

    221 // ServeHTTP serves a single HTTP request.
    222 func (rp *reverseProxy) ServeHTTP(rw http.ResponseWriter, req *http.Request) {
    223 	var err error
    224 	var abortResp *http.Response
    225 	var transport http.RoundTripper
    226 
    227 	proxyReq := req.Clone(req.Context())
    228 	tags := map[string]interface{}{
    229 		"route_host": proxyReq.Host,
    230 		"method":     proxyReq.Method,
    231 		"protocol":   proxyReq.Proto,
    232 		"tls":        proxyReq.TLS != nil,
    233 	}
    234 
    235 	// Match virtual host
    236 	vhost, ok := rp.matchHost(proxyReq.Host)
    237 	if !ok {
    238 		metrics.Client.Incr(metricProxyUnknownHost, tags)
    239 		zap.L().Debug("unknown virtual host", zap.String("host", proxyReq.Host))
    240 		rp.respond(rw, errUnknownHost.response())
    241 		return
    242 	}
    243 
    244 	// Match upstream from route
    245 	upstream, ok := rp.matchRoute(proxyReq, vhost.Upstreams, vhost.Routes)
    246 	if !ok {
    247 		metrics.Client.Incr(metricProxyUnknownRoute, tags)
    248 		zap.L().Debug("unknown route", zap.String("virtual_host", vhost.Name))
    249 		rp.respond(rw, errUnknownRoute.response())
    250 		return
    251 	}
    252 
    253 	tags = util.MergeMaps(tags, map[string]interface{}{
    254 		"virtual_host": vhost.Name,
    255 		"upstream":     upstream.Name,
    256 	})
    257 
    258 	metrics.Client.Incr(metricProxyServe, tags)
    259 	metrics.Client.Gauge(metricProxyTotalServed, float64(atomic.LoadInt64(rp.sequenceID)), tags)
    260 	defer func() { atomic.AddInt64(rp.sequenceID, 1) }()
    261 
    262 	// Reverse proxy callbacks
    263 	rpModifyResponse := func(proxyResp *http.Response) error {
    264 		metrics.Client.Incr(metricProxySuccess, tags)
    265 
    266 		proxyResp, err = rp.filterChains[vhost.Name].ProcessResponse(proxyResp)
    267 		return err
    268 	}
    269 	rpErrorHandler := func(errResp *response) internalErrorHandler {
    270 		return func(proxyRW http.ResponseWriter, proxyReq *http.Request, err error) {
    271 			metrics.Client.Incr(
    272 				metricProxyError,
    273 				util.MergeMaps(tags, map[string]interface{}{"code": errResp.code}),
    274 			)
    275 
    276 			hub := sentry.CurrentHub().Clone()
    277 			hub.ConfigureScope(func(scope *sentry.Scope) {
    278 				scope.SetTag("virtual_host", vhost.Name)
    279 				scope.SetTag("upstream", upstream.Address.String())
    280 				scope.SetTag("route_host", proxyReq.Host)
    281 				scope.SetTag("method", proxyReq.Method)
    282 				scope.SetTag("protocol", proxyReq.Proto)
    283 				scope.SetExtra("url", proxyReq.RequestURI)
    284 				scope.SetExtra("client", proxyReq.RemoteAddr)
    285 				scope.SetExtra("code", errResp.code)
    286 			})
    287 			go hub.CaptureException(err)
    288 
    289 			zap.L().Error(
    290 				"reverse proxy error",
    291 				zap.String("virtual_host", vhost.Name),
    292 				zap.Stringer("upstream", upstream.Address),
    293 				zap.String("route_host", proxyReq.Host),
    294 				zap.String("method", proxyReq.Method),
    295 				zap.String("protocol", proxyReq.Proto),
    296 				zap.String("url", proxyReq.RequestURI),
    297 				zap.Int("status", errResp.status),
    298 				zap.String("code", errResp.code),
    299 				zap.String("message", errResp.message),
    300 				zap.Error(err),
    301 			)
    302 
    303 			if !rp.opts.EnableErrorSemantics {
    304 				errResp = errObscuredInternal
    305 			}
    306 
    307 			if rp.opts.EnableErrorPropagation {
    308 				errResp.tags = map[string]string{"error": err.Error()}
    309 			}
    310 
    311 			resp := errResp.response()
    312 			resp.Request = proxyReq
    313 
    314 			rpModifyResponse(resp)
    315 			rp.respond(proxyRW, resp)
    316 		}
    317 	}
    318 
    319 	// Apply request filters
    320 	proxyReq, abortResp, err = rp.filterChains[vhost.Name].ProcessRequest(proxyReq)
    321 	if err != nil {
    322 		rpErrorHandler(errFilterProcess)(rw, proxyReq, err)
    323 		return
    324 	} else if abortResp != nil {
    325 		metrics.Client.Incr(metricProxyEarlyTerminate, tags)
    326 		zap.L().Debug(
    327 			"early termination of request lifecycle before proxy",
    328 			zap.Int("status", abortResp.StatusCode),
    329 		)
    330 
    331 		rpModifyResponse(abortResp)
    332 		rp.respond(rw, abortResp)
    333 		return
    334 	}
    335 
    336 	// Conditionally fail-fast on upstream health check failures
    337 	if hp, ok := rp.healthProbes[upstream]; ok {
    338 		if err := hp.error(); err != nil && upstream.HealthCheck.EnableCircuitBreak {
    339 			metrics.Client.Incr(metricProxyCircuitBreak, tags)
    340 			rpErrorHandler(errUpstreamUnhealthy)(rw, proxyReq, err)
    341 			return
    342 		}
    343 	}
    344 
    345 	// Prepare proxy request to upstream
    346 	proxyReq.URL.Scheme = "http"
    347 	if upstream.TLSContext != nil {
    348 		proxyReq.URL.Scheme = "https"
    349 	}
    350 
    351 	// Transport initialization
    352 	tr, ok := rp.transports.Load(upstream)
    353 	if ok {
    354 		transport = tr.(http.RoundTripper)
    355 		metrics.Client.Incr(metricProxyTransportRecycle, tags)
    356 	} else {
    357 		transport, err = upstream.ClientTransport()
    358 		if err != nil {
    359 			rpErrorHandler(errBadTransport)(rw, proxyReq, err)
    360 			return
    361 		}
    362 
    363 		rp.transports.Store(upstream, transport)
    364 		metrics.Client.Incr(metricProxyTransportInitialize, tags)
    365 	}
    366 
    367 	backend := httputil.ReverseProxy{
    368 		Transport: transport,
    369 		// Instruct the proxy to flush responses back to the client immediately, for
    370 		// compatibility with gRPC streams.
    371 		FlushInterval: -1,
    372 		Director: func(clientReq *http.Request) {
    373 			// The Director routine is a noop because the request passed to the reverse
    374 			// proxy has already been cascaded through all request filters.
    375 		},
    376 		ModifyResponse: rpModifyResponse,
    377 		ErrorHandler:   rp.errorClassifierHandler(rpErrorHandler),
    378 		ErrorLog:       log.StdWarnLogger,
    379 	}
    380 
    381 	backend.ServeHTTP(rw, proxyReq)
    382 }

    384 // matchHost attempts to match the hostname requested by the client in its Host header with a
    385 // virtual host entry in configuration.
    386 func (rp *reverseProxy) matchHost(reqHost string) (*config.VirtualHost, bool) {
    387 	for _, vhost := range rp.vhosts {
    388 		if vhost.Host.MatchString(reqHost) {
    389 			return vhost, true
    390 		}
    391 	}
    392 
    393 	return nil, false
    394 }

    396 // matchRoute attempts to match the inbound request to an upstream based on routing policy rules in
    397 // configuration.
    398 func (rp *reverseProxy) matchRoute(
    399 	req *http.Request,
    400 	upstreams []*config.Upstream,
    401 	routes []*config.Route,
    402 ) (*config.Upstream, bool) {
    403 	for _, route := range routes {
    404 		if route.Match.Host != nil && !route.Match.Host.MatchString(req.Host) {
    405 			continue
    406 		} else if route.Match.Method != nil && !route.Match.Method.MatchString(req.Method) {
    407 			continue
    408 		} else if route.Match.Path != nil && !route.Match.Path.MatchString(req.URL.Path) {
    409 			continue
    410 		} else if route.Match.Header != nil {
    411 			value := req.Header.Get(route.Match.Header.Key)
    412 			if value == "" || !route.Match.Header.Value.MatchString(value) {
    413 				continue
    414 			}
    415 		}
    416 
    417 		for _, upstream := range upstreams {
    418 			if upstream.Name == route.Upstream {
    419 				return upstream, true
    420 			}
    421 		}
    422 	}
    423 
    424 	return nil, false
    425 }

    427 // respond writes an HTTP response from a structured response struct.
    428 func (rp *reverseProxy) respond(rw http.ResponseWriter, resp *http.Response) error {
    429 	if resp.Header != nil {
    430 		for key, value := range resp.Header {
    431 			rw.Header()[key] = value
    432 		}
    433 	}
    434 
    435 	rw.WriteHeader(resp.StatusCode)
    436 
    437 	if _, err := io.Copy(rw, resp.Body); err != nil {
    438 		return err
    439 	}
    440 
    441 	return resp.Body.Close()
    442 }

    444 // errorClassifierHandler is an internalErrorHandler that attempts to select the most semantically
    445 // meaningful response for an internal error raised by the reverse proxy.
    446 func (rp *reverseProxy) errorClassifierHandler(handler func(errResp *response) internalErrorHandler) internalErrorHandler {
    447 	var classifier func(error) *response
    448 
    449 	classifier = func(err error) *response {
    450 		if netErr, ok := err.(net.Error); ok && netErr.Timeout() {
    451 			return errUpstreamTimeout
    452 		} else if _, ok := err.(*net.OpError); ok {
    453 			return errUpstreamUnavailable
    454 		} else if unwrapped := errors.Unwrap(err); unwrapped != nil {
    455 			// Identify the first meaningful error in the error stack.
    456 			return classifier(unwrapped)
    457 		}
    458 
    459 		return errProxyInternal
    460 	}
    461 
    462 	return func(proxyRW http.ResponseWriter, proxyReq *http.Request, err error) {
    463 		handler(classifier(err))(proxyRW, proxyReq, err)
    464 	}
    465 }
