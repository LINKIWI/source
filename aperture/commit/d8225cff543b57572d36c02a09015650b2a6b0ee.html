+package aperture

+import (
+	"testing"
+
+	"github.com/stretchr/testify/assert"

+func TestConfigValidationFailure(t *testing.T) {
+	t.Parallel()
+
+	cases := []*Config{
+		// Multiple targets
+		{
+			Address: "addr",
+			Socket:  "sock",
+		},
+		// Address parse failure
+		{
+			Address: "addr",
+		},
+		// Socket stat failure
+		{
+			Socket: "sock",
+		},
+		// Invalid buffer size
+		{
+			Address:    "localhost:8125",
+			BufferSize: -1,
+		},
+		// Invalid sample rate
+		{
+			Address:    "localhost:8125",
+			SampleRate: 1.1,
+		},
+		{
+			Address:    "localhost:8125",
+			SampleRate: -1.2,
+		},
+	}
+
+	for _, cfg := range cases {
+		assert.Error(t, cfg.validate())
+	}
+}
+
+func TestConfigValidationSuccess(t *testing.T) {
+	t.Parallel()
+
+	cases := []*Config{
+		{
+			Address: "localhost:8125",
+		},
+		{
+			Address:    "localhost:8125",
+			BufferSize: 1,
+		},
+		{
+			Address:    "localhost:8125",
+			SampleRate: 0.8,
+			BufferSize: 2,
+		},
+		{
+			Address: "localhost:8125",
+			Prefix:  "bar",
+			DefaultTags: map[string]interface{}{
+				"host": "foo",
+				"ncpu": 6,
+			},
+		},
+	}
+
+	for _, cfg := range cases {
+		assert.NoError(t, cfg.validate())
+	}
+}
