     56 // ReadASCIICommand reads a full ASCII command from the underlying io.Reader, exercising heuristics
     57 // to determine how far into the stream data should be read.
     58 func (r *Reader) ReadASCIICommand() ([]byte, error) {
     59 	buf, err := r.readUntil(crlf)
     60 	if err != nil {
     61 		return buf, err
     62 	}
     63 
     64 	// Storage commands are the only protocol request messages that do conclude on the first
     65 	// CRLF sequence. The read length following the first CRLF is context-dependent on the
     66 	// claimed data size in the storage header. This routine performs a low-fidelity parse of
     67 	// the storage header and, assuming it is valid, attempts to read ahead exactly the number
     68 	// of bytes stated in the header for the data size.
     69 	for _, cmd := range storageCommands {
     70 		if bytes.HasPrefix(buf, cmd) {
     71 			components := bytes.Fields(buf)
     72 			if len(components) < 5 {
     73 				return buf, fmt.Errorf(
     74 					"protocol: bad storage header format: components=%d",
     75 					len(components),
     76 				)
     77 			}
     78 
     79 			size, err := strconv.Atoi(string(components[4]))
     80 			if err != nil {
     81 				return buf, fmt.Errorf(
     82 					"protocol: error parsing size in storage header: err=%v",
     83 					err,
     84 				)
     85 			}
     86 
     87 			data := make([]byte, size)
     88 			n, err := r.Read(data)
     89 			buf = append(buf, data...)
     90 
     91 			if n != size {
     92 				return buf, fmt.Errorf(
     93 					"protocol: cannot accommodate purported storage data "+
     94 						"size from buffer: expect=%d actual=%d",
     95 					size,
     96 					n,
     97 				)
     98 			}
     99 
    100 			tailer, err := r.readUntil(crlf)
    101 			buf = append(buf, tailer...)
    102 
    103 			return buf, err
    104 		}
    105 	}
    106 
    107 	return buf, nil
    108 }

    110 // readUntil incrementally reads from the buffered stream in chunks until the delimiter sequence is
    111 // located. Only those bytes that represent the fully-formed message are consumed from the buffered
    112 // reader. If the sequence is located before EOF, a byte slice with the full message contents is
    113 // returned; otherwise, an error indicates the command is malformed.
    114 func (r *Reader) readUntil(sequence []byte) ([]byte, error) {
    115 	var terminate bool
    116 
    117 	buf := make([]byte, 0, 512)
    118 
    119 	for !terminate {
    120 		chunk, err := r.b.Peek(scanChunkSize)
    121 		if err != nil {
    122 			if err == io.EOF && len(buf) > 0 {
    123 				// Terminate the chunk-read loop on the next iteration, relevant
    124 				// only when a nonzero buffer has already been accumulated.
    125 				terminate = true
    126 			} else {
    127 				return buf, err
    128 			}
    129 		}
    130 
    131 		buf = append(buf, chunk...)
    132 
    133 		idx := bytes.Index(buf, sequence)
    134 		if idx != -1 {
    135 			offset := idx + len(sequence)
    136 			r.b.Discard(offset - len(buf) + len(chunk))
    137 			return buf[:offset], nil
    138 		}
    139 
    140 		r.b.Discard(len(chunk))
    141 	}
    142 
    143 	return buf, ErrMalformedRequest
    144 }
