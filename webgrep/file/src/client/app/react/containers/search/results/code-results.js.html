     12 /**
     13  * List of results for matching code snippets.
     14  */
     15 class CodeResultsContainer extends PureComponent {
     16   static propTypes = {
     17     repositories: PropTypes.object.isRequired,
     18     snippets: PropTypes.array.isRequired,
     19     onSearchQueryChange: PropTypes.func.isRequired,
     20     onSearchRepositoryFilePathChange: PropTypes.func.isRequired,
     21     // HOC props
     22     hash: PropTypes.number,
     23     setHash: PropTypes.func.isRequired,
     24   };
     25 
     26   static defaultProps = {
     27     hash: null,
     28   };
     29 
     30   componentDidMount() {
     31     const { hash } = this.props;
     32 
     33     if (hash) {
     34       this._scrollPermalink(hash);
     35     }
     36   }
     37 
     38   componentDidUpdate(prevProps) {
     39     if (this.props.hash !== prevProps.hash) {
     40       this._scrollPermalink(this.props.hash);
     41     }
     42   }
     43 
     44   navigateResults = this._navigateResults.bind(this);
     45 
     46   _scrollPermalink(idx) {  // eslint-disable-line class-methods-use-this
     47     const [node] = document.getElementsByName(`#R${idx}`);
     48     if (node) {
     49       node.scrollIntoView({ behavior: 'smooth', block: 'start' });
     50     }
     51   }
     52 
     53   _navigateResults(delta) {
     54     const { snippets, hash, setHash } = this.props;
     55 
     56     return (evt) => {
     57       // Don't allow results navigation when focused on any input fields
     58       if (!INPUT_FIELD_IDS.includes(evt.target.id)) {
     59         if (hash === null) {
     60           setHash(0);
     61         } else {
     62           // Bidirectionally cycle through all available snippets
     63           setHash(modulo((hash) + delta, snippets.length));
     64         }
     65       }
     66     };
     67   }
     68 
     69   render() {
     70     const {
     71       snippets,
     72       repositories,
     73       hash,
     74       onSearchQueryChange,
     75       onSearchRepositoryFilePathChange,
     76     } = this.props;
     77 
     78     return (
     79       <Fragment>
     80         <KeyboardListener
     81           character="j"
     82           handler={this.navigateResults(1)}
     83         />
     84         <KeyboardListener
     85           character="k"
     86           handler={this.navigateResults(-1)}
     87         />
     88 
     89         {snippets.map((snippet, idx) => (
     90           <Spacing
     91             key={`${snippet.repo}-${snippet.path}`}
     92             name={`#R${idx}`}
     93             style={hash === idx ? {
     94               borderRadius: '3px',
     95               boxShadow: `0 0 0 1.5px ${background.primary.DELTA}`,
     96               transition: transition.all.DEFAULT,
     97             } : {}}
     98             bottom={idx < snippets.length - 1}
     99           >
    100             <CodeSnippetContainer
    101               position={idx + 1}
    102               permalink={`#R${idx}`}
    103               repositories={repositories}
    104               snippet={snippet}
    105               onSearchQueryChange={onSearchQueryChange}
    106               onSearchRepositoryFilePathChange={onSearchRepositoryFilePathChange}
    107               isHighlighted={hash === idx}
    108             />
    109           </Spacing>
    110         ))}
    111       </Fragment>
    112     );
    113   }
    114 }
