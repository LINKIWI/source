    407 // ClientTransport creates a new http.RoundTripper transport derived from the upstream connection
    408 // configuration. Note that this method initializes new resources and the returned transport should
    409 // thus be reused across multiple connections for this upstream in order to avoid resource leaks.
    410 func (u *Upstream) ClientTransport() (http.RoundTripper, error) {
    411 	var err error
    412 	var tlsCfg *tls.Config
    413 
    414 	if u.TLSContext != nil {
    415 		zap.L().Debug(
    416 			"creating client TLS origination context",
    417 			zap.String("upstream", u.Address.String()),
    418 		)
    419 
    420 		tlsCfg, err = u.TLSContext.ClientConfig()
    421 		if err != nil {
    422 			return nil, &util.Error{
    423 				Namespace:    "config",
    424 				Message:      "error creating client TLS origination context",
    425 				StackedError: err,
    426 			}
    427 		}
    428 	}
    429 
    430 	switch u.Transport {
    431 	case "http2":
    432 		zap.L().Debug(
    433 			"creating client HTTP/2 transport",
    434 			zap.String("upstream", u.Address.String()),
    435 			zap.Stringer("proxy", u.Proxy),
    436 			zap.Any("connection", u.Connection),
    437 		)
    438 
    439 		return &http2.Transport{
    440 			AllowHTTP:       true,
    441 			TLSClientConfig: tlsCfg,
    442 			DialTLS: func(network string, addr string, cfg *tls.Config) (net.Conn, error) {
    443 				var dialer proxy.Dialer
    444 				var cancel context.CancelFunc
    445 
    446 				dialer = &net.Dialer{
    447 					Timeout:   u.Connection.ConnectTimeout,
    448 					KeepAlive: u.Connection.KeepaliveInterval,
    449 				}
    450 
    451 				if u.TLSContext != nil {
    452 					dialer = &tls.Dialer{
    453 						NetDialer: dialer.(*net.Dialer),
    454 						Config:    cfg,
    455 					}
    456 				}
    457 
    458 				if u.Proxy != nil {
    459 					dialer, err = proxy.FromURL(u.Proxy.URL, dialer)
    460 					if err != nil {
    461 						return nil, &util.Error{
    462 							Namespace: "config",
    463 							Message:   "error creating HTTP/2 proxy dialer",
    464 							Tags: map[string]interface{}{
    465 								"name":  u.Name,
    466 								"addr":  u.Address,
    467 								"proxy": u.Proxy,
    468 							},
    469 							StackedError: err,
    470 						}
    471 					}
    472 				}
    473 
    474 				network, addr = u.Address.Address()
    475 				conn, err := dialer.Dial(network, addr)
    476 				if err != nil {
    477 					return nil, &util.Error{
    478 						Namespace: "config",
    479 						Message:   "error dialing upstream address over HTTP/2",
    480 						Tags: map[string]interface{}{
    481 							"name": u.Name,
    482 							"addr": u.Address,
    483 						},
    484 						StackedError: err,
    485 					}
    486 				}
    487 
    488 				if u.TLSContext != nil {
    489 					ctx := context.Background()
    490 					if u.TLSContext.HandshakeTimeout > 0 {
    491 						ctx, cancel = context.WithTimeout(
    492 							context.Background(),
    493 							u.TLSContext.HandshakeTimeout,
    494 						)
    495 						defer cancel()
    496 					}
    497 
    498 					if err := conn.(*tls.Conn).HandshakeContext(ctx); err != nil {
    499 						return nil, &util.Error{
    500 							Namespace: "config",
    501 							Message:   "error during upstream TLS handshake",
    502 							Tags: map[string]interface{}{
    503 								"name": u.Name,
    504 								"addr": u.Address,
    505 							},
    506 							StackedError: err,
    507 						}
    508 					}
    509 				}
    510 
    511 				return newTimeoutConn(
    512 					u.Connection.ReadTimeout,
    513 					u.Connection.WriteTimeout,
    514 					conn,
    515 				), nil
    516 			},
    517 		}, nil
    518 	default:

    520 
    521 		zap.L().Debug(
    522 			"creating client HTTP/1.1 transport",
    523 			zap.String("upstream", u.Address.String()),
    524 			zap.Stringer("proxy", u.Proxy),
    525 			zap.Any("connection", u.Connection),
    526 		)

    594 // ClientConfig transforms the TLS context into a tls.Config appropriate for purposes of TLS
    595 // origination from a client.
    596 func (t *TLSContext) ClientConfig() (*tls.Config, error) {
    597 	var verifyConnection func(tls.ConnectionState) error
    598 	var minVersion uint16
    599 
    600 	cs, err := newCertificateStore(t, func() (*x509.CertPool, error) {
    601 		return x509.SystemCertPool()
    602 	})
    603 	if err != nil {
    604 		return nil, &util.Error{
    605 			Namespace:    "config",
    606 			Message:      "failed to initialize client certificate store",
    607 			StackedError: err,
    608 		}
    609 	}
    610 
    611 	if t.VerifyPeer != nil {
    612 		pv := &peerVerifier{
    613 			pattern: t.VerifyPeer,
    614 			roots:   cs.cas,
    615 		}
    616 
    617 		verifyConnection = pv.verifyServer
    618 	}
    619 
    620 	switch t.ProtocolVersion {
    621 	case "lenient":
    622 		minVersion = tls.VersionTLS11
    623 	case "standard":
    624 		minVersion = tls.VersionTLS12
    625 	case "modern":
    626 		minVersion = tls.VersionTLS13
    627 	default:
    628 		minVersion = tls.VersionTLS12
    629 	}
    630 
    631 	return &tls.Config{
    632 		GetClientCertificate: cs.getClientCertificate,
    633 		RootCAs:              cs.cas,
    634 		// Defer to the overridden connection verification in place of default validation
    635 		// when explicit peer verification is enabled.
    636 		InsecureSkipVerify: t.VerifyMode == "none" ||
    637 			t.VerifyMode == "relaxed" ||
    638 			t.VerifyPeer != nil,
    639 		MinVersion:         minVersion,
    640 		ClientSessionCache: tls.NewLRUClientSessionCache(128),
    641 		VerifyConnection:   verifyConnection,
    642 	}, nil
    643 }
    644 
    645 // ServerConfig transforms the TLS context into a tls.Config appropriate for purposes of TLS
    646 // termination at a server.
    647 func (t *TLSContext) ServerConfig() (*tls.Config, error) {
    648 	var verifyMode tls.ClientAuthType
    649 	var verifyConnection func(tls.ConnectionState) error
    650 	var minVersion uint16
    651 
    652 	// Explicitly use a nil (empty) seed CA certificate pool in order to use only the CAs
    653 	// specified in TLS configuration for client certificate authentication.
    654 	cs, err := newCertificateStore(t, nil)
    655 	if err != nil {
    656 		return nil, &util.Error{
    657 			Namespace:    "config",
    658 			Message:      "failed to initialize server certificate store",
    659 			StackedError: err,
    660 		}
    661 	}
    662 
    663 	switch t.VerifyMode {
    664 	case "none":
    665 		verifyMode = tls.NoClientCert
    666 	case "relaxed":
    667 		verifyMode = tls.VerifyClientCertIfGiven
    668 	case "strict":
    669 		verifyMode = tls.RequireAndVerifyClientCert
    670 	default:
    671 		verifyMode = tls.NoClientCert
    672 	}
    673 
    674 	if t.VerifyPeer != nil {
    675 		pv := &peerVerifier{
    676 			pattern: t.VerifyPeer,
    677 			roots:   cs.cas,
    678 		}
    679 
    680 		verifyConnection = pv.verifyClient
    681 	}
    682 
    683 	switch t.ProtocolVersion {
    684 	case "lenient":
    685 		minVersion = tls.VersionTLS11
    686 	case "standard":
    687 		minVersion = tls.VersionTLS12
    688 	case "modern":
    689 		minVersion = tls.VersionTLS13
    690 	default:
    691 		minVersion = tls.VersionTLS12
    692 	}
    693 
    694 	return &tls.Config{
    695 		GetCertificate:   cs.getServerCertificate,
    696 		ClientCAs:        cs.cas,
    697 		ClientAuth:       verifyMode,
    698 		MinVersion:       minVersion,
    699 		VerifyConnection: verifyConnection,
    700 	}, nil
    701 }
