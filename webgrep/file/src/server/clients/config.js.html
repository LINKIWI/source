      1 import fs from 'fs';

      5 /**
      6  * Client for interacting with a YAML configuration file on disk.
      7  */
      8 export default class ConfigClient {
      9   /**
     10    * Create a config client instance.
     11    *
     12    * @param {String} path Path to the config file on disk.
     13    */
     14   constructor(path) {
     15     this.config = yaml.safeLoad(fs.readFileSync(path));
     16   }
     17 
     18   /**
     19    * Get a config value by a path descriptor, where keys are delimited by a dot.
     20    *
     21    * @param {String} path Path to the config key.
     22    * @param {String} delimiter Delimiter character of the path descriptor.
     23    * @returns {*} Corresponding config value as defined in the config file.
     24    */
     25   get(path, delimiter = '.') {
     26     return objLookup(this.config, path.split(delimiter));
     27   }
