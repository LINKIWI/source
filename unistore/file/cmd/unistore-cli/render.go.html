     41 // renderer provides the capability to render formatted messages to an output.
     42 type renderer interface {
     43 	// proto is used for rendering Protobuf messages.
     44 	proto(message proto.Message) error
     45 
     46 	// any is used for interface{} messages of arbitrary type.
     47 	any(message interface{}) error
     48 }

     50 // newRenderer creates a new renderer for the desired output format.
     51 func newRenderer(format string, output io.Writer) renderer {
     52 	switch format {
     53 	case outputFormatJSON:
     54 		return &jsonRenderer{output}
     55 	case outputFormatYAML:
     56 		return &yamlRenderer{output}
     57 	case outputFormatHuman:
     58 		return &humanRenderer{output}
     59 	case outputFormatDefault, outputFormatText:
     60 		return &textRenderer{output}
     61 	default:
     62 		return &defaultRenderer{output}
     63 	}
     64 }

     86 // proto reuses the implementation for interface{} types.
     87 func (j *jsonRenderer) proto(message proto.Message) error {

     91 // any marshals any type implementing json.Marshaler or any struct with JSON tags.
     92 func (j *jsonRenderer) any(message interface{}) error {
     93 	rendered, err := json.MarshalIndent(message, "", defaultIndent)
     94 	if err != nil {
     95 		return err
     96 	}
     97 
     98 	_, err = fmt.Fprintln(j.output, strings.TrimSpace(string(rendered)))
     99 	return err
    100 }

    152 // proto renders a human-readable representation of supported Protobuf messages.
    153 func (h *humanRenderer) proto(message proto.Message) error {
    154 	switch m := message.(type) {
    155 	case *service.InfoResponse:
    156 		return h.protoInfoResponse(m)
    157 	case *service.ListBucketsResponse:
    158 		return h.protoListBucketsResponse(m)
    159 	case *service.ListObjectsResponse:
    160 		return h.protoListObjectsResponse(m)
    161 	default:
    162 		return errUnsupportedMessage
    163 	}
    164 }

    166 // protoInfoResponse renders a service.InfoResponse.
    167 func (h *humanRenderer) protoInfoResponse(response *service.InfoResponse) error {
    168 	var renderNode func(node *common.Node) []string
    169 
    170 	renderNode = func(node *common.Node) []string {
    171 		var out []string
    172 
    173 		childrenConstants := make(map[string]string)
    174 		childrenNodes := make(map[string]*common.Node)
    175 
    176 		for param, value := range node.Children {
    177 			switch c := value.Child.(type) {
    178 			case *common.Node_Value_Value:
    179 				childrenConstants[param] = c.Value
    180 			case *common.Node_Value_Node:
    181 				childrenNodes[param] = c.Node
    182 			}
    183 		}
    184 
    185 		if len(childrenConstants) > 0 {
    186 			var params []string
    187 			for param, value := range childrenConstants {
    188 				params = append(params, fmt.Sprintf("%s: %s", param, value))
    189 			}
    190 
    191 			out = append(out, fmt.Sprintf("%s [%s]", node.Name, strings.Join(params, ", ")))
    192 		} else {
    193 			out = append(out, node.Name)
    194 		}
    195 
    196 		for _, n := range childrenNodes {
    197 			for _, line := range renderNode(n) {
    198 				out = append(out, fmt.Sprintf("%s%s", defaultIndent, line))
    199 			}
    200 		}
    201 
    202 		return out
    203 	}
    204 
    205 	fmt.Fprintln(h.output, "server version:")
    206 	fmt.Fprintf(h.output, "%sunistore/%s\n\n", defaultIndent, response.Version)
    207 	fmt.Fprintln(h.output, "server listener:")
    208 	fmt.Fprintf(h.output, "%s%s\n\n", defaultIndent, response.Address)
    209 	fmt.Fprintln(h.output, "server backend composition tree:")
    210 	for _, line := range renderNode(response.Backend) {
    211 		fmt.Fprintf(h.output, "%s%s\n", defaultIndent, line)
    212 	}
    213 
    214 	return nil
    215 }

    217 // protoListBucketsResponse renders a service.ListBucketsResponse.
    218 func (h *humanRenderer) protoListBucketsResponse(response *service.ListBucketsResponse) error {
    219 	var objects service.ListObjectsResponse
    220 
    221 	// Reuse the existing renderer for ListObjects by reshaping the ListBuckets response to
    222 	// a list of objects with Object.DIRECTORY type.
    223 
    224 	for _, bucket := range response.Buckets {
    225 		head := &service.HeadObjectResponse{
    226 			Metadata: &common.Metadata{
    227 				Key:          bucket.Bucket.Name,
    228 				ModifiedTime: bucket.Bucket.Timestamp,
    229 				ObjectType:   common.Object_DIRECTORY,
    230 			},
    231 		}
    232 
    233 		objects.Objects = append(objects.Objects, head)
    234 	}
    235 
    236 	return h.protoListObjectsResponse(&objects)
    237 }

    239 // protoListObjectsResponse renders a service.ListObjectsResponse.
    240 func (h *humanRenderer) protoListObjectsResponse(response *service.ListObjectsResponse) error {
    241 	// Sort the objects alphabetically by key name, while grouping directories at the
    242 	// beginning of the list.
    243 	sort.Slice(response.Objects, func(i int, j int) bool {
    244 		l := response.Objects[i]
    245 		r := response.Objects[j]
    246 
    247 		if l.Metadata.ObjectType == common.Object_DIRECTORY &&
    248 			r.Metadata.ObjectType == common.Object_DIRECTORY {
    249 			// Both objects are directories; sort alphabetically
    250 			return l.Metadata.Key < r.Metadata.Key
    251 		} else if l.Metadata.ObjectType == common.Object_DIRECTORY {
    252 			// The left object is a directory; always prefer it
    253 			return true
    254 		} else if r.Metadata.ObjectType == common.Object_DIRECTORY {
    255 			// The right object is a directory; always prefer it
    256 			return false
    257 		} else {
    258 			// Both objects are non-directories; sort alphabetically
    259 			return l.Metadata.Key < r.Metadata.Key
    260 		}
    261 	})
    262 
    263 	for _, object := range response.Objects {
    264 		switch object.Metadata.ObjectType {
    265 		case common.Object_DIRECTORY:
    266 			if !strings.HasSuffix(object.Metadata.Key, "/") {
    267 				object.Metadata.Key = fmt.Sprintf("%s/", object.Metadata.Key)
    268 			}
    269 
    270 			_, err := fmt.Fprintf(
    271 				h.output,
    272 				"%v\t%s\t%s\n",
    273 				object.Metadata.ModifiedTime.AsTime().Format(time.RFC3339),
    274 				"-",
    275 				object.Metadata.Key,
    276 			)
    277 			if err != nil {
    278 				return err
    279 			}
    280 
    281 		case common.Object_REGULAR:
    282 			_, err := fmt.Fprintf(
    283 				h.output,
    284 				"%v\t%s\t%s\n",
    285 				object.Metadata.ModifiedTime.AsTime().Format(time.RFC3339),
    286 				humanize.Bytes(object.Metadata.Size),
    287 				object.Metadata.Key,
    288 			)
    289 			if err != nil {
    290 				return err
    291 			}
    292 		}
    293 	}
    294 
    295 	return nil
    296 }
