+import PropTypes from 'prop-types';
+import React from 'react';
+import { Button, Modal, Spacing, Text } from 'react-elemental';
+import RepositorySelector from 'client/app/react/components/search/query/repository-selector';
+import { text } from 'client/app/util/style/color';
+
+/**
+ * Modal providing repository filter options.
+ */
+const RepositoryFilter = ({ repositories, onHide, onRepositoryToggle }) => (
+  <Modal
+    size="beta"
+    onHide={onHide}
+    style={{ borderRadius: '3px', boxShadow: '0px 12px 24px -2px rgba(0, 0, 0, 0.1)' }}
+  >
+    <Spacing size="large" top right bottom left padding>
+      <Spacing bottom>
+        <Spacing size="micro" bottom>
+          <Text size="kilo" uppercase bold>
+            Repositories
+          </Text>
+        </Spacing>
+        <Text color={text.dark.BETA} size="kilo">
+          Filter results by specific repositories.
+        </Text>
+      </Spacing>
+
+      <Spacing bottom>
+        {repositories.map((repo) => (
+          <RepositorySelector
+            key={repo.name}
+            name={repo.name}
+            remote={repo.remote}
+            isSelected={repo.isSelected}
+            onClick={() => onRepositoryToggle(repo)}
+          />
+        ))}
+      </Spacing>
+
+      <div style={{ display: 'flex', justifyContent: 'flex-end' }}>
+        <Button text="OK" onClick={onHide} />
+      </div>
+    </Spacing>
+  </Modal>
+);
+
+RepositoryFilter.propTypes = {
+  repositories: PropTypes.arrayOf(PropTypes.shape({
+    name: PropTypes.string.isRequired,
+    remote: PropTypes.string.isRequired,
+    isSelected: PropTypes.bool.isRequired,
+  }).isRequired).isRequired,
+  onHide: PropTypes.func.isRequired,
+  onRepositoryToggle: PropTypes.func.isRequired,
+};
+
+export default RepositoryFilter;
