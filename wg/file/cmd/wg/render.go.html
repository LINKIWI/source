     30 // tableResultsRenderer is an implementation of the default renderer style, which prints results
     31 // as a table with columns for the version, repository, file path, and result contents (code line
     32 // or file name). It is intended to be accessible to humans while remaining machine-parsable by
     33 // standard Unix tools.
     34 type tableResultsRenderer struct{}

     36 // RenderCodeSearchResults renders a table of version, repository, file path, line number, and line
     37 // contents.
     38 func (d *tableResultsRenderer) RenderCodeSearchResults(code []webgrep.CodeSearchResult) (string, error) {
     39 	table := cli.NewTable()
     40 
     41 	for _, result := range code {
     42 		for _, line := range result.Lines {
     43 			source := line.Line
     44 			if len(line.Bounds) == 2 {
     45 				// Apply highlighting to the matching portion of the source
     46 				// line, if applicable
     47 				source = strings.Join([]string{
     48 					line.Line[:line.Bounds[0]],
     49 					cli.Highlight(line.Line[line.Bounds[0]:line.Bounds[1]]),
     50 					line.Line[line.Bounds[1]:],
     51 				}, "")
     52 			}
     53 
     54 			row := []string{
     55 				result.Version,
     56 				result.Repository,
     57 				result.Path,
     58 				strconv.Itoa(line.Number),
     59 				fmt.Sprintf("|%s", source),
     60 			}
     61 
     62 			if err := table.Add(row); err != nil {
     63 				return "", err
     64 			}
     65 		}
     66 	}
     67 
     68 	return table.String(), nil
     69 }

     96 // stackedResultsRenderer is an implementation of a renderer style that organizes results as stacked
     97 // blocks grouped by repository and/or matching file path name. It resembles the results layout of
     98 // the webgrep UI and is generally intended for consumption by humans due to its machine-unfriendly
     99 // format.
    100 type stackedResultsRenderer struct{}

    102 // RenderCodeSearchResults renders blocks of code, each with a header indicating the source
    103 // repository and file path.
    104 func (s *stackedResultsRenderer) RenderCodeSearchResults(code []webgrep.CodeSearchResult) (string, error) {
    105 	var lines []string
    106 
    107 	for idx, result := range code {
    108 		block := cli.NewTable()
    109 		header := fmt.Sprintf("%s > %s", result.Repository, result.Path)
    110 
    111 		lines = append(lines, cli.Bold(header))
    112 		lines = append(lines, cli.Bold(strings.Repeat("-", len(header))))
    113 
    114 		for _, line := range result.Lines {
    115 			source := line.Line
    116 			if len(line.Bounds) == 2 {
    117 				source = strings.Join([]string{
    118 					line.Line[:line.Bounds[0]],
    119 					cli.Highlight(line.Line[line.Bounds[0]:line.Bounds[1]]),
    120 					line.Line[line.Bounds[1]:],
    121 				}, "")
    122 			}
    123 
    124 			block.Add([]string{
    125 				strconv.Itoa(line.Number),
    126 				fmt.Sprintf("|%s", source),
    127 			})
    128 		}
    129 
    130 		lines = append(lines, block.String())
    131 
    132 		if idx < len(code)-1 {
    133 			lines = append(lines, "")
    134 		}
    135 	}
    136 
    137 	return strings.Join(lines, "\n"), nil
    138 }

    140 // RenderFileSearchResults renders blocks of file paths, each with a header indicating the source
    141 // repository by which it is grouped.
    142 func (s *stackedResultsRenderer) RenderFileSearchResults(files []webgrep.FileSearchResult) (string, error) {
    143 	var lines []string
    144 	var idx int
    145 
    146 	repoFiles := make(map[string][]webgrep.FileSearchResult)
    147 
    148 	for _, result := range files {
    149 		repoFiles[result.Repository] = append(repoFiles[result.Repository], result)
    150 	}
    151 
    152 	for repo, result := range repoFiles {
    153 		lines = append(lines, cli.Bold(repo))
    154 		lines = append(lines, cli.Bold(strings.Repeat("-", len(repo))))
    155 
    156 		for _, file := range result {
    157 			path := strings.Join([]string{
    158 				file.Path[:file.Bounds[0]],
    159 				cli.Highlight(file.Path[file.Bounds[0]:file.Bounds[1]]),
    160 				file.Path[file.Bounds[1]:],
    161 			}, "")
    162 
    163 			lines = append(lines, path)
    164 		}
    165 
    166 		if idx < len(repoFiles)-1 {
    167 			lines = append(lines, "")
    168 		}
    169 
    170 		idx++
    171 	}
    172 
    173 	return strings.Join(lines, "\n"), nil
    174 }
