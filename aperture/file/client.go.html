     15 // Statsd is an interface for a client providing metrics reporting to a statsd backend.
     16 type Statsd interface {
     17 	Count(name string, delta int64, tags map[string]interface{}) error
     18 	Incr(name string, tags map[string]interface{}) error
     19 	Decr(name string, tags map[string]interface{}) error
     20 	Gauge(name string, value float64, tags map[string]interface{}) error
     21 	Size(name string, bytes int64, tags map[string]interface{}) error
     22 	Timing(name string, duration time.Duration, tags map[string]interface{}) error
     23 	TimingMs(name string, duration float64, tags map[string]interface{}) error
     24 	Histogram(name string, value float64, tags map[string]interface{}) error
     25 	Close() error
     26 }

     34 // NewClient creates a client with the specified configuration.
     35 func NewClient(cfg *Config) (*Client, error) {
     36 	if err := cfg.validate(); err != nil {
     37 		return nil, errors.Stack("client", "client config validation failed", err)
     38 	}
     39 
     40 	/* Transport instantiation */
     41 
     42 	var backends []transport.Factory
     43 
     44 	tFactory := func() (transport.Transport, error) {
     45 		return transport.NewNoop(), nil
     46 	}
     47 
     48 	if cfg.Address != "" {
     49 		for _, addr := range strings.Split(cfg.Address, ",") {
     50 			func(addr string) {
     51 				// Instantiation of the transport client specified by the address
     52 
     53 				uri, _ := data.ParseAddressURI(addr)
     54 
     55 				switch uri.Scheme {
     56 				case "udp":
     57 					tFactory = func() (transport.Transport, error) {
     58 						return transport.NewUDP(uri.Authority)
     59 					}
     60 				case "tcp":
     61 					tFactory = func() (transport.Transport, error) {
     62 						opts := transport.TCPSocketConfig{
     63 							ConnectTimeout:    cfg.TransportConnectTimeout,
     64 							SendTimeout:       cfg.TransportSendTimeout,
     65 							KeepAliveInterval: cfg.TransportKeepaliveInterval,
     66 						}
     67 
     68 						if cfg.Proxy == "" {
     69 							return transport.NewDirectTCP(
     70 								uri.Authority,
     71 								opts,
     72 							)
     73 						}
     74 
     75 						proxy, _ := data.ParseAddressURI(cfg.Proxy)
     76 						return transport.NewProxiedTCP(
     77 							uri.Authority,
     78 							proxy.Scheme,
     79 							proxy.Authority,
     80 							opts,
     81 						)
     82 					}
     83 				case "unix":
     84 					tFactory = func() (transport.Transport, error) {
     85 						return transport.NewUDS(uri.Authority)
     86 					}
     87 				case "file":
     88 					tFactory = func() (transport.Transport, error) {
     89 						return transport.NewFile(uri.Authority)
     90 					}
     91 				}
     92 
     93 				// Transport usage abstractions on top of backends
     94 
     95 				backendTFactory := tFactory
     96 				tFactory = func() (transport.Transport, error) {
     97 					return transport.NewFramed(backendTFactory, []byte("\n"))
     98 				}
     99 
    100 				if cfg.LazyTransportInitialization {
    101 					backendTFactory := tFactory
    102 					tFactory = func() (transport.Transport, error) {
    103 						return transport.NewLazy(backendTFactory)
    104 					}
    105 				}
    106 
    107 				if cfg.TransportProbeInterval > 0 {
    108 					backendTFactory := tFactory
    109 					tFactory = func() (transport.Transport, error) {
    110 						return transport.NewReconnecting(
    111 							backendTFactory,
    112 							cfg.TransportProbeInterval,
    113 							1,
    114 						)
    115 					}
    116 				}
    117 
    118 				backends = append(backends, tFactory)
    119 			}(addr)
    120 		}
    121 	}
    122 
    123 	if len(backends) == 1 {
    124 		tFactory = backends[0]
    125 	} else if len(backends) > 1 {
    126 		tFactory = func() (transport.Transport, error) {
    127 			return transport.NewComposite(backends, transport.All)
    128 		}
    129 	}
    130 
    131 	if cfg.BufferSize > 0 {
    132 		backendTFactory := tFactory
    133 		tFactory = func() (transport.Transport, error) {
    134 			return transport.NewBuffered(backendTFactory, cfg.BufferSize)
    135 		}
    136 	}
    137 
    138 	tport, err := tFactory()
    139 	if err != nil {
    140 		return nil, errors.Stack("client", "failed to create transport", err)
    141 	}
    142 
    143 	/* Configuration defaults */
    144 
    145 	if cfg.Logger == nil {
    146 		cfg.Logger = logger.NewNoop()
    147 	}
    148 
    149 	if cfg.Serializer == nil {
    150 		cfg.Serializer = protocol.DefaultStatsdLineSerializer
    151 	}
    152 
    153 	if cfg.SampleRate == 0.0 {
    154 		cfg.SampleRate = 1.0
    155 	}
    156 
    157 	return &Client{
    158 		cfg:   cfg,
    159 		tport: tport,
    160 	}, nil
    161 }

    163 // Count emits a counter metric.
    164 func (s *Client) Count(name string, delta int64, tags map[string]interface{}) error {
    165 	counter := &protocol.Counter{
    166 		Timestamp:  time.Now(),
    167 		Name:       s.formatName(name),
    168 		Value:      delta,
    169 		SampleRate: s.cfg.SampleRate,
    170 		Tags:       s.mergeTags(tags),
    171 	}
    172 
    173 	s.cfg.Logger.LogCounter(counter)
    174 
    175 	line, err := s.cfg.Serializer.SerializeCounter(counter)
    176 	if err != nil {
    177 		return err
    178 	}
    179 
    180 	if _, err := s.tport.Send([]byte(line)); err != nil {
    181 		return err
    182 	}
    183 
    184 	return nil
    185 }

    220 // Size is used to describe the size of a payload in byte units. In practice, this is merely an
    221 // alias for emitting a timer metric.
    222 func (s *Client) Size(name string, bytes int64, tags map[string]interface{}) error {
    223 	// Size metrics share the same semantics as timing metrics; they are interpreted and
    224 	// aggregated in the same way.
    225 	return s.TimingMs(name, float64(bytes), tags)
    226 }

    286 // formatName adds a prefix and suffix, if specified, to the metric name.
    287 func (s *Client) formatName(metric string) string {
    288 	var components []string
    289 
    290 	if s.cfg.Prefix != "" {
    291 		components = append(components, s.cfg.Prefix)
    292 	}
    293 
    294 	components = append(components, metric)
    295 
    296 	if s.cfg.Suffix != "" {
    297 		components = append(components, s.cfg.Suffix)
    298 	}
    299 
    300 	return strings.Join(components, ".")
    301 }

    303 // mergeTags merges configuration-supplied default tag with the per-metric tags. Metric tags take
    304 // precedence over client-global default tags (i.e., it can override default tags).
    305 func (s *Client) mergeTags(tags map[string]interface{}) map[string]string {
    306 	mergedTags := make(map[string]string)
    307 
    308 	for key, value := range s.cfg.DefaultTags {
    309 		mergedTags[key] = fmt.Sprintf("%v", value)
    310 	}
    311 
    312 	if tags != nil {
    313 		for key, value := range tags {
    314 			mergedTags[key] = fmt.Sprintf("%v", value)
    315 		}
    316 	}
    317 
    318 	return mergedTags
    319 }
