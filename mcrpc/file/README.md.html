      1 # mcrpc
      2 
      3 **mcrpc** is a library for building [memcache protocol](https://github.com/memcached/memcached/blob/master/doc/protocol.txt)
      4 servers with arbitrary backing implementations.
      5 
      6 ## Usage
      7 
      8 Add the library module dependency:
      9 
     10 ```bash
     11 $ go get lib.kevinlin.info/mcrpc
     12 ```
     13 
     14 Define a server handler and start a server:
     15 
     16 ```go
     17 package main
     18 
     19 import (
     20 	"context"
     21 	"net"
     22 
     23 	"lib.kevinlin.info/mcrpc"
     24 	"lib.kevinlin.info/mcrpc/lib"
     25 	"lib.kevinlin.info/mcrpc/protocol"
     26 )
     27 
     28 type dummyHandler struct {
     29 	lib.NoopHandler
     30 }
     31 
     32 func (d *dummyHandler) Version(ctx context.Context, request *protocol.VersionRequest) (*protocol.VersionResponse, error) {
     33 	return &protocol.VersionResponse{Version: "dummy"}, nil
     34 }
     35 
     36 func main() {
     37 	ln, err := net.Listen("tcp", "localhost:11211")
     38 	if err != nil {
     39 		panic(err)
     40 	}
     41 
     42 	srv, err := mcrpc.NewServer(&dummyHandler{})
     43 	if err != nil {
     44 		panic(err)
     45 	}
     46 
     47 	srv.Serve(ln)
     48 }
     49 ```
