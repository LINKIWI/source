      8 // Description of the values, labels, and navigation targets of all tabs, in order.
      9 export const NAVIGATION_TABS = [
     10   { value: 'search', label: 'Search', href: '/search' },
     11   { value: 'admin', label: 'Admin', href: '/admin' },
     12 ];

     14 /**
     15  * Single anchor-enabled tab.

     17 const Tab = ({ label, href, isSelected }) => (
     18   <Link to={href} style={{ textDecoration: 'none' }}>
     19     <Spacing top bottom padding>
     20       <Spacing size="small" left right padding>
     21         <Text
     22           size="kilo"
     23           bold={isSelected}
     24           color={isSelected ? text.light.ALPHA : text.light.BETA}
     25           style={{
     26             opacity: isSelected ? 1 : 0.7,
     27             transition: transition.all.DEFAULT,
     28           }}
     29           uppercase
     30         >
     31           {label}
     32         </Text>
     33       </Spacing>
     34     </Spacing>
     35   </Link>
     36 );

     38 Tab.propTypes = {
     39   label: PropTypes.string.isRequired,
     40   href: PropTypes.string.isRequired,
     41   isSelected: PropTypes.bool.isRequired,
     42 };
     43 
     44 /**
     45  * Set of switchable navigation tabs.
     46  */
     47 const NavigationTabs = ({ tab }) => (
     48   <Tabs
     49     value={tab}
     50     options={NAVIGATION_TABS.map((descriptor) => ({

     52       label: (
     53         <Tab

     55           href={descriptor.href}

     58       ),
     59     }))}
     60     secondary
     61     invert
     62     fit
     63   />
     64 );
     65 
     66 NavigationTabs.propTypes = {
     67   tab: PropTypes.oneOf(NAVIGATION_TABS.map((descriptor) => descriptor.value)),
     68 };
     69 
     70 NavigationTabs.defaultProps = {
     71   tab: null,
     72 };

     74 export default NavigationTabs;
