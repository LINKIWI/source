     44 		return nil, &util.Error{
     45 			Namespace: "config",
     46 			Message:   "error deserializing config file contents",
     47 			Tags: map[string]interface{}{
     48 				"path": path,
     49 			},
     50 			StackedError: err,
     51 		}
     52 	}
     53 
     54 	if err := cfg.Validate(); err != nil {
     55 		return nil, &util.Error{
     56 			Namespace: "config",

     62 		}
     63 	}

    620 				return newTimeoutConn(
    621 					u.Connection.ReadTimeout,
    622 					u.Connection.WriteTimeout,
    623 					conn,
    624 				), nil
    625 			},
    626 		}, nil
    627 	}
    628 }

    630 // ClientConfig transforms the TLS context into a tls.Config appropriate for purposes of TLS
    631 // origination from a client.
    632 func (t *TLSContext) ClientConfig() (*tls.Config, error) {
    633 	var verifyConnection func(tls.ConnectionState) error
    634 	var minVersion uint16
    635 
    636 	cs, err := newCertificateStore(t, func() (*x509.CertPool, error) {
    637 		return x509.SystemCertPool()
    638 	})
    639 	if err != nil {
    640 		return nil, &util.Error{
    641 			Namespace:    "config",
    642 			Message:      "failed to initialize client certificate store",
    643 			StackedError: err,
    644 		}
    645 	}
    646 
    647 	if t.VerifyPeer != nil {
    648 		pv := &peerVerifier{
    649 			pattern: t.VerifyPeer,
    650 			roots:   cs.cas,
    651 		}
    652 
    653 		verifyConnection = pv.verifyServer
    654 	}
    655 
    656 	switch t.ProtocolVersion {
    657 	case "lenient":
    658 		minVersion = tls.VersionTLS11
    659 	case "standard":
    660 		minVersion = tls.VersionTLS12
    661 	case "modern":
    662 		minVersion = tls.VersionTLS13
    663 	default:
    664 		minVersion = tls.VersionTLS12
    665 	}
    666 
    667 	return &tls.Config{
    668 		GetClientCertificate: cs.getClientCertificate,
    669 		RootCAs:              cs.cas,
    670 		// Defer to the overridden connection verification in place of default validation
    671 		// when explicit peer verification is enabled.
    672 		InsecureSkipVerify: t.VerifyMode == "none" ||
    673 			t.VerifyMode == "relaxed" ||
    674 			t.VerifyPeer != nil,
    675 		MinVersion:         minVersion,
    676 		ClientSessionCache: tls.NewLRUClientSessionCache(128),
    677 		VerifyConnection:   verifyConnection,
    678 	}, nil
    679 }

    681 // ServerConfig transforms the TLS context into a tls.Config appropriate for purposes of TLS
    682 // termination at a server.
    683 func (t *TLSContext) ServerConfig() (*tls.Config, error) {
    684 	var verifyMode tls.ClientAuthType
    685 	var verifyConnection func(tls.ConnectionState) error
    686 	var minVersion uint16
    687 
    688 	// Explicitly use a nil (empty) seed CA certificate pool in order to use only the CAs
    689 	// specified in TLS configuration for client certificate authentication.
    690 	cs, err := newCertificateStore(t, nil)
    691 	if err != nil {
    692 		return nil, &util.Error{
    693 			Namespace:    "config",
    694 			Message:      "failed to initialize server certificate store",
    695 			StackedError: err,
    696 		}
    697 	}
    698 
    699 	switch t.VerifyMode {
    700 	case "none":
    701 		verifyMode = tls.NoClientCert
    702 	case "relaxed":
    703 		verifyMode = tls.VerifyClientCertIfGiven
    704 	case "strict":
    705 		verifyMode = tls.RequireAndVerifyClientCert
    706 	default:
    707 		verifyMode = tls.NoClientCert
    708 	}
    709 
    710 	if t.VerifyPeer != nil {
    711 		pv := &peerVerifier{
    712 			pattern: t.VerifyPeer,
    713 			roots:   cs.cas,
    714 		}
    715 
    716 		verifyConnection = pv.verifyClient
    717 	}
    718 
    719 	switch t.ProtocolVersion {
    720 	case "lenient":
    721 		minVersion = tls.VersionTLS11
    722 	case "standard":
    723 		minVersion = tls.VersionTLS12
    724 	case "modern":
    725 		minVersion = tls.VersionTLS13
    726 	default:
    727 		minVersion = tls.VersionTLS12
    728 	}
    729 
    730 	return &tls.Config{
    731 		GetCertificate:   cs.getServerCertificate,
    732 		ClientCAs:        cs.cas,
    733 		ClientAuth:       verifyMode,
    734 		MinVersion:       minVersion,
    735 		VerifyConnection: verifyConnection,
    736 	}, nil
    737 }
