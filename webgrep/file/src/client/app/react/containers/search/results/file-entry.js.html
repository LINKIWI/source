     60   _handleFilePathClick() {
     61     const { file, position, recordTelemetryEvent } = this.props;
     62 
     63     recordTelemetryEvent(TELEMETRY_ACTIONS.CLICK_FILE_RESULT_PATH, 1, { repo: file.repo });
     64     recordTelemetryEvent(
     65       TELEMETRY_ACTIONS.CLICK_FILE_RESULT_POSITION,
     66       position,
     67       { repo: file.repo },
     68     );
     69   }
     70 
     71   _handleFilePathCopy() {
     72     const { file, clipboard, toast, recordTelemetryEvent } = this.props;
     73 
     74     recordTelemetryEvent(TELEMETRY_ACTIONS.CLIPBOARD_WRITE);
     75     clipboard.write(file.path, (err) => toast(err ?
     76       'There was an error writing to the system clipboard.' :
     77       `Copied to clipboard: ${file.path}`));
     78   }
     79 
     80   _handleSourcePreview() {
     81     const { recordTelemetryEvent, showSourcePreview } = this.props;
     82 
     83     recordTelemetryEvent(TELEMETRY_ACTIONS.SOURCE_PREVIEW);
     84     showSourcePreview();
     85   }
     86 
     87   _handleSingleFileSearch() {
     88     const { file, toast, onSearchRepositoryFilePathChange } = this.props;
     89 
     90     onSearchRepositoryFilePathChange(file.repo, file.path);
     91     toast(`Applied search filters: repository ${file.repo} and file path ${file.path}.`);
     92     scroll();
     93   }

     95   render() {
     96     const {
     97       file,
     98       repositories,
     99       windowWidth,
    100       isOverflowMenuVisible,

    116     const containerStyle = {
