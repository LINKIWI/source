     32 // Validate validates the parsed configuration struct and returns a non-nil error on validation
     33 // errors.
     34 func (c *Config) Validate() error {
     35 	for _, relay := range c.Relays {
     36 		if relay.Name == "" {
     37 			return fmt.Errorf("config: relay name is not populated")
     38 		} else if relay.LogFile.Pattern == "" {
     39 			return fmt.Errorf("config: relay log file path is not populated")
     40 		} else if len(relay.Delimiter) > 1 {
     41 			return fmt.Errorf("config: relay delimiter must be omitted or exactly one character")
     42 		} else if relay.BufferLength < 0 {
     43 			return fmt.Errorf("config: buffer length must be a nonnegative integer")
     44 		} else if _, _, err := net.SplitHostPort(relay.ProxyAddress); relay.ProxyAddress != "" && err != nil {
     45 			return fmt.Errorf("config: failed to parse proxy server address: err=%v", err)
     46 		} else if _, _, err := net.SplitHostPort(relay.KafkaAddress); err != nil {
     47 			return fmt.Errorf("config: failed to parse Kafka address: err=%v", err)
     48 		}
     49 
     50 		switch relay.Serializer {
     51 		case "", SerializerLine, SerializerJSON:
     52 		default:
     53 			return fmt.Errorf("config: unknown relay serializer identifier")
     54 		}
     55 
     56 		switch relay.SeekPosition {
     57 		case "", SeekPositionStart, SeekPositionEnd:
     58 		default:
     59 			return fmt.Errorf("config: unknown seek position identifier")
     60 		}
     61 
     62 		for _, tagID := range relay.TagIdentifiers {
     63 			switch tagID {
     64 			case TagHostname, TagVersion:
     65 			default:
     66 				return fmt.Errorf("config: unknown relay tag identifier: tag=%s", tagID)
     67 			}
     68 		}
     69 	}
     70 
     71 	return nil
     72 }
