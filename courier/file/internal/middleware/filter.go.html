     55 // ProcessRequest chains the request through each filter sequentially.
     56 func (c *composedFilter) ProcessRequest(clientReq *http.Request) (*http.Request, *http.Response, error) {
     57 	var processed []Filter
     58 	var resp *http.Response
     59 	var err error
     60 
     61 	proxyReq := clientReq
     62 	tags := map[string]interface{}{
     63 		"filter":     "composed",
     64 		"route_host": proxyReq.Host,
     65 		"method":     proxyReq.Method,
     66 		"protocol":   proxyReq.Proto,
     67 	}
     68 
     69 	defer func() {
     70 		metrics.Client.Size(metricFilterCardinality, int64(len(processed)), tags)
     71 	}()
     72 
     73 	// Pass the request incrementally into all filters in order, but terminate early if any of
     74 	// them errors or requests a response be sent back to the client immediately.
     75 	for _, filter := range c.filters {
     76 		proxyReq, resp, err = filter.ProcessRequest(proxyReq)
     77 
     78 		if resp == nil && err == nil {
     79 			// As long as the request processing didn't error and was not terminated
     80 			// early (i.e. the request was either passed along without error), it should
     81 			// be queued for future response processing.
     82 			// Prepend the successfully processed filter so that response processing can
     83 			// occur in reverse stack order after the reverse proxy completes.
     84 			processed = append([]Filter{filter}, processed...)
     85 		}
     86 
     87 		if proxyReq != nil {
     88 			proxyReq = proxyReq.WithContext(
     89 				context.WithValue(
     90 					proxyReq.Context(),
     91 					ctxComposedProcessedFilters,
     92 					processed,
     93 				),
     94 			)
     95 		}
     96 
     97 		if resp != nil && resp.Request != nil {
     98 			resp.Request = resp.Request.WithContext(
     99 				context.WithValue(
    100 					resp.Request.Context(),
    101 					ctxComposedProcessedFilters,
    102 					processed,
    103 				),
    104 			)
    105 		}
    106 
    107 		if err != nil || resp != nil {
    108 			return proxyReq, resp, err
    109 		}
    110 	}
    111 
    112 	return proxyReq, resp, err
    113 }

    115 // ProcessResponse chains the response through each filter sequentially.
    116 func (c *composedFilter) ProcessResponse(proxyResp *http.Response) (*http.Response, error) {
    117 	var err error
    118 
    119 	clientResp := proxyResp
    120 
    121 	processed, ok := clientResp.Request.Context().Value(ctxComposedProcessedFilters).([]Filter)
    122 	if !ok {
    123 		return clientResp, nil
    124 	}
    125 
    126 	for _, filter := range processed {
    127 		clientResp, err = filter.ProcessResponse(clientResp)
    128 		if err != nil {
    129 			return clientResp, err
    130 		}
    131 	}
    132 
    133 	return clientResp, err
    134 }

    136 // ProcessResponse runs the underlying filter's ProcessResponse asynchronously and immediately

    138 func (a *asyncFilter) ProcessResponse(proxyResp *http.Response) (*http.Response, error) {
    139 	go a.Filter.ProcessResponse(proxyResp)

    144 // ProcessRequest runs the underlying filter's ProcessRequest and emits a counter and timer.
    145 func (i *instrumentedFilter) ProcessRequest(clientReq *http.Request) (*http.Request, *http.Response, error) {
    146 	stopwatch := lib.NewStopwatch()
    147 	tags := map[string]interface{}{
    148 		"filter":     i.name,
    149 		"route_host": clientReq.Host,
    150 		"method":     clientReq.Method,
    151 		"protocol":   clientReq.Proto,
    152 	}
    153 
    154 	defer func() {
    155 		metrics.Client.Timing(metricFilterRequestDuration, stopwatch.Elapsed(), tags)
    156 	}()
    157 
    158 	metrics.Client.Incr(metricFilterRequestProcess, tags)
    159 
    160 	return i.Filter.ProcessRequest(clientReq)
    161 }
