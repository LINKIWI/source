     85   _handleQueryChange(evt, cb) {
     86     const { form, initialMatchLimit, handleFormChange } = this.props;
     87 
     88     // Changes to the search query should reset the number of max matches when it has only seen
     89     // incremental changes (i.e. loading more search results). A special case is when the max
     90     // matches limit is set to "all matches," in which case it should not be altered.
     91 
     92     if (form.maxMatches === MAX_MATCHES.UNLIMITED) {
     93       handleFormChange('query')(evt);
     94       this.invoke({ query: evt.target.value }, cb);
     95     } else {
     96       this.props.handleFormChange('query')(evt);
     97       this.props.handleFormChange('maxMatches')(initialMatchLimit);
     98       this.invoke({ query: evt.target.value, maxMatches: initialMatchLimit }, cb);
     99     }
    100   }
    101 
    102   _handleRegexChange(state, cb) {
    103     this.props.handleFormChange('regex')(state);
    104     this.invoke({ regex: state }, cb);
    105   }
    106 
    107   _handleCaseSensitivityChange(state, cb) {
    108     this.props.handleFormChange('caseSensitive')(state);
    109     this.invoke({ caseSensitive: state }, cb);
    110   }
    111 
    112   _handleMaxMatchesChange(value, cb) {
    113     const { initialMatchLimit, handleFormChange } = this.props;
    114 
    115     // null max matches implies resetting to the default/initial match limit, as supplied in
    116     // preferences.
    117 
    118     if (value !== null) {
    119       handleFormChange('maxMatches')(value);
    120       this.invoke({ maxMatches: value }, cb);
    121     } else {
    122       handleFormChange('maxMatches')(initialMatchLimit);
    123       this.invoke({ maxMatches: initialMatchLimit }, cb);
    124     }
    125   }
    126 
    127   _handleFilePathChange(evt, cb) {
    128     this.props.handleFormChange('file')(evt);
    129     this.invoke({ file: evt.target.value }, cb);
    130   }
    131 
    132   _handleRepositoriesChange(repos, cb) {
    133     this.props.handleFormChange('repos')(repos);
    134     this.invoke({ repos }, cb);
    135   }

    137   _invoke(overrides = {}, cb) {
    138     const { results, form, indexIdentity, contextLines, initialMatchLimit } = this.props;
    139     const searchParams = { ...form, ...overrides };
    140 
    141     // Only re-encode URL state when a form parameter changes
    142     if (Object.keys(overrides).length) {
    143       encodeURLState(
    144         searchParams.query && searchParams,
    145         // Conditionally serialize request parameters into URL state, to keep the URL shorter
    146         {
    147           regex: (val) => val,
    148           caseSensitive: (val) => val,
    149           repos: (val) => val.length > 0,
    150           maxMatches: (val) => val !== initialMatchLimit,
    151           file: (val) => val,
    152         },
    153         // Special case serialization for non-primitive field values
    154         {
    155           repos: URLStateSerializer.array,
    156           query: URLStateSerializer.string,
    157           file: URLStateSerializer.string,
    158         },
    159       );
    160     }
    161 
    162     const request = {
    163       ...searchParams,
    164       // The index identity is used to "fingerprint" the client's search request context.
    165       // Any discrepancy between the client-asserted and server-asserted index identities implies
    166       // that the search index has changed since the client last updated its knowledge of the
    167       // server-side index metadata.
    168       indexIdentity,
    169       // Customize the number of context lines based on the global preference value.
    170       context: contextLines,
    171     };
    172 
    173     results.sendMessage(request);
    174 
    175     if (cb) {
    176       this.forceUpdate(cb);
    177     }
    178   }

    180   render() {
    181     const { results, form, title, width, actions } = this.props;
    182     const searchTitle = form.query ? `${form.query} - ${title}` : title;
    183     const isCompact = breakpoints.reduce((acc, breakpoint) =>
    184       (width < breakpoint.threshold ? breakpoint.compact : acc), false);
    185     const connectivity = (() => {
    186       if (results.err) {
    187         return CONNECTIVITY_STATE.DISCONNECTED;
    188       }
    189 
    190       if (results.isConnected) {
    191         return CONNECTIVITY_STATE.CONNECTED;
    192       }
    193 
    194       return CONNECTIVITY_STATE.CONNECTING;
    195     })();
    196 
    197     return (
    198       <Fragment>
    199         <Helmet>
    200           <title>
    201             {searchTitle}
    202           </title>
    203         </Helmet>
    204 
    205         <Layout width={width}>
    206           <Spacing size="huge" bottom>
    207             <Spacing size="large" top>
    208               <SearchQueryContainer
    209                 value={form.query}
    210                 regex={form.regex}
    211                 caseSensitive={form.caseSensitive}
    212                 maxMatches={form.maxMatches}
    213                 filePath={form.file}
    214                 filteredRepos={form.repos}
    215                 results={results.message}
    216                 onQueryChange={this.handleQueryChange}
    217                 onRegexChange={this.handleRegexChange}
    218                 onCaseSensitivityChange={this.handleCaseSensitivityChange}
    219                 onMaxMatchesChange={this.handleMaxMatchesChange}
    220                 onFilePathChange={this.handleFilePathChange}
    221                 onRepositoriesChange={this.handleRepositoriesChange}
    222                 onQueryRecord={actions.recordSearchHistoryItem}
    223                 connectivity={connectivity}
    224                 isCompact={isCompact}
    225               />
    226             </Spacing>
    227           </Spacing>
    228 
    229           <Spacing size="huge" bottom>
    230             <SearchResultsContainer
    231               maxMatches={form.maxMatches}
    232               results={results.message}
    233               onQueryChange={this.handleQueryChange}
    234               onQueryRecord={actions.recordSearchHistoryItem}
    235               onMaxMatchesChange={this.handleMaxMatchesChange}
    236               onFilePathChange={this.handleFilePathChange}
    237               onRepositoriesChange={this.handleRepositoriesChange}
    238             />
    239           </Spacing>
    240         </Layout>
    241       </Fragment>
    242     );
    243   }
