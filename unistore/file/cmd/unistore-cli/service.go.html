    132 // runDownload is a command line interface for the GetObject and StreamGetObject RPCs.
    133 func runDownload(cmd *cobra.Command, args []string) error {
    134 	rdr := newRenderer(flagFormat, os.Stderr)
    135 	bucket, _ := cmd.Flags().GetString("bucket")
    136 	rangeStart, _ := cmd.Flags().GetUint64("range-start")
    137 	rangeEnd, _ := cmd.Flags().GetUint64("range-end")
    138 	stream, _ := cmd.Flags().GetBool("stream")
    139 	chunkSize, _ := cmd.Flags().GetUint64("chunk-size")
    140 	progress, _ := cmd.Flags().GetBool("progress")
    141 
    142 	store, rpc, err := newClientConfig(flagConfig, flagStore)
    143 	if err != nil {
    144 		return err
    145 	}
    146 
    147 	client, err := newUnistoreClient(store, rpc)
    148 	if err != nil {
    149 		return err
    150 	}
    151 
    152 	get := &service.GetObjectRequest{
    153 		Resource: &common.Resource{
    154 			Backend: common.Backend(common.Backend_value[strings.ToUpper(store.Backend)]),
    155 			Bucket:  bucket,
    156 		},
    157 		Key: args[0],
    158 	}
    159 
    160 	if rangeStart > 0 || rangeEnd > 0 {
    161 		get.Range = &common.Range{
    162 			Unit:  "bytes",
    163 			Start: rangeStart,
    164 			End:   rangeEnd,
    165 		}
    166 	}
    167 
    168 	if !stream {
    169 		if flagVerbose {
    170 			rdr.proto(get)
    171 		}
    172 
    173 		response, err := client.GetObject(context.Background(), get)
    174 		if err != nil {
    175 			return err
    176 		}
    177 
    178 		if flagVerbose {
    179 			rdr.proto(response)
    180 		}
    181 
    182 		_, err = os.Stdout.Write(response.Data)
    183 		return err
    184 	}
    185 
    186 	request := &service.GetObjectStreamRequest{
    187 		Request:   get,
    188 		ChunkSize: chunkSize,
    189 	}
    190 
    191 	if flagVerbose {
    192 		rdr.proto(request)
    193 	}
    194 
    195 	response, err := client.StreamGetObject(context.Background(), request)
    196 	if err != nil {
    197 		return err
    198 	}
    199 
    200 	start := time.Now()
    201 	downloaded := new(uint64)
    202 
    203 	if progress {
    204 		interval := 5 * time.Second
    205 		ticker := time.NewTicker(interval)
    206 		defer ticker.Stop()
    207 
    208 		go func() {
    209 			prev := atomic.LoadUint64(downloaded)
    210 
    211 			for range ticker.C {
    212 				current := atomic.LoadUint64(downloaded)
    213 				fmt.Fprintf(
    214 					os.Stderr,
    215 					"%v: downloaded %s (%s/s)\n",
    216 					time.Since(start).Round(interval),
    217 					humanize.Bytes(current),
    218 					humanize.Bytes((current-prev)/5),
    219 				)
    220 				prev = current
    221 			}
    222 		}()
    223 	}
    224 
    225 	for {
    226 		message, err := response.Recv()
    227 		if err == io.EOF {
    228 			break
    229 		} else if err != nil {
    230 			return err
    231 		}
    232 
    233 		atomic.AddUint64(downloaded, uint64(len(message.Response.Data)))
    234 
    235 		if flagVerbose {
    236 			rdr.proto(message)
    237 		}
    238 
    239 		if _, err = os.Stdout.Write(message.Response.Data); err != nil {
    240 			return err
    241 		}
    242 	}
    243 
    244 	if progress {
    245 		fmt.Fprintf(
    246 			os.Stderr,
    247 			"%v: completed download of %s\n",
    248 			time.Since(start),
    249 			humanize.Bytes(atomic.LoadUint64(downloaded)),
    250 		)
    251 	}
    252 
    253 	return nil
    254 }

    256 // runUpload is a command line interface for the PutObject and StreamPutObject RPCs.
    257 func runUpload(cmd *cobra.Command, args []string) error {
    258 	rdr := newRenderer(flagFormat, os.Stdout)
    259 	bucket, _ := cmd.Flags().GetString("bucket")
    260 	timestamp, _ := cmd.Flags().GetUint64("timestamp")
    261 	stream, _ := cmd.Flags().GetBool("stream")
    262 	chunkSize, _ := cmd.Flags().GetUint64("chunk-size")
    263 	checksum, _ := cmd.Flags().GetString("checksum")
    264 	progress, _ := cmd.Flags().GetBool("progress")
    265 
    266 	store, rpc, err := newClientConfig(flagConfig, flagStore)
    267 	if err != nil {
    268 		return err
    269 	}
    270 
    271 	client, err := newUnistoreClient(store, rpc)
    272 	if err != nil {
    273 		return err
    274 	}
    275 
    276 	attributes := &common.Attributes{}
    277 	if timestamp > 0 {
    278 		attributes.Timestamp = timestamppb.New(time.Unix(0, int64(timestamp)))
    279 	}
    280 
    281 	put := &service.PutObjectRequest{
    282 		Resource: &common.Resource{
    283 			Backend: common.Backend(common.Backend_value[strings.ToUpper(store.Backend)]),
    284 			Bucket:  bucket,
    285 		},
    286 		Key:        args[0],
    287 		Checksum:   checksum,
    288 		Attributes: attributes,
    289 	}
    290 
    291 	if !stream {
    292 		data, err := io.ReadAll(os.Stdin)
    293 		if err != nil {
    294 			return err
    295 		}
    296 
    297 		put.Data = data
    298 
    299 		if flagVerbose {
    300 			rdr.proto(put)
    301 		}
    302 
    303 		response, err := client.PutObject(context.Background(), put)
    304 		if err != nil {
    305 			return err
    306 		}
    307 
    308 		return rdr.proto(response)
    309 	}
    310 
    311 	request, err := client.StreamPutObject(context.Background())
    312 	if err != nil {
    313 		return err
    314 	}
    315 
    316 	start := time.Now()
    317 	uploaded := new(uint64)
    318 
    319 	if progress {
    320 		interval := 5 * time.Second
    321 		ticker := time.NewTicker(interval)
    322 		defer ticker.Stop()
    323 
    324 		go func() {
    325 			prev := atomic.LoadUint64(uploaded)
    326 
    327 			for range ticker.C {
    328 				current := atomic.LoadUint64(uploaded)
    329 				fmt.Printf(
    330 					"%v: uploaded %s (%s/s)\n",
    331 					time.Since(start).Round(interval),
    332 					humanize.Bytes(current),
    333 					humanize.Bytes((current-prev)/5),
    334 				)
    335 				prev = current
    336 			}
    337 		}()
    338 	}
    339 
    340 	for {
    341 		chunk := make([]byte, chunkSize)
    342 		n, err := os.Stdin.Read(chunk)
    343 		if err == io.EOF {
    344 			break
    345 		} else if err != nil {
    346 			return err
    347 		}
    348 
    349 		message := &service.PutObjectStreamRequest{Request: put}
    350 		message.Request.Data = chunk[:n]
    351 
    352 		atomic.AddUint64(uploaded, uint64(n))
    353 
    354 		if flagVerbose {
    355 			rdr.proto(message)
    356 		}
    357 
    358 		if err := request.Send(message); err != nil {
    359 			return err
    360 		}
    361 	}
    362 
    363 	response, err := request.CloseAndRecv()
    364 	if err != nil {
    365 		return err
    366 	}
    367 
    368 	if progress {
    369 		fmt.Printf(
    370 			"%v: completed upload of %s\n",
    371 			time.Since(start),
    372 			humanize.Bytes(atomic.LoadUint64(uploaded)),
    373 		)
    374 	}
    375 
    376 	return rdr.proto(response)
    377 }
