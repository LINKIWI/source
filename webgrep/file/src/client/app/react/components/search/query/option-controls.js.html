     11 // Reasonably robust regular expression patterns for capturing various components of a file path
     12 // that may be interesting as an automatically populated file path filter suggestion.
     13 const ROOT_DIRECTORY_PATTERN = /^([^/]+\/)/;
     14 const FILE_EXTENSION_PATTERN = /(\.[^./]+)$/;
     15 
     16 /**
     17  * Controls for query options.
     18  */
     19 class QueryOptionControls extends Component {
     20   static propTypes = {
     21     regex: PropTypes.bool.isRequired,
     22     caseSensitive: PropTypes.bool.isRequired,
     23     maxMatches: PropTypes.number.isRequired,
     24     filePath: PropTypes.string.isRequired,
     25     repositories: PropTypes.array.isRequired,
     26     searchResults: PropTypes.arrayOf(PropTypes.shape({
     27       path: PropTypes.string.isRequired,
     28     })).isRequired,
     29     onRegexChange: PropTypes.func.isRequired,
     30     onCaseSensitivityChange: PropTypes.func.isRequired,
     31     onMaxMatchesChange: PropTypes.func.isRequired,
     32     onFilePathChange: PropTypes.func.isRequired,
     33     onRepositoryToggle: PropTypes.func.isRequired,
     34     isCompact: PropTypes.bool.isRequired,
     35     // HOC props
     36     isPathPatternFlyoutVisible: PropTypes.bool.isRequired,
     37     showPathPatternFlyout: PropTypes.func.isRequired,
     38     hidePathPatternFlyout: PropTypes.func.isRequired,
     39     isRepositoriesFlyoutVisible: PropTypes.bool.isRequired,
     40     showRepositoriesFlyout: PropTypes.func.isRequired,
     41     hideRepositoriesFlyout: PropTypes.func.isRequired,
     42   };

     44   componentDidUpdate(prevProps) {
     45     // Automatically focus the file path and repository filter fields when the flyout is opened.
     46     // It's hackily wrapped in a setTimeout to work around a race condition with the toggle stealing
     47     // focus on click.
     48 
     49     if (
     50       this.props.isPathPatternFlyoutVisible &&
     51       !prevProps.isPathPatternFlyoutVisible &&
     52       this.filePathFieldRef.current
     53     ) {
     54       window.setTimeout(() => this.filePathFieldRef.current.focus(), 100);
     55     }
     56 
     57     if (
     58       this.props.isRepositoriesFlyoutVisible &&
     59       !prevProps.isRepositoriesFlyoutVisible &&
     60       this.repositoriesFieldRef.current
     61     ) {
     62       window.setTimeout(() => this.repositoriesFieldRef.current.focus(), 100);
     63     }
     64   }
     65 
     66   filePathToggleRef = createRef();
     67 
     68   filePathFieldRef = createRef();
     69 
     70   repositoriesToggleRef = createRef();
     71 
     72   repositoriesFieldRef = createRef();
     73 
     74   render() {
     75     const {
     76       regex,
     77       caseSensitive,
     78       maxMatches,
     79       filePath,
     80       repositories,
     81       searchResults,
     82       onRegexChange,
     83       onCaseSensitivityChange,
     84       onMaxMatchesChange,
     85       onFilePathChange,
     86       onRepositoryToggle,
     87       isRepositoriesFlyoutVisible,
     88       showRepositoriesFlyout,
     89       hideRepositoriesFlyout,
     90       isPathPatternFlyoutVisible,
     91       showPathPatternFlyout,
     92       hidePathPatternFlyout,
     93       isCompact,
     94     } = this.props;
     95 
     96     const filteredRepos = repositories.filter((repo) => repo.isSelected);
     97 
     98     const suggestions = isPathPatternFlyoutVisible && (() => {
     99       // Data manipulation routine to match each result path against a regular expression, and order
    100       // the resulting capturing groups by descending order of occurrence frequency.
    101       const orderByRegexMatchFrequency = (pattern) => {
    102         const frequency = searchResults
    103           // Match all paths against the desired pattern, extracting the first capture group as long
    104           // as it is available.
    105           .map((result) => result.path.match(pattern))
    106           .filter((match) => match && match.length)
    107           .map(([_, capture]) => capture)
    108           // Transform it into a mapping of file extension to its occurrence frequency.
    109           .reduce((freq, capture) => ({
    110             ...freq,
    111             [capture]: (freq[capture] || 0) + 1,
    112           }), {});
    113 
    114         return Object.keys(frequency)
    115           // Rank the extensions in descending order of occurrence frequency.
    116           .sort((a, b) => frequency[b] - frequency[a]);
    117       };
    118 
    119       const directories = orderByRegexMatchFrequency(ROOT_DIRECTORY_PATTERN).slice(0, 5);
    120       const extensions = orderByRegexMatchFrequency(FILE_EXTENSION_PATTERN).slice(0, 5);
    121 
    122       // It's not particularly useful to provide suggestions if all current search results already
    123       // share the same extension.
    124       return [
    125         ...directories.length > 1 ? directories : [],
    126         ...extensions.length > 1 ? extensions : [],
    127       ];
    128     })();
    129 
    130     const containerStyle = {
    131       display: 'flex',
    132       ...isCompact && {
    133         flexWrap: 'wrap',
    134         marginBottom: '-10px',
    135       },
    136     };
    137 
    138     const toggleStyle = {
    139       marginBottom: 0,
    140       ...isCompact && {
    141         marginBottom: '10px',
    142       },
    143     };
    144 
    145     const flyoutToggleStyle = {
    146       position: 'relative',
    147       ...isCompact && {
    148         position: 'unset',
    149       },
    150     };
    151 
    152     return (
    153       <div style={containerStyle}>
    154         <Spacing size="small" style={toggleStyle} right>
    155           <Toggle
    156             isActive={regex}
    157             description="Search using regular expressions"
    158             onClick={() => onRegexChange(!regex)}
    159           >
    160             Regex
    161           </Toggle>
    162         </Spacing>
    163 
    164         <Spacing size="small" style={toggleStyle} right>
    165           <Toggle
    166             isActive={caseSensitive}
    167             description="Respect text casing"
    168             onClick={() => onCaseSensitivityChange(!caseSensitive)}
    169           >
    170             Case sensitive
    171           </Toggle>
    172         </Spacing>
    173 
    174         <Spacing size="small" style={flyoutToggleStyle} right>
    175           <Toggle
    176             ref={this.filePathToggleRef}
    177             isActive={filePath.length > 0}
    178             description="File path pattern to filter"
    179             onClick={isPathPatternFlyoutVisible ? hidePathPatternFlyout : showPathPatternFlyout}
    180           >
    181             File path
    182           </Toggle>
    183 
    184           <Flyout
    185             triggerRef={this.filePathToggleRef}
    186             isOpen={isPathPatternFlyoutVisible}
    187             isCompact={isCompact}
    188             onHide={hidePathPatternFlyout}
    189           >
    190             <FilePathFilter
    191               ref={this.filePathFieldRef}
    192               value={filePath}
    193               suggestions={suggestions || []}
    194               onChange={onFilePathChange}
    195               onHide={hidePathPatternFlyout}
    196             />
    197           </Flyout>
    198         </Spacing>
    199 
    200         <Spacing size="small" style={flyoutToggleStyle} right>
    201           <Toggle
    202             ref={this.repositoriesToggleRef}
    203             isActive={filteredRepos.length > 0}
    204             description={filteredRepos.length ?
    205               [
    206                 'Including',
    207                 filteredRepos.length,
    208                 filteredRepos.length === 1 ? 'repository' : 'repositories',
    209               ].join(' ') :
    210               'Select repositories to filter'}
    211             onClick={isRepositoriesFlyoutVisible ? hideRepositoriesFlyout : showRepositoriesFlyout}
    212           >
    213             Repositories
    214           </Toggle>
    215 
    216           <Flyout
    217             triggerRef={this.repositoriesToggleRef}
    218             isOpen={isRepositoriesFlyoutVisible}
    219             isCompact={isCompact}
    220             onHide={hideRepositoriesFlyout}
    221           >
    222             <RepositoryFilter
    223               ref={this.repositoriesFieldRef}
    224               repositories={repositories}
    225               onRepositoryToggle={onRepositoryToggle}
    226               onHide={hideRepositoriesFlyout}
    227             />
    228           </Flyout>
    229         </Spacing>
    230 
    231         <div style={toggleStyle}>
    232           <Toggle
    233             isActive={maxMatches === MAX_MATCHES.UNLIMITED}
    234             description="Exhaust all results (slow)"
    235             onClick={() => onMaxMatchesChange(maxMatches === MAX_MATCHES.UNLIMITED ?
    236               null : MAX_MATCHES.UNLIMITED)}
    237           >
    238             All matches
    239           </Toggle>
    240         </div>
    241       </div>
    242     );
    243   }
    244 }
