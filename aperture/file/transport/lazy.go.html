      7 // Lazy is an abstraction over a Transport that initializes the underlying transport only when the
      8 // client attempts to make its first transmission, rather than at construction time.
      9 type Lazy struct {
     10 	backendFactory Factory
     11 	mutex          sync.Mutex
     12 	Transport
     13 }
     14 
     15 // NewLazy creates a Transport that lazily initializes the underlying network transport client.
     16 func NewLazy(backendFactory Factory) (Transport, error) {
     17 	return &Lazy{backendFactory: backendFactory}, nil
     18 }
     19 
     20 // Send idempotently initializes the underlying transport, followed by using the transport to send
     21 // the requested data.
     22 func (t *Lazy) Send(data []byte) (int, error) {
     23 	if err := t.init(); err != nil {
     24 		return 0, err
     25 	}
     26 
     27 	return t.Transport.Send(data)
     28 }
     29 
     30 // Close defers to the underlying transport implementation, if initialized, to close the transport.
     31 func (t *Lazy) Close() error {
     32 	if t.Transport == nil {
     33 		return nil
     34 	}
     35 
     36 	return t.Transport.Close()
     37 }
     38 
     39 // init statefully but idempotently initializes the underlying transport. This method noops if the
     40 // transport is already initialized; otherwise, it initializes and sets the transport for future
     41 // reuse on subsequent transmissions. The transport is not set in case of initialization errors.
     42 func (t *Lazy) init() error {
     43 	t.mutex.Lock()
     44 	defer t.mutex.Unlock()
     45 
     46 	// Transport is already initialized; abort immediately.
     47 	if t.Transport != nil {
     48 		return nil
     49 	}
     50 
     51 	tport, err := t.backendFactory()
     52 	if err != nil {
     53 		return err
     54 	}
     55 
     56 	t.Transport = tport
     57 
     58 	return nil
     59 }
