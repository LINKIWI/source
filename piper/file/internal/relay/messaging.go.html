     13 // Producer is an abstraction over a sarama.SyncProducer that exposes a limited API surface.
     14 type Producer struct {
     15 	client     sarama.SyncProducer
     16 	serializer MessageSerializer

     19 // NewProducer creates a new Kafka producer for a particular target broker.
     20 func NewProducer(
     21 	serializer MessageSerializer,
     22 	socks5 string,
     23 	address string,
     24 	acks int,
     25 	retries int,
     26 	timeout time.Duration,
     27 ) (*Producer, error) {
     28 	kafkaCfg := sarama.NewConfig()
     29 	kafkaCfg.Producer.RequiredAcks = sarama.RequiredAcks(acks)
     30 	kafkaCfg.Producer.Retry.Backoff = 5 * time.Second
     31 	kafkaCfg.Producer.Return.Successes = true
     32 	kafkaCfg.Producer.Return.Errors = true
     33 	if retries != 0 {
     34 		kafkaCfg.Producer.Retry.Max = retries
     35 	}
     36 	if timeout != 0 {
     37 		kafkaCfg.Producer.Timeout = timeout
     38 	}
     39 	if socks5 != "" {
     40 		dialer, err := proxy.SOCKS5("tcp", socks5, nil, proxy.Direct)
     41 		if err != nil {
     42 			return nil, err
     43 		}
     44 
     45 		kafkaCfg.Net.Proxy.Enable = true
     46 		kafkaCfg.Net.Proxy.Dialer = dialer
     47 	}
     48 
     49 	client, err := sarama.NewSyncProducer([]string{address}, kafkaCfg)
     50 	if err != nil {
     51 		return nil, fmt.Errorf("relay: failed to create client: err=%v", err)
     52 	}
     53 
     54 	return &Producer{
     55 		client:     client,
     56 		serializer: serializer,
     57 	}, nil
     58 }

     60 // WriteMessage synchronously writes a tail.Message to a particular Kafka topic, blocking until the
     61 // write has been sent and has reached the Kafka acknowledgement level specified in configuration.
     62 // It returns the length of the produced content in bytes and an error as applicable.
     63 func (p *Producer) WriteMessage(topic string, message tail.Message) (int, error) {
     64 	serialized, err := p.serializer.Serialize(message)
     65 	if err != nil {
     66 		return 0, err
     67 	}
     68 
     69 	value := sarama.ByteEncoder(serialized)
     70 
     71 	if _, _, err := p.client.SendMessage(&sarama.ProducerMessage{
     72 		Topic:     topic,
     73 		Value:     value,
     74 		Timestamp: message.Timestamp,
     75 	}); err != nil {
     76 		return 0, fmt.Errorf("relay: failed to send message: err=%v", err)
     77 	}
     78 
     79 	return value.Length(), nil
     80 }
