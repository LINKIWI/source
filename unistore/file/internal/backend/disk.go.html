    105 // GetObject opens a file handle for the requested file followed by reading its contents in full.
    106 func (d *Disk) GetObject(ctx context.Context, request *service.GetObjectRequest) (*service.GetObjectResponse, error) {
    107 	target, err := newFileTarget(d.client, request.Resource.Bucket, request.Key)
    108 	if err != nil {
    109 		return nil, status.Error(codes.PermissionDenied, err.Error())
    110 	}
    111 
    112 	file, stat, err := d.client.ReadFile(target.resolve())
    113 	if err != nil {
    114 		return nil, d.createError(err)
    115 	}
    116 
    117 	defer file.Close()
    118 
    119 	if !stat.Mode.IsRegular() {
    120 		return nil, status.Errorf(
    121 			codes.InvalidArgument,
    122 			"backend: cannot serve non-regular file: file=%s mode=%v",
    123 			stat.Path,
    124 			stat.Mode,
    125 		)
    126 	}
    127 
    128 	n := stat.Size
    129 
    130 	if request.Range != nil {
    131 		if request.Range.Unit != "bytes" {
    132 			return nil, status.Errorf(
    133 				codes.InvalidArgument,
    134 				"backend: unsupported range unit: unit=%s",
    135 				request.Range.Unit,
    136 			)
    137 		} else if request.Range.End > 0 && request.Range.End <= request.Range.Start {
    138 			return nil, status.Errorf(
    139 				codes.OutOfRange,
    140 				"backend: invalid range specification: start=%d end=%d",
    141 				request.Range.Start,
    142 				request.Range.End,
    143 			)
    144 		} else if request.Range.Start >= stat.Size {
    145 			return nil, status.Errorf(
    146 				codes.OutOfRange,
    147 				"backend: range start index is beyond file boundary: start=%d size=%d",
    148 				request.Range.Start,
    149 				stat.Size,
    150 			)
    151 		}
    152 
    153 		if _, err := file.Seek(int64(request.Range.Start), io.SeekStart); err != nil {
    154 			return nil, d.createError(err)
    155 		}
    156 
    157 		// Read only the number of bytes requested by the range specification, up to the
    158 		// end of the file. A zero end index implicitly reads to the end of the file.
    159 		end := request.Range.End
    160 		if end == 0 || end > stat.Size {
    161 			end = stat.Size
    162 		}
    163 		n = end - request.Range.Start
    164 	}
    165 
    166 	data := make([]byte, n)
    167 	if _, err := io.ReadFull(file, data); err != nil {
    168 		return nil, d.createError(err)
    169 	}
    170 
    171 	attributes, err := schemas.UnmarshalAttributes(
    172 		stat.Attributes,
    173 		&schemas.AttributesEncodingOptions{Prefix: "user"},
    174 	)
    175 	if err != nil {
    176 		return nil, d.createError(err)
    177 	}
    178 
    179 	return &service.GetObjectResponse{
    180 		Data: data,
    181 		Metadata: &common.Metadata{
    182 			Id:           stat.BaseName,
    183 			Key:          target.key,
    184 			Size:         stat.Size,
    185 			ObjectType:   common.Object_REGULAR,
    186 			ModifiedTime: timestamppb.New(stat.ModifiedTime),
    187 			Attributes:   attributes,
    188 		},
    189 	}, nil
    190 }

    192 // StreamGetObject reads the requested file in chunks, each of which is packaged as a single
    193 // response instance over the return channel.
    194 func (d *Disk) StreamGetObject(ctx context.Context, request *service.GetObjectStreamRequest) (<-chan *service.GetObjectStreamResponse, chan error) {
    195 	stream := make(chan *service.GetObjectStreamResponse)
    196 	errs := make(chan error, 1)
    197 
    198 	if request.ChunkSize == 0 {
    199 		errs <- status.Error(
    200 			codes.InvalidArgument,
    201 			"backend: disk streaming get chunk size must be greater than zero",
    202 		)
    203 		return nil, errs
    204 	}
    205 
    206 	target, err := newFileTarget(d.client, request.Request.Resource.Bucket, request.Request.Key)
    207 	if err != nil {
    208 		errs <- status.Error(codes.PermissionDenied, err.Error())
    209 		return nil, errs
    210 	}
    211 
    212 	file, stat, err := d.client.ReadFile(target.resolve())
    213 	if err != nil {
    214 		errs <- d.createError(err)
    215 		return nil, errs
    216 	}
    217 
    218 	if !stat.Mode.IsRegular() {
    219 		file.Close()
    220 		errs <- status.Errorf(
    221 			codes.InvalidArgument,
    222 			"backend: cannot serve non-regular file: file=%s mode=%v",
    223 			stat.Path,
    224 			stat.Mode,
    225 		)
    226 		return nil, errs
    227 	}
    228 
    229 	attributes, err := schemas.UnmarshalAttributes(
    230 		stat.Attributes,
    231 		&schemas.AttributesEncodingOptions{Prefix: "user"},
    232 	)
    233 	if err != nil {
    234 		file.Close()
    235 		errs <- d.createError(err)
    236 		return nil, errs
    237 	}
    238 
    239 	if request.Request.Range != nil {
    240 		if request.Request.Range.Unit != "bytes" {
    241 			file.Close()
    242 			errs <- status.Errorf(
    243 				codes.InvalidArgument,
    244 				"backend: unsupported range unit: unit=%s",
    245 				request.Request.Range.Unit,
    246 			)
    247 			return nil, errs
    248 		} else if request.Request.Range.End > 0 && request.Request.Range.End <= request.Request.Range.Start {
    249 			file.Close()
    250 			errs <- status.Errorf(
    251 				codes.OutOfRange,
    252 				"backend: invalid range specification: start=%d end=%d",
    253 				request.Request.Range.Start,
    254 				request.Request.Range.End,
    255 			)
    256 			return nil, errs
    257 		} else if request.Request.Range.Start >= stat.Size {
    258 			file.Close()
    259 			errs <- status.Errorf(
    260 				codes.OutOfRange,
    261 				"backend: range start index is beyond file boundary: start=%d size=%d",
    262 				request.Request.Range.Start,
    263 				stat.Size,
    264 			)
    265 			return nil, errs
    266 		}
    267 	}
    268 
    269 	go func() {
    270 		defer file.Close()
    271 
    272 		offset := uint64(0)
    273 		end := stat.Size
    274 		total := stat.Size
    275 
    276 		if request.Request.Range != nil {
    277 			offset = request.Request.Range.Start
    278 			end = request.Request.Range.End
    279 			if end == 0 || end > stat.Size {
    280 				end = stat.Size
    281 			}
    282 			total = end - request.Request.Range.Start
    283 
    284 			if _, err := file.Seek(int64(offset), io.SeekStart); err != nil {
    285 				errs <- d.createError(err)
    286 				return
    287 			}
    288 		}
    289 
    290 		// Wrap the file with a buffered reader in order to normalize the number of actual
    291 		// underlying read(2) syscalls on the file descriptor, regardless of the chunk size
    292 		// requested by the client.
    293 		reader := bufio.NewReader(file)
    294 		metadata := &common.Metadata{
    295 			Id:           stat.BaseName,
    296 			Key:          target.key,
    297 			Size:         stat.Size,
    298 			ObjectType:   common.Object_REGULAR,
    299 			ModifiedTime: timestamppb.New(stat.ModifiedTime),
    300 			Attributes:   attributes,
    301 		}
    302 
    303 		for {
    304 			n := request.ChunkSize
    305 			if n > end-offset {
    306 				n = end - offset
    307 			}
    308 
    309 			chunk := make([]byte, n)
    310 
    311 			if _, err := io.ReadFull(reader, chunk); err != nil {
    312 				errs <- d.createError(err)
    313 				break
    314 			}
    315 
    316 			response := &service.GetObjectStreamResponse{
    317 				Response: &service.GetObjectResponse{
    318 					Metadata: metadata,
    319 					Data:     chunk,
    320 				},
    321 				Range: &common.Range{
    322 					Unit:  "bytes",
    323 					Start: offset,
    324 					End:   offset + n,
    325 					Total: total,
    326 				},
    327 			}
    328 
    329 			stream <- response
    330 			offset += n
    331 
    332 			if offset >= end {
    333 				break
    334 			}
    335 		}
    336 
    337 		close(stream)
    338 	}()
    339 
    340 	return stream, errs
    341 }

    373 	}
    374 
    375 	if err := d.client.WriteAttributes(target.resolve(), attributes); err != nil {
    376 		return nil, d.createError(err)

    399 	var err error
    400 	var target *fileTarget
    401 	var file disk.WriteSeekCloser
    402 	var request *service.PutObjectStreamRequest
    403 
    404 	for request = range stream {
    405 		// Initialize the file descriptor for only the first request in the channel
    406 		if file == nil {
    407 			target, err = newFileTarget(d.client, request.Request.Resource.Bucket, request.Request.Key)
    408 			if err != nil {
    409 				return nil, status.Error(codes.PermissionDenied, err.Error())
    410 			}
    411 
    412 			if err := d.client.CreateDirectory(target.parent()); err != nil {
    413 				return nil, d.createError(err)
    414 			}
    415 
    416 			file, err = d.client.WriteFile(target.resolve())
    417 			if err != nil {
    418 				return nil, d.createError(err)
    419 			}
    420 
    421 			defer file.Close()
    422 		}
    423 
    424 		if request.Range != nil && request.Range.Start > 0 && request.Range.End > request.Range.Start {
    425 			if uint64(len(request.Request.Data)) != request.Range.End-request.Range.Start {
    426 				return nil, status.Error(
    427 					codes.InvalidArgument,
    428 					"backend: write payload size is inconsistent with range specification",
    429 				)
    430 			}
    431 
    432 			if _, err := file.Seek(int64(request.Range.Start), io.SeekStart); err != nil {
    433 				return nil, status.Error(codes.OutOfRange, err.Error())
    434 			}
    435 		}
    436 
    437 		if _, err := file.Write(request.Request.Data); err != nil {
    438 			return nil, d.createError(err)
    439 		}
    440 	}
    441 
    442 	// Use attributes from the latest (most recent) request from the stream.
    443 	attributes, err := schemas.MarshalAttributes(
    444 		request.Request.Attributes,
    445 		&schemas.AttributesEncodingOptions{Prefix: "user"},
    446 	)
    447 	if err != nil {
    448 		return nil, d.createError(err)
    449 	}
    450 
    451 	if err := d.client.WriteAttributes(target.resolve(), attributes); err != nil {
    452 		return nil, d.createError(err)
    453 	}
    454 
    455 	stat, err := d.client.StatFile(target.resolve())
    456 	if err != nil {
    457 		return nil, d.createError(err)
    458 	}
    459 
    460 	return &service.PutObjectStreamResponse{
    461 		Response: &service.PutObjectResponse{
    462 			Metadata: &common.Metadata{
    463 				Id:           stat.BaseName,
    464 				Key:          target.key,
    465 				Size:         stat.Size,

    607 // createError creates a semantically accurate gRPC status for errored filesystem operations.
    608 func (d *Disk) createError(err error) error {
    609 	switch {
    610 	case os.IsNotExist(err):
    611 		return status.Error(codes.NotFound, err.Error())
    612 	case os.IsPermission(err):
    613 		return status.Error(codes.PermissionDenied, err.Error())
    614 	case os.IsTimeout(err):
    615 		return status.Error(codes.DeadlineExceeded, err.Error())
    616 	case errors.Is(err, os.ErrInvalid):
    617 		return status.Error(codes.InvalidArgument, err.Error())
    618 	case errors.Is(err, syscall.ENOTEMPTY):
    619 		return status.Error(codes.FailedPrecondition, err.Error())
    620 	default:
    621 		return status.Error(codes.Unknown, err.Error())
    622 	}
    623 }

    633 // newFileTarget creates a new file target while validating the desired path.
    634 func newFileTarget(client *disk.Client, bucket string, key string) (*fileTarget, error) {
    635 	zap.L().Debug(
    636 		"creating disk file target",
    637 		zap.String("bucket", bucket),
    638 		zap.String("key", key),
    639 	)
    640 
    641 	if strings.Contains(bucket, string(os.PathSeparator)) {
    642 		return nil, fmt.Errorf("bucket name contains illegal characters: bucket=%s", bucket)
    643 	}
    644 
    645 	canonicalKey := filepath.Clean(key)
    646 	resolved := filepath.Join(bucket, canonicalKey)
    647 
    648 	// Require the resolved path to start with the bucket after a pass through filepath.Clean.
    649 	// Since at this point it is known that the bucket does not contain nested directories, this
    650 	// thwarts potential directory traversal attacks by enforcing that the requested key does
    651 	// not resolve to be outside the allowed file tree.
    652 	if canonicalKey != "." && !strings.HasPrefix(resolved, fmt.Sprintf("%s%c", bucket, os.PathSeparator)) {
    653 		return nil, fmt.Errorf("resolved file path is invalid: key=%s", key)
    654 	}
    655 
    656 	return &fileTarget{
    657 		client: client,
    658 		bucket: bucket,
    659 		key:    canonicalKey,
    660 	}, nil
    661 }
    662 
    663 // resolve resolves the file path targeted by the supplied bucket and key.
    664 func (f *fileTarget) resolve() string {
    665 	return filepath.Join(f.bucket, f.key)
    666 }
    667 
    668 // parent returns the parent directory of the target. If the target is itself a directory, the
    669 // return value is the same as that of resolve. If the target is a file or file prefix, the nearest
    670 // path ancestor is returned.
    671 func (f *fileTarget) parent() string {
    672 	resolved := f.resolve()
    673 
    674 	// The requested path may itself already be a directory, in which case the parsed dirname
    675 	// should be discarded.
    676 	p := filepath.Dir(resolved)
    677 	stat, err := f.client.StatFile(resolved)
    678 	if p == "." || (err == nil && stat.Mode.IsDir()) {
    679 		p = resolved
    680 	}
    681 
    682 	return p
    683 }
